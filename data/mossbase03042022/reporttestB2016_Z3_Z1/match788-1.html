function<double(double)> BaricentricnaInterpolacija(vector<pair<double,double>> v,int d)

        for(int j=max(1,i-d);j<=min(i,int(v.size())-d);j++)

                p*=(1/(v[i-1].first-v[k-1].first));
            }
            p*=pow(-1,j-1);
            suma+=p;
        }
        w[i-1]=suma;
    }
    return [v,w] (double x)
    {
       for(int i=0;i<v.size();i++)
       {
           if(x==v[i].first)
           return v[i].second;
       }
       double suma1(0),suma2(0);

    };
}
