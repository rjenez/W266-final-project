double w (int brojcvorova, int d, int i, vector<pair<double, double>> grupa)
{
    int n1;
    double rezd1, rezd2, rez(0);
    
    n1=minimum(i,brojcvorova-d);
    
    for (int k=maksimum(1,i-d); k<=n1; k++)
    {
        rezd1=pow((-1),(k-1));
        rezd2=1;
        for (int j=k; j<=k+d; j++)
        {
            if (j!=i)
            {


                rezd2=rezd2*(1./(grupa[i-1].first-grupa[j-1].first));
            }
        }
        
        rez=rez+rezd1*rezd2;
    }
    
    return rez;
}
