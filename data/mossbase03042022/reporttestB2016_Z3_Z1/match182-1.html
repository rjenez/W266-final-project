std::function<double(double)>BaricentricnaInterpolacija(std::vector<double>v, int d){
   int n=v.size();
    std::vector<double>w;
    for(int i=0; i<n; i++) {
        int min,max;
        if(i<n-d)  min=i;
        else min=n-d;
        if(i>n-d) max=i;
        else max=n-d;
    
        for(int k=max;k!=min;k++) {
            double proizvod(1);
            for(int j=k; j!=(k+d); j++) {
                double clan=1/(v[i]-v[j]);
                proizvod*=clan;
            }
           w.push_back(proizvod); 
        }
        
    }
    return [v, w, d](double x) {
        double suma1(0),suma2(0);
        int n=v.size();
        for(int i=0; i<n; i++) {
         double clan=(w[i])/(x-v[i].first))*(v[i].second);
        suma1+=clan;
        }
        for(int i=0; i<n; i++) {
            double clan=w[i]/(x-v[i].first);
            suma2+=clan;
        }
        return suma1/suma2;
    };
}
