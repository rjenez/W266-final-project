/*Provjeravamo da li se rijec na koju smo naisli nalazi u nizu rijeci*/
int provjera_rijeci(char* s, char *rijeci){
    int vrati=1;
    while(*s!='\0' && *s!=32 && *rijeci!='\0'){
        if(*s!=*rijeci)
            vrati= 0;
        rijeci++;
        s++;
        if(*rijeci=='\0' || *s=='\0')
			break;
    }
 	if(*s!=32 && *s!='\0' && *rijeci=='\0')
 		vrati=0;
 	if(*rijeci!='\0')
 		vrati=0;
    return vrati;
}
int duzina_rijeci(char* s){
    int brojac=0;
    while((*s!='\0') && ((*s>='A' && *s<='Z') || (*s>='a' && *s<='z'))){
        brojac++;
        s++;
    }
    return brojac;
}
