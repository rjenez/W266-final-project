#include <stdio.h>

int uporedi(char *prva, char *druga, int duzina){
	char *p=prva, *d=druga;
	int brojac=0;
	while(*d!='\0'){
		if(*p!=*d)
			return 0;
		p++;
		d++;
		brojac++;
	}
	if(*d=='\0'&&brojac!=duzina)
			return 0;
	return 1;
}

		char *pok=prva;
		while(pok!=kraj-duzina){
			*pok=*(pok+duzina);
			pok++;
			
		}
		*pok='\0';
		return tekst;
	}
	
	if(brojac!=duzina){
		if(brojac==1){
			while(*kraj!='\0') kraj++;
			char *pok=prva+1;
			while(pok!=kraj-duzina+1){
				*pok=*(pok+duzina-1);
				pok++;
			}
			*pok='\0';
			*prva=*druga;
			return tekst;
		}
		while(*kraj!='\0') kraj++;
		if(brojac>duzina){
			while(kraj!=p){
				*(kraj+brojac-duzina)=*kraj;
				kraj--;
			}
		}
		if(brojac<duzina){
			char *oduzima=p+duzina-brojac-1;
			while(oduzima!=kraj-(duzina-brojac)){
				*oduzima=*(oduzima+duzina-brojac);
				oduzima++;
			}
			*oduzima='\0';
			
		}
		
	}
	
	
	for(i=0;i<brojac;i++)
		*(p+i)=*(d+i);
	return tekst;
	
}
