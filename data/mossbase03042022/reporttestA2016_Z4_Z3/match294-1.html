#include <stdio.h>
#include <string.h>


int ProvjeriRijecnik(char **rijeci,char** zamjene,int broj_rijeci)
{
	int jeste=1;
	int i;
	char *temp,*temp1;
	
		for(i=0;i<broj_rijeci;i++)
		{   
			temp=rijeci[i];
			temp1=zamjene[i];
			while(*temp!='\0' || *temp1!='\0')
			{
				if(*temp!=*temp1){jeste=0;break;}
				temp++;
				temp1++;
			}
			if(jeste==0) break;
		}
	
		return jeste;
}
