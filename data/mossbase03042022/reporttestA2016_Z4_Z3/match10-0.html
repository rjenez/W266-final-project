int uporedi(char *rijec1, char *rijec2, int duzina){
	char *m=rijec1, *n=rijec2;
	int brojac=0;
	while(*n!='\0'){
		if(*m!=*n)
			return 0;
		m++;
		n++;
		brojac++;
	}
	if(*n=='\0'&&brojac!=duzina)		
			return 0;
	return 1;
}
