{

    char* p = s1, *q = s2;

    int x = 1;
    while ((*p != ' ' && *p != '\0') && (*q != ' ' && *q != '\0') ) {
        if (*p != *q) {
            x = 0;
            break;
        }
        p++;
        q++;
    }
    if (x) {

        if ((*p != ' ' && *p != '\0') || (*q != ' ' && *q != '\0'))
            x = 0;
    }
    return x;
}
