#include <stdio.h>

int provjera (char* text, char* zamjena)
{
	int je_li_isto=1;
	char* reset=text;
	char* prvoslovo=zamjena;
	while (*zamjena!='\0' && *text!='\0') {
		if(*text!=*zamjena)
			je_li_isto=0;
		zamjena++;
		text++;
	}
	if(*zamjena!='\0' || (*text!='\0' && *text!=' ')) je_li_isto=0;
	zamjena=prvoslovo;
	text=reset;
	return je_li_isto;
}

			if(provjera(tekst,rijeci[i])) {
