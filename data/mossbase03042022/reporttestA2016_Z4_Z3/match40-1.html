#include <stdio.h>
#include <ctype.h>




int JesuIsti(char *p,char *temp)
{
	int	isti=1;

		while(*temp!='\0')
		{
			if(*temp++!=*p++){isti=0;break;}
			if(*temp=='\0' && (*p!=' ' && *p!='\0')){isti=0;break;}
		}
	return isti;
		
}

void ZamijeniRijeci(char *recenica,char *rijec)
{
	while(*rijec!='\0')
	{
		*recenica++=*rijec++;
	}
	recenica++;
}

int Rjecnik(char **rijeci,char **zamjene,int broj_rijeci)
{
	int i,isti=1;
	char *pom,*pom1;
	
		for(i=0;i<broj_rijeci;i++)
		{
			pom=rijeci[i];
			pom1=zamjene[i];
			while(*pom!='\0' || *pom1!='\0')
			{
				if(*pom++!=*pom1++)isti=0;
			}
			if (isti==0) break;
		}
		return isti;
}
