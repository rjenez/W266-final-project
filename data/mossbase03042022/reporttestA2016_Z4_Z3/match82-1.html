#include <stdio.h>



int provjera(char* p,char* q){
    int jesu=1;
    while(*q!='\0'){
        if(*q++!=*p++){
            jesu=0;
            break;
        }
        if(*q=='\0' && (*p!=' ' && *p!='\0')){
            jesu=0;
            break;
        }
    }
    return jesu;
}

void zamjena(char* rec,char* rijec){
    while(*rijec!='\0'){
        *rec++=*rijec++;
    }
    rec++;
}

}
