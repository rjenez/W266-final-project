#include <stdio.h>
void unesi(char* s, int velicina){
	char znak = getchar();
	int i;
	if(znak == '\n') znak=getchar();
	i=0;
	while(i<velicina && znak != '\n'){
		*s = znak;
		znak = getchar();
		i++;
		s++;
	}
	*s = '\0';
}


int rijec(char* s){
	if((*s != ' ' && *s != '\0')){
		return 1;
	}else{
		return 0;
	}
}
