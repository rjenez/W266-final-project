char* trazirijec(char* s1, char* s2) {
	char* poc=s1;
	char* p;
	char* q;
	
	while(*s1!='\0') {
		p=s1;
		q=s2;
		
		while(*p++==*q++) 
		if(*q=='\0' && (*p==' ' || *p=='\0') && (s1==poc || *(s1-1)==' ' ))
		return s1;
		
		s1++;
	}
	return NULL;
	
	return s1;
}
