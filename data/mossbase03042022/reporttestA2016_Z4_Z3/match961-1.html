#include <stdio.h>

int je_li_ista_rijec(char *s1, char *s2) {
	char *p=s1;
	char *q=s2;
	
	while (*p == *q) {
		
		if ((*p == ' ' || *p == '\0') && (*q == ' ' || *q == '\0'))
			return 1;
		p++;
		q++;
	}
	if ((*p == ' ' || *p == '\0') && (*q == ' ' || *q == '\0'))
			return 1;
	
	return 0;
}

void izbaci_jednu_rijec(char* s) {

	while(*s != ' ' && *s != '\0')
		s++;
		


	while (*s != '\0') {
		*brisac++=*s++;
	}
	*brisac='\0';
}

		pom++;
	
	while(pom != s){
		
		*(pom + duzina) = *pom;
		pom--;
	}
	*(pom + duzina) = *pom;
	
	for(i = 0; i < duzina; i++){

	}
}
