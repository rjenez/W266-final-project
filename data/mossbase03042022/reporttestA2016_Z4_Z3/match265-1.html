#include <stdio.h>
#include <ctype.h>

int da_li_su_iste(char *p,char *pom)
{
	int	iste=1;

		while(*pom!='\0')
		{
			if(*pom++!=*p++) {
				iste=0;
				break;
			}
			if(*pom=='\0' && (*p!=' ' && *p!='\0')) {
				iste=0;
				break;
			}
		}
	return iste;
		
}

void zamjeni_rijeci(char *rec, char *rijec)
{

}
