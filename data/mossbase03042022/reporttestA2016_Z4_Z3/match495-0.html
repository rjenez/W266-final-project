int DuzinaRj(char* s)
{
    int duz = 0;

    while (*s != ' ' && *s != '\0') {
        duz ++;
        s++;
    }
    return duz;
}

{

    int x = 1;
    while ((*p != ' ' && *p != '\0') && (*q != ' ' && *q != '\0') ) {
        if (*p != *q) {
            x = 0;
            break;
        }
        p++;
        q++;
    }
    if (x) {

        if ((*p != ' ' && *p != '\0') || (*q != ' ' && *q != '\0'))
            x = 0;
    }
    return x;
}
