#include <stdio.h>

int duzina(char *p)
{
	int brojac=0;
	while(*p!='\0')
	{
		brojac++;


		p++;
	}
	
	return brojac;
}

char* zamjena_rijeci(char* tekst, char** rijeci, char** zamjene, int broj_rijeci)
{
	char *p1=tekst;
	char *p2;
	int i,iste,razlika,j,duzinaRijeci,brojac;

			p1++;

		p2=p1;

				pomocni2=p1;

				{
					if(*pomocni1!=*pomocni2)

						iste=0;
