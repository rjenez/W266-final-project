void KorigujString(std::string &s){
    auto pok(s.begin());
    while(pok != s.end()){
        *pok=tolower(*pok);
        if(!((*pok >= 'a' && *pok <='z') || (*pok >='0' && *pok <='9') || *pok == ' ')) *pok = ' ';
        pok++;
    }
}

int Brojac(std::string s){
    int brojac(0);
    KorigujString(s);
    auto pok(s.begin());
    while(pok != s.end()){
        if(*pok == ' '){
            pok++;
            continue;
        }
        std::string pomocni;
        while((*pok >= 'a' && *pok <= 'z') || (*pok >= '0' && *pok <='9')){
            pomocni += *pok;
            pok++;
        }
        if(pomocni.length() != 0) brojac++;
    }
    return brojac;
}
