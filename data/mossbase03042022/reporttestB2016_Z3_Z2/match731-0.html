void  PrebaciVelikaSlovaUMala(std::string &s){
    
    for(int i = 0; i < s.length(); i++){
        if(s[i]>='A' && s[i]<='Z') s[i] = tolower(s[i]);
    }
}
std::set<int> PozicijeNaKojimSePojavljuje(std::string rijec, std::string recenica, int pocni_od){

}
