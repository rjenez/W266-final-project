bool dalijebroj (char a){

    if(a>='0' && a<='9') return true;

}

bool dalijeslovo (char a){
    if((a>='A' && a<='Z') || (a>='a' && a<='z')) return true;
    return false;
}



std::string pretvoriumala(std::string s){
    for(int i(0);i<s.length();i++){
        if(s[i]>='A' && s[i]<='Z') s[i]+='a'-'A';
    }
    return s;
}
