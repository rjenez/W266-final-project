void IspisiIndeksPojmova(map<string,set<int>> mapa){
        for(auto it=mapa.begin(); it!=mapa.end(); it++){
            auto skup(it->second);
            int brojac(0);
            cout << it->first << ": ";
            for (auto x : skup) {
                brojac++;
                if (brojac!=skup.size()) cout << x << ",";
                else cout << x;
                
        } 
        cout << endl; }  
}

}
