bool UporediMe(std::string s1,std::string s2) {
    for(int i=0;i<s1.length();i++) {
        s1[i]=std::toupper(s1[i]);
    }
    for(int i=0;i<s2.length();i++) {
        s2[i]=std::toupper(s2[i]);
    }
    return s1==s2;
}

        s2[i]=std::tolower(s2[i]);

}
