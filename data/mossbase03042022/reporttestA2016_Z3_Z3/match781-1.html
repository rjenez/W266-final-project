#include <stdio.h>

int main() {
	int a[200][200]={0};
	int m,n,p,br=0,treba,k,j,i,e,s,treba1;

			printf("Brojevi nisu u trazenom opsegu.\n");

			scanf("%d",&a[i][j]);
		}
	}
	for(i=0;i<m-1;i++){
		for(p=i+1;p<m;p++){
			br=0;

			}
				if(br==n) {


					for(s=p;s<m-1;s++){
						for(k=0;k<n;k++){
							treba=a[s][k];
							a[s][k]=a[s+1][k];
							a[s+1][k]=treba;
						}
					}
				m--;
				p--;
				} 
				
		}
		
	}
