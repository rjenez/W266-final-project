		for(k=i+1;k<m;k++){
		for(j=0;j<n;j++)if(matrica[i][j]!=matrica[k][j])break;

		          	matrica[p][j]=matrica[p+1][j];
			}
				}
	        	m--;
				k--;		
		} 
		
		}
		
	}
	for(i=0;i<m;i++){
	for(j=0;j<n;j++){
		kopija[j][i]=matrica[i][j];
	}
	}

		for(i=0;i<m;i++){

				}
