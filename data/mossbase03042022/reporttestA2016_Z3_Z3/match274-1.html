int izbacivanje(int mat[200][200],int m ,int n){
	int i,j,k,l,t,s,radi=1,temp;
	for(k=0;k<n;k++){
		for(j=k+1;j<n;j++){


			if(mat[0][j]== mat[0][k]){
				radi=1;
			for(l=0;l<m;l++)	
				if(mat[l][j]!=mat[l][k])radi=0;
			if(radi==1){
				for(s=0;s<m;s++)
					for(t=j;t<n;t++)
						mat[s][t]=mat[s][t+1];
				n--;
				j--;
				}
			}		
		}
	}return n;
}

void rotacija(int niz[200][200],int m){

			temp=niz[i][j];
			niz[i][j]=niz[j][i];
			niz[j][i]= temp;
		}
	}}
}
