#include <stdio.h>



void f1(int mat[200][200],int *niz,int velicina,int m){
	int i;
	for(i=0;i<velicina;i++){
		niz[i]=mat[m][i];
	}
}

int f2(int*n1,int*n2,int velicina){
	int i;
	int jednak=1;


	for(i=0;i<velicina;i++){
		if(n1[i]!=n2[i]){
			jednak=0;break;
		}
	}
	return jednak;
}

void f3(int mat[200][200],int brr,int brk,int m){
	int i,j;
	for(i=m;i<brr-1;i++){
		for(j=0;j<brk;j++){
			mat[i][j]=mat[i+1][j];
		}
	}
}

void f5(int mat[200][200],int brr,int brk,int n){

	for(i=0;i<brr;i++){

			mat[i][j]=mat[i][j+1];
		}
	}
}
