#include <stdio.h>

int duplikat (int* niz,int k, int* niz1, int vel) {
    int j;
    for(j=k+1;j<vel;j++) {
        if(niz[k]==niz1[j]) return j;
    }
    return 0;
}

void sortiraj (int* niz,int vel) {
    int* p=niz;
    int* q=niz;
    int temp;
    while(p<niz+vel-1) {
        q=p+1;
        while(q<niz+vel) {
            if(*p>*q) {
                temp=*p;
                *p=*q;
                *q=temp;
            }
            q++;    
        }
        p++;
    }
}

int izbaci_visak (int* niz, int  vel) {
    int*p=niz;
    int temp,i,j;
    for(i=0;i<vel;i++) {
        for(j=i+1;j<vel;j++) {
            if(*(p+i)==*(p+j)) {
                temp=*(p+vel-1);
                *(p+vel-1)=*(p+j);
                *(p+j)=temp;
                vel--;
                j--;
            }
        }
    }
    return vel;
}

    int i;
    for(i=0;i<vel;i++) {
        if(niz[i]==broj) return 1;
    }
    return 0;
}
