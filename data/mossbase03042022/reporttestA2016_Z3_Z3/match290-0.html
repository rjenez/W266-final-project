#include <stdio.h>


int IzbaciRed(int mat[][200],int m,int n)
{   
	int i,j,k,p,q;


	int pret;
	
	for(i=0;i<m;i++)
	{
		for(j=i+1;j<m;j++)
		{
			pret=1;
			for(k=0;k<n;k++)
			{
			   if(mat[i][k]!=mat[j][k]) pret=0;
			}
			if(pret==1)
			{
				for(p=j;p<m-1;p++)


					for(q=0;q<n;q++)
						mat[p][q]=mat[p+1][q];
					m--;
					j--;
			}
		}
	}
	return m;
}


int IzbaciKolone(int mat[][200],int m,int n)
{   
	int i,j,k,p,q;


	int pret;
	
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			pret=1;
			for(k=0;k<m;k++)
			{ 
			   if(mat[k][i]!=mat[k][j]) pret=0;
			}
			if(pret==1)
			{   
				for(p=j;p<n-1;p++)
					for(q=0;q<m;q++)


						mat[q][p]=mat[q][p+1];
					n--;
					j--;
			}
		}
	}
	return n;
}
