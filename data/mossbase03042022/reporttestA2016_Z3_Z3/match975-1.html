	        scanf("%d",&mat[i][j]);
	    }

	    for(z=0;z<i;z++){

	    	for(j=0;j<b;j++){
	    		//ako u matrici nema takvih redova program se prekida
	    		if(mat[i][j] != mat[z][j]){

	    	}
	    }
	    //ako postoje takvi redovi onda izbacujemo prvi odozdo takav red
	    	if(p==0){

	    		for(j=0;j<b;j++)


	    		mat[k][j]=mat[k+1][j];
	    	}

	        }
