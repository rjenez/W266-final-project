        std::pair<int, int> pom=pomVrijeme.Ocitaj();
        sat2=pom.first;
        min2=pom.second;
        
        if(dan1==dan2 && mjesec1==mjesec2 && godina1==godina2 && sat1==sat2 && min1==min2){
            delete preg[i];
            preg[i]=nullptr;
            for(int j=i; j<broj_pregleda-1; j++){
                preg[j]=preg[j+1];
            }
            
            preg[broj_pregleda-1]=nullptr;
            broj_pregleda--;
            
            return;
        }
        
    }
}
