#include <stdio.h>
int main ()
{
    int n,i,j;
    do {
        printf ("Unesite broj n: ");
        scanf("%d" ,&n);

            printf("Pogresan unos\n");
        }

    } while (n<=0 || n>50);
    if (n==1) {

    }

    for(i=0; i<n ; i++) {
        for(j=0; j<4*n-3 ; j++) {
