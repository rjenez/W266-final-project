#include <stdio.h>
int main() {


	int i,j,n;
	
	do {
	printf("Unesite broj n: ");
	scanf("%d",&n);

		
	if (n<=0||n>50)
	{
		printf("Pogresan unos");
   		printf ("\n");
	}
	else 
	{
	for (i=0; i<n; i++) {
	  for(j=0; j<(4*n-1); j++) {
	  	if (i==0 && n==1)
	  		printf("*");
	  	else
	  		if(j==i || j==2*n-i-2 || j==2*n+i-2 || j==4*n-i-4) 
	  			 printf("*");
	  		else 
	  			 printf(" ");
	  }
   		printf ("\n");
	}
	}
	} while (n<=0||n>50);
   return 0;
}
