	ponavljanje:

	{
		printf ("Pogresan unos\n");

	}
	
	if (n==1)
    {   
    	printf("***");

    }
  
    else

	{
		for (j=0;j<(4*n-3);j++)

		    printf("*");
		    else if (j-i-(4*n-3)/2==0)

		}

	}
