#include <stdio.h>
#include <math.h>

int provjera(char* s) {
	char* p;
	int br = 0;
	
	
	if(*s=='\0') return 0;
	if(*s!='(') return 0;
	
	p = s;	
	if(*p=='(' && *(p+1)=='(') return 0;
	


	while(*p!='\0') {
		if(*p!='(' && *p!=')' && *p!=',' && *p!='.' && *p!='-' && (*p<'0' || *p>'9')) return 0;
		if(*p==')' && *(p+1)!='\0' && (*(p+2)!='(' || *(p+1)!=',')) return 0;
		if(*p=='(' && *(p+1)==',') return 0;
		if(*p==',' && *(p+1)==')') return 0;
		if(*p=='.' && (*(p-1)<'0' || *(p-1)>'9')) return 0;
		if(*p=='-' && *(p+1)==',') return 0;
		if(*p=='-' && *(p+1)==')') return 0;
		p++;
		
	}
	p--;
	if(*p!=')') return 0;
	
	p = s;
	
	while(*p!='\0') {
		br = 0;
		while(*p!=')') {
			if(*p==',') br++;	
			p++;
		}
		if(br!=1) return 0;
		p++;
		
		if(*p==',') p++;
	}
	return 1;
}
