int PorediNiz(double niz1[], int vel1, double niz2[], int vel2){
	int i, j;
	for(i=0; i<vel1; i++){
		for(j=i+1; j<vel2; j++)
			if(fabs(niz1[i]-niz2[j])<EPS) return 1; 
	} 
	return 0; /*  - ne postoji isti element u nizu1 i nizu2 */
}
