bool slovo(char a)
{
    return (toupper(a)>='A'&&toupper(a)<='Z' || a>='0'&&a<='9');
}

int slova(string s)
{
    int br=0;
    for(int i=0; i<s.size(); i++)
        if(slovo(s[i])) br++;
    return br;
}

struct Dijete
{
    string ime;
    Dijete *slj;
};

Dijete *transformisi(vector<string> v)
{
    Dijete *p=nullptr;


    Dijete *t=nullptr;
    for(int i=0; i<v.size(); i++)
    {
        Dijete *n= new Dijete;
        n->ime=v[i];
        if(i==0) { p=n; t=n;}
        else { t->slj=n; t=n;}
        if(i==v.size()-1) n->slj=p;
    }
    return p;
}
