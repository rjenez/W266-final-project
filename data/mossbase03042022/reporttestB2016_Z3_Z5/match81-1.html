}

int duzinaStringa(const string &s)
{
    int duzina{0};
    for(char c : s)
    {
        if((c>='A' && c<='Z') || (c>='a' && c<='z') || (c>='0' && c<='9'))
            duzina++;
    }
    return duzina;
}
