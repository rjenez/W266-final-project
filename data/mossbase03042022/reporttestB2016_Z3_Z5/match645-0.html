bool ValidnoSlovo(char c) {
  return ((c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z') || (c >= '0' && c <= '9'));
}
int Duzina(std::string x) {
  int ret = 0;
  for (int i = 0; i < x.length(); i++)
    if (ValidnoSlovo(x[i])) ret++;
  return ret;
}

}
