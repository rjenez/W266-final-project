struct Dijete

int BrojSlova(std::string s)
{
    int brojac(0);
    for(int i(0); i < s.size(); i++)
    {
        if(s[i] >= '0' && s[i] <= '9')
            brojac++;
        else if(s[i] >= 'a' && s[i] <= 'z')
            brojac++;
        else if(s[i] >= 'A' && s[i] <= 'Z')
            brojac++;
    }
    return brojac;
}
