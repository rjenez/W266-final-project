vector<int> PobrojiTimove(int n,int k)
{
    vector<int> v(k);
    for(int i=0;i<v.size();i++)
    {
        if(i<n%k)
            v[i]=(n/k)+1;
        else v[i]=n/k;
        
    }
    return v;
}
