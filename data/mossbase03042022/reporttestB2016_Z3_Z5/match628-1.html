std::vector<int>Djeca(std::vector<std::string>s, int n){
    int m=s.size()/n;
    std::vector<int>ret(n);
    int p=s.size();
    for(int i=0; i<n; i++){
        ret[i]=m;
        if(p%n!=0){
            ret[i]=m+1;
            p--;
        }
    }
    return ret;
}
