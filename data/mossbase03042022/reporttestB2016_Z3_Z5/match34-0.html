struct Dijete {

Dijete *napraviListu(std::vector<std::string> &vektor){
    Dijete *prvi = nullptr;
    Dijete *trenutni ;
    
    for(int i(0); i < vektor.size(); i++){
        
        Dijete *tmp(new Dijete);
        tmp->ime = vektor[i];
        tmp->sljedeci = nullptr;
        
        if(prvi == nullptr) prvi = tmp;
        else trenutni->sljedeci = tmp;
        trenutni = tmp;
        
    }
    trenutni->sljedeci = prvi;
    
    return prvi;
}

}
