int BrojSlova(std::string s){
    int i(0), brojac(0);
    while(i!=s.length()){
        if((s[i]>='A' && s[i]<='Z') || (s[i]>='a' && s[i]<='z') || (s[i]>='0' && s[i]<='9'))
            brojac++;
        i++;
    }
    return brojac;
}
