        zaduzenje=std::make_shared<Korisnik>(korisnik);

    void RazduziKnjigu() {
        zaduzenje=nullptr;
    }
    bool DaLiJeZaduzena()const {
        if(!zaduzenje) return false;
        return true;
    }

};
