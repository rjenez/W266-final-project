      pok=&korisnik;

      Knjiga::pok=nullptr;

        if(mapa_korisnika.count(evbroj)==1) throw std::logic_error("Korisnik vec postoji");

       std::map<int,std::shared_ptr<Korisnik>>::iterator i=mapa_korisnika.find(clbroj);
