#define EPSILON 0.0001

	if(fabs(a2 - a1) > EPSILON)
	{
		x = (b2 - b1) / (a1 - a2);
		y = a1 * x + b1;
		printf("Prave se sijeku u tacci (%.1f,%.1f)", x, y);

	else
	{

	}
	return 0;
}
