double prosjek_ocjena(struct Student student) {

	for (i = 0; i < student.br_ocjena; i++) {
		
		if (student.ocjene[i] == 5)

	    suma = suma + student.ocjene[i];
	
	}
	
	if (student.br_ocjena == 0)  
	    return 5.0;
	
	return suma / student.br_ocjena;
	
}
