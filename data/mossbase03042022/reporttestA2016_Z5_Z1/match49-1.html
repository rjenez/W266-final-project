float racunanje_prosjeka (int *ocjena, int broj_ocjena) {

	return prosjek;
}

		if (*prez1<*prez2) return -1;
		if (*prez1>*prez2) return 1;
		prez1++;
		prez2++;
	}
	if (*prez1!='\0') return 1;
	if (*prez2!='\0') return -1;
	while (*ime1!='\0' && *ime2!='\0') {
		if(*ime1<*ime2) return -1;

		ime1++;
		ime2++;
	}
	if (*ime1!='\0') return 1;
	if (*ime2!='\0') return -1;
	return 0;
}
