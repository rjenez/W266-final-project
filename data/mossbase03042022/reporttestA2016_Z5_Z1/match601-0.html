double prosjek_ocjena (int *ocjene, int vel) {
	float suma=0;
	int i;
	if (vel==0)
		return 5;
	for (i=0; i<vel; i++) {
		if (ocjene[i]==5)
			return 5;
		suma+=ocjene[i];
	}
	suma/=vel;
	return suma;
}
