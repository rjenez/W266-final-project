#include <stdio.h>

struct Student {
	char ime[20], prezime[20];
	int ocjene[50];
	int br_ocjena;
};

// funkcija racuna prosjek ocjena
double prosjek_ocjena (struct Student x)
{
	int i=0, suma=0;
	double prosjek;
	for(i=0; i<x.br_ocjena; i++) {
		suma+=x.ocjene[i];
		if(x.ocjene[i]==5 || x.br_ocjena==0)
			return 5;
	}
	if(suma==0)
		return 5;


	prosjek=(double)suma/x.br_ocjena;
	return prosjek;
}
