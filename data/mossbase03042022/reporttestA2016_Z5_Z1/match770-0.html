/* funkcija za unos stringa */
void unesi(char niz[], int velicina){
    int i=0;
    char znak;
    do{
        znak=getchar();
    }while(znak=='\n');
    
    while(znak!='\n' && i<velicina-1){
        niz[i++]=znak;
        znak=getchar();
    }
    niz[i]='\0';
}
