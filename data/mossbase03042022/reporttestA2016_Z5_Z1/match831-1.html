void unesi(char niz[], int velicina) {
    char znak = getchar();
    if (znak == '\n') znak=getchar();
    int i = 0;
    while (i < velicina-1 && znak != '\n') {
        niz[i] = znak;
        i++;
        znak = getchar();
    }
    niz[i]='\0';
}

int prebroji(char *niz){ /*funkcija za prebrojavanje slova u riječi*/
	int i=0, brojac=0;
	while(niz[i]!='\0'){
		brojac++;
		i++;
	}
	return brojac;
}
