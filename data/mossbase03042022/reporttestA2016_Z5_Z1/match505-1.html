void unesi(char *niz, int velicina) {
    char znak = getchar();
    if (znak == '\n') znak=getchar();
    int i = 0;
    while (i < velicina-1 && znak != '\n') {
        niz[i] = znak;
        i++;
        znak = getchar();
    }
    niz[i]='\0';
}

float prosjek_ocjene(struct Student s){
    int k,suma=0;
    float y;
    
    if(s.br_ocjena<1) return 0;
    
    for (k=0; k<s.br_ocjena;k++){
        if (s.ocjene[k] == 5.0 )
            return 5.0;
        suma+=s.ocjene[k];
    }
    y=(float)suma/s.br_ocjena;

    if(suma==0) 
        y=5.0;
    return y;
}
