int poredi_rijeci(char *prva,char* druga)
{
	while(*prva!='\0'&&*druga!='\0')
	{
		if(*prva>*druga) return 0;
		if(*druga>*prva) return 1;
		druga++;
		prva++;
	}
	if(*prva!='\0'&&*druga=='\0') return 0;
	if(*prva=='\0'&&*druga!='\0') return 1;
	return -1;
}
