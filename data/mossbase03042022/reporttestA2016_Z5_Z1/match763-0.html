void unesi(char *s, int velicina) {
	char znak=getchar();
	int i=0;
	if(znak=='\n') znak=getchar();
	while(i<velicina-1 && znak!='\n') {
		*s=znak;
		s++;
		znak=getchar();
	}
	*s='\0';
}

float prosjek_ocjena(int *niz, int velicina) {
	float suma=0, prosjek;
	int i;
	for(i=0; i<velicina; i++) {

			prosjek=5;
			break;
		}
		else {
			suma+=niz[i];
		}
	}
	prosjek=suma/velicina;
	return prosjek;
}
