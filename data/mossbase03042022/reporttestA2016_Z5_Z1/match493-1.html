/*Funckija racuna prosjek svakog studenta pojedinacno u zavsnosti koliko ima ocjena*/
float prosjek(int niz[],int vel){   
	int i=0,suma=0;

	for(i=0;i<vel;i++){
		if (niz[i]==5) return 5;    
		suma+=niz[i];     
	}
	return (float)suma/vel;         
}
