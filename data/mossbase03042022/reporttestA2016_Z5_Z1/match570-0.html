void unesi(char niz[], int velicina)
{
    char znak = getchar();
    if (znak == '\n') znak = getchar();
    int i = 0;
    while (i<velicina-1 && znak != '\n')
    {
        niz[i]=znak;
        i++;
        znak=getchar();
    }
    niz[i]='\0';
}

double izracunajProsjek(int ocjene[], int brOcjena)
{
	int i;
	double prosjek = 0.0;
	
	if(brOcjena == 0)
	{
		prosjek = 5.0;
		return prosjek;
	}
	
	for(i = 0; i < brOcjena; i++)
	{
		if(ocjene[i] == 5.0)
		{
			prosjek = 5.0;
			return prosjek;
		}
		prosjek += ocjene[i];
	}
	
	return prosjek / brOcjena;
}
