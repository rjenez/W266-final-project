float prosjek(int ocjene[], int vel)
{
	int i=0;
	float suma=0;
	for(i=0; i<vel; i++)
		suma+=ocjene[i];
	return suma/vel;
}

int prosjek_5(int ocjene[], int vel)
{
	int i=0;
	if(vel==0)
		return 1;
	for(i=0; i<vel; i++)
		if(ocjene[i]==5)
			return 1;
	return 0;
}

int isti_string(char *p1, char *p2)

	vel1=strlen(s1); vel2=strlen(s2);
	 
	if(vel1<vel2)
	{
		for(i=0; i<vel1; i++)
		{
			if(s1[i] == s2[i])
				continue;
			else if(s1[i] > s2[i])
				return 2;
		}
		return 1;
	}
	else if(vel2<vel1)
	{
		for(i=0; i<vel2; i++)
		{
			if(s1[i] == s2[i])
				continue;
			else if(s1[i] < s2[i])
				return 2;
		}
		return 1;
	}
	else if(vel1 == vel2)
	{
		for(i=0; i<vel1; i++)
			if(s1[i] > s2[i])
				return 2;
		return 1;
	}

}
