		s = prezime2; p = prezime1; flag = 2;
	}
	
	while (*p!='\0') {
		if (*p == *s) {
			p++; s++; continue;
		} else if (*p < *s) {
			if (flag == 1) return 0;
			if (flag == 2) return 1;

			if (flag == 2) return 0;
