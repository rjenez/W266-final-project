struct Student {
    char ime[20], prezime[20];
    int ocjene[50];
	int br_ocjena;
};

float funkcija_prosjek(struct Student s){
	float sum=0;
	int j=0;
	float p;
	if(s.br_ocjena==0) return 5;
		else {
			for(j=0;j<s.br_ocjena;j++){
				if(s.ocjene[j]==5) return 5; 
				sum=sum+s.ocjene[j];
			}
			p=sum/s.br_ocjena;
			
			
	return p;
}}
