float prosj(int *ocj,int brocj){
	int suma=0,i;
	for(i=0;i<brocj;i++){
		if(ocj[i]==5) return 5;
		suma=suma+ocj[i];
	}
	if (suma==0) return 5;
	return (float)suma/brocj;
}

		if(*p1<*p2) return 1;
		if(*p1>*p2) return 2;
		p1++;
		p2++;
	}
	if(*p1!=0) return 2;
	if(*p2!=0) return 1;
	while(*i1!=0 && *i2!=0){
		if(*i1<*i2) return 1;
		if(*i1>*i2) return 2;
		i1++;
		i2++;
	}
	if(*i1!=0) return 2;
	if(*i2!=0) return 1;
	return 0;
}
