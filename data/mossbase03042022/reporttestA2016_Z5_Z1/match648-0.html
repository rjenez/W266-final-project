double p_ocjena(struct Student m) {
    double suma=0,prosjek;
    int i;
	for(i=0; i<m.br_ocjena; i++) {
		suma=suma+m.ocjene[i];
		if(m.ocjene[i]==0 || m.ocjene[i]==5) return 5;
	}
	if(suma==0) return 5;
	prosjek=suma/m.br_ocjena; 
	return prosjek;
}
