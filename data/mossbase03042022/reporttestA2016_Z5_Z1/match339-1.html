float prosjek_ocjena(int ocjene[], int broj_ocjena) {
    float prosjek, suma_ocjena = 0;
    int i;
    for (i = 0; i < broj_ocjena; i++) {
        suma_ocjena = suma_ocjena + ocjene[i];
        if (ocjene[i] == 5 || broj_ocjena == 0) return 5.0;
    }
    prosjek = suma_ocjena / broj_ocjena;
    return prosjek;
}
