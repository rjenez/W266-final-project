struct Student {

//funkcija za leksikografsko poredjenje
int leksik(char* r1, char* r2)
{
	while(*r1 != '\0' && *r2 != '\0') {
		if(*r1 < *r2)
			return -1;
		if(*r1 > *r2)
			return 1;
		r1++;
		r2++;
	}
	if(*r1 =='\0' && *r2 == '\0')
		return 1;
		
	return 0;

}
//funkcija za racunanje prosjeka
float pr(struct Student stud)
{
	int j;
	float s_avg=0;
	if(stud.br_ocjena==0)
		return 5.0;

	for(j=0; j<stud.br_ocjena; j++) {
		s_avg+=stud.ocjene[j];
		if(stud.ocjene[j]==5)
			return 5.0;
	}
	return s_avg/stud.br_ocjena;
}
