int leksikon(char *p,char *q){
	
	while(*p != '\0'){
		
		if(*p > *q)return 1;
		if(*p < *q)return 0;
		if(*q == '\0')return 1;
		p++;
	}	
	if(*p == '\0' && *q!='\0')return 0;
	//ovdje je receno u zadatku da mozemo ispisati bilo kojeg
	if(*p =='\0' && *q == '\0')return 1;
}

void intcopy(int *p, int *q,int vel){
	int i = 0;
	
	while(i<vel){
		*p = *q;
		p++;
		q++;
		i++;
	}
}
