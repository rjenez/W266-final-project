#include <stdio.h>

int strcmp(char* s1, char* s2)
{
	while (*s1 != '\0' && *s2 != '\0') {
		if (*s1 < *s2) return -1;
		if (*s1 > *s2) return 1;
		s1++;
		s2++;
	}

	if(*s1=='\0'&&*s2=='\0') return 0;
	if (*s1 != '\0') return 1;
	if (*s2 != '\0') return -1;


	return 0;

}

struct Student {

	char ime[20], prezime[20];
	int ocjene[50];
	int br_ocjena;
};

/*funkcija koja računa prosjek jednog studenta*/
double prosjek (struct Student st)
{

	int i;
	double sum=0;

	if(st.br_ocjena==0) {
		return 5;
	}



	for(i=0; i<st.br_ocjena; i++) {

		if(st.ocjene[i]==5) {
			return 5;
		}

		sum=sum+st.ocjene[i];
	}
	return sum/st.br_ocjena;
}
