void unesi(char* s, int broj){
	char znak = getchar();
	int i;
	if(znak == '\n') znak=getchar();
	i=0;
	while(i<broj && znak!='\n'){
		s[i]=znak;
		znak= getchar();
		i++;
	}


	s[i]='\0';
}
