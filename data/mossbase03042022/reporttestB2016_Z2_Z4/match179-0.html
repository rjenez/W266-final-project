int PotencijalniKrivci(char **&znak, std::vector<std::string> vektor){
	int n=vektor.size();
	int brojac(0);
	try{
		char **niz(new char*[n]);
		for(int i(0);i<n;i++){
			niz[i]=nullptr;
		}
		try{
			for(int i(0);i<n;i++){
				int m=vektor[i].size();
				niz[i]=new char[m+1];


				for(int j(0);j<m;j++){
				niz[i][j]=vektor[i][j];
			}
			niz[i][m]='\0';
			brojac++;
			}
			znak=niz;
		}
			catch(std::bad_alloc){
				for(int i(0);i<n;i++) delete[] niz[i];
				delete[] niz;
				throw;
			}
		}
		catch(std::bad_alloc){
			std::cout<<"Problemi sa memorijom!\n";
		}
		return brojac;
}
