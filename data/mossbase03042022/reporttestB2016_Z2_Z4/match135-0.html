int PotencijalniKrivci(char **&pok,std::vector<std::string> v){
	try{
		char **c(new char*[v.size()]);
		for(int i=0;i<v.size();i++) c[i]=nullptr;
		try{
			for(int i=0;i<v.size();i++) {
				c[i]=new char[v[i].length()+1];
				std::strcpy(c[i],v[i].c_str());
				}
			pok=c;
			return v.size();
		}
		catch(std::bad_alloc){
			for(int i=0;i<v.size();i++) delete[] c[i];
			delete c;
			throw;
		}
	}
	catch(std::bad_alloc){
		throw;
	}
}

	int brojac(0);
	bool pronadjen(false);
	for (int i=0;i<vel;i++){

			brojac++;
			delete[] pok[i];
			pok[i]=nullptr;
			pronadjen=true;
		}
		if(brojac>10){
			try{
				vel=vel-11;
				char** novi(new char *[vel]);

					for(int k=0;k<vel+11;j++){
						if(pok[k]!=nullptr){
							novi[j]=pok[k];

					}
				}
				delete[] pok;
				pok=novi;

}

}
