int PotencijalniKrivci(char** &p, std::vector<std::string> v){
	int i(0);
	try{
		p = nullptr;
		p = new char*[v.size()];
		for(i = 0; i < v.size(); i++){
			p[i] = nullptr;
			p[i] = new char[v[i].size() + 1];
			for(int j = 0; j < v[i].size(); j++){
				p[i][j] = v[i][j];
				if(j == 0) p[i][v[i].size()] = '\0';
			}
		}
	}catch(...){
		for(int k = 0; k < i; k++) delete[] p[k];
		delete[] p;
		throw std::bad_alloc();
	}
	return i;
}
