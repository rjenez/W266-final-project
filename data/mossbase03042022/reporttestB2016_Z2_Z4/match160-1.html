int PotencijalniKrivci (char **&ref, std::vector<std::string> imena) {
	int duz=imena.size();
	char **pok(new char*[duz]);
	for(int i=0; i<duz; i++) {
		pok[i]=new char[imena[i].size()+1];
		for(int j=0; j<imena[i].size(); j++){
			pok[i][j]=imena[i][j];
		}
		pok[i][imena[i].size()]='\0';
	}
	ref=pok;

	return duz;
}
