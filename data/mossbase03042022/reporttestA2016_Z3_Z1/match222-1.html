#include <stdio.h>



int daj_prost()
{
	int i;
	/*prost postavljamo kao staticku promjenljivu jer 
	nam treba da svaki sljedeci put imamo sljedeci prost broj*/
	static int prost=1;
	prost++;
	/*racunanje prostog broja*/
	for(i=2;i<prost;i++)
	{
		if(prost%i==0)
		{
			prost++;
			i=2;
		}
	}
	return prost;
	
}
