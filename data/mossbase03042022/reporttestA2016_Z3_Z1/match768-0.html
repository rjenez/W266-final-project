#include <stdio.h>
#include <math.h>
int daj_prost() {
	int flag,i=2,x;
	static int n=2;
	do{
		flag=1;
		for(i=2;i<=sqrt(n);i++) {
			if(n%i==0) flag=0;
		}
		if(flag==1) {
			x=n;
			n++;
			return x;
		}
		n++;
	}while(flag==0);
}

}
