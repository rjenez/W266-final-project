#include <stdio.h>

int prost(int x) {
	int i;
	if(x == 1) return 0;


	for(i = 2; i < x; i++) {
		if(x % i == 0) return 0;
	}
	return 1;
}

int daj_prost(){
	static int broj=1;
	broj++;
	
	while(!prost(broj)) broj++;
	
	return broj;
}
