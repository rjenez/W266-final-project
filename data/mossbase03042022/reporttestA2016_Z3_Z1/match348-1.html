#include <stdio.h>

int daj_prost(){
	static int l=2;
	int i,provjera;
	do {
		provjera=1;	
		for(i=2; i<=(sqrt(l)); i++) {
			if(l==2){ 
				break; 
			}
			else if(l%i==0) { 
				provjera=0;
			}
		}
		if(provjera==0) {
			l++;
		}
		else break;
		
	} while(1==1);
	l++;


	return l-1;
}
