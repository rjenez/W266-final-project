#include <stdio.h>

int prost(int x){
	int i;
	
	for(i=2; i<x; i++){
		if(x%i == 0) return 0;
	}
	return 1;
}

int daj_prost(){
	
	static int broj = 1;
	int i;
	
	for(i=broj+1;;i++){
		if(prost(i)){
			broj = i;
			return broj;
		}
	}
}

}
