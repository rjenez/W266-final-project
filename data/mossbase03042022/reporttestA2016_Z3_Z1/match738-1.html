#include <stdio.h>



int daj_prost() {
	static int n = 1 ;
	int prost,i,j;
	for(i=n+1; ; i++) {
		prost=1;
		for(j=i/2; j>1; j--) {
			if(i%j==0) {
				prost = 0;
				break;
			}
		}
		if(prost) {
			n=i;
			break;
		}
	}
	return n;
}
