int daj_prost()
{
	static int x=2;
	int i;
    /*petlja se ponavlja dok ne pronadje slijedeci prost broj*/
	while(1) {
		int prost = 1;
	/*provjeravamo da li je vrijednost promjenjive x prost broj*/
		for(i=2; i<x; i++) {
			if(x%i==0) {
				prost = 0;
				break;
			}
		}
		if (prost) return x++;
		else x++;
	}
}
