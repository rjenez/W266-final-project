#include <stdio.h>





int DaLiJeProst(int t){
	int brojac=0;
	int i;
	for(i=1; i<t; i++){
		if(t%i==0) brojac++;
	}
	if(brojac==1) return 1;
	else return 0;
}

int daj_prost(){
	static int x=2;
	while(1){
		if(DaLiJeProst(x)==1){
			return x++;
		}
		x++;
	}
}
