#include <stdio.h>
#include <math.h>

int daj_prost (){

	static int broj=1;
	int je_prost, i;

	do {
		broj++;
		je_prost=1;
		
		/*provjera da li je broj prost*/
		for (i=2; i<=sqrt(broj); i++) { 
			if (broj%i==0) {
				je_prost=0;
				break;
			}
		}
	} while (je_prost==0);

	return broj;
}
