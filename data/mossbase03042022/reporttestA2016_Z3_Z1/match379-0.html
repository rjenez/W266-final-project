#include <stdio.h>

int daj_prost ()
{

    static int prost=1;
    int i, brprost=0;

    do {
        brprost=0;
        prost++;
        for (i=1; i<=prost; i++) {
            if (prost%i==0) {
                brprost++;
            }
        }
    } while (brprost !=2);   /*provjeravanje prostog broja*/


    return prost;
}
