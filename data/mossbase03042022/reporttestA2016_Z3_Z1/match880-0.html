#include <stdio.h>

int Da_Li_Je_Prost (int n) {
	int prost=1,i;
	for (i=2; i<=n/2; i++) {
		if (n%i==0) {
			prost=0;
			break;
		}
	}
	return prost;
}

int daj_prost() {
	static int prost_broj=1;
	do {
		prost_broj++;
	} while (Da_Li_Je_Prost(prost_broj)==0);
	return prost_broj;
}
