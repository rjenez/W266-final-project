#include <stdio.h>

int daj_prost(){
	static int n=2;
	for(;;){
		int i, broj=1;
		for(i=2;i<n;i++){
			if(n%i==0) broj=0;
			if(broj==0) {n++;break;}
			}
			if(broj==1) {
				broj=n;
				n++;
				return broj;
		}
	}
}
