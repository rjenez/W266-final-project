int daj_prost() {
	static int prost=1;
	int i, j;
	for(i=prost+1; ; i++) {
		for(j=2; j<i; j++) {
			if(i%j==0) break;
		}
		if(j==i) {
			prost=i;
			return i;
		}
	}
}
