#include <stdio.h>

int daj_prost()
{
	static int i=1;
/*static jer treba svaki put kada pozovem daj_prost() da daje sljedeci prost, inace bi uvijek isao od i*/
	int dodatak,j;
	if(i==1)
	{
		i++;
		return 2;
	}
	dodatak=1;
	i++;
/*provjerava da li je i prost broj*/
	do
	{
		for(j=2;j<i;j++)
		{
			if(i%j==0)
			{
				dodatak=0;
				break;
			}
			else dodatak=1;
		}
		i++;
	}while(dodatak==0);
	i--;


	return i;
}
