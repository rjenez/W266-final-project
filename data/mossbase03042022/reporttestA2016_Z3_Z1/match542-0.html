#include <stdio.h>

int daj_prost(){
	int j,prost;
	static int i=1;
	while(1){
		i++;
			prost=1;
		for(j=2;j<i;j++){
			if(i%j==0){
				prost=0;
				break;
			}
		}
		if(prost)
			break;
	}
	return i;
}
