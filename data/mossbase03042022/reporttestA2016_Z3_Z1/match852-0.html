#include <stdio.h>

int da_li_je_prost_broj(int broj){
	int kazi=1, i;
	if(broj<2) return 0;
	for (i=2; i<broj; i++){
		if(broj%i==0) {kazi=0; break;}
	}
	return kazi;
}

int daj_prost(){
	static int a=1;
	a++;


	while(da_li_je_prost_broj(a)!=1) a++;
	return a;

}
