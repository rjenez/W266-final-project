#include <stdio.h>
int daj_prost(){
	    /*racunamo prost broj najblizi trenutnom clanu intervala*/ 
		static int zadnji_prost=1; 
		int broj, j, prost=0;
		
        /*racunamo sljedeci prost broj*/                  
		broj=zadnji_prost;        	
		do{
		  	broj++;	
			for (j=2; j<broj; j++)
        		if (broj % j == 0) break;
			if (j == broj) {
	   			prost = 1;
   				zadnji_prost=broj;
	    	} 
		} while (prost == 0);


		return zadnji_prost;
}
