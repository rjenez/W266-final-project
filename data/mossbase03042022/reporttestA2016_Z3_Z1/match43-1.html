#include <stdio.h>


int daj_prost(){
	static int x=1;
	int i,prost;
	do{
		x++;
		prost=1;
		if(x==2)
		break;
		else{
			for(i=2;i<x;i++){
				if(x%i==0){
					prost=0;
					break;
				}
			}
		}
	}
	while (prost==0);
	return x;
}
