#include <stdio.h>

int iz_it_prost(int n)
{
	int i;
	for(i = 2; i < n; i++)
	{
		if(n%i == 0)
		  return 0;
	}
	return 1;
}

int prost_po_redu(int n)
{
	int i, br = 0;
	for(i = 2; i <= n; i++)
	{
		if(iz_it_prost(i) == 1)
		  br++;
	}
	return br;
}

int daj_prost()
{
	static int x = 2;
	
	while(iz_it_prost(x) == 0)
	{
		x++;
	}
	x++;
	return x - 1;
}
