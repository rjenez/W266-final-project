/* Funkcija koja izbacuje svaki slijedeci prost broj */
int daj_prost() {
	int prost=1,i;


	static int broj=1;
    while(1){
    	broj++;
    	if (broj==2){
    		return broj;
    	}
    	else {
    		for (i=2;i<broj;i++){
    			if (broj%i==0){
    				prost=0;
    				break;
    			}
    		}
    		if (prost){
    			return broj;
    		}
    		else {
    			prost=1;
    		}
    		
    	}
    }
}
