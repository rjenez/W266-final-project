#include <stdio.h>

int daj_prost() {
	static int snimi=0,izlaz=1,var=-1;


	int i,j,prost_broj=1;
	i=++izlaz;
	while(i>0) { 
		prost_broj=1;
		for(j=i-1;j>1; j--)
		if(i%j==0) prost_broj=0;
		if(prost_broj==1)var++;
		if(snimi==var){izlaz=i;snimi++;break;}
		i++;
	}
	return izlaz;
}

int main() {
	int i, broj_A=0, broj_B=0,radi=1;
	long long int suma_prostih=0;

	else radi=0;}

	while(radi<broj_B){

}
