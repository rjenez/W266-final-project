#include <stdio.h>

int daj_prost() {
	static int prost_br=1;
	int i=0;
	


	while (prost_br > i){
		prost_br++;
		for (i=2; i<prost_br; i++){
			if (prost_br % i == 0) break;
		}
	}
	return prost_br;
}
