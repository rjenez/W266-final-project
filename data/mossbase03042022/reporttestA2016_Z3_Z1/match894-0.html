#include<stdio.h>
int daj_prost() {
	static int n=2; /* Svakim pozivom vraca sljedeci prost broj */
	int i;
	if (n==2) {
		n++;
		return 2;
	}
	for(i=2; i<n; i++) {
		if(n%i==0) {
			n++;
			i=1;
		}
	}


	n++;
	return n-1;
}
