int daj_prost() {
    static int prost=1;
    int i, kontrolna=1;
    while(kontrolna){
        prost++;
        if(prost==2) return prost;
        for(i=2; i<prost; i++) {
            if(prost%i==0){
                break;
                kontrolna=2;
            }
            if(i==prost-1){
                kontrolna = 0;
                return prost;
            }
        }
    }
}
