int jeProst(int br) {
	
	int i = 2;
	
	if(br ==0 || br == 1 || br == 2) return 0;
	
	for(i = 2; i < br; i++) {
		
		if(br % i == 0) return 0;
	}
	return 1;
}


int daj_prost() {

	if(prost >= 2) prvi_put = 0;

		prost = 2;
		prost++;
		return prost - 1;
	}
	
	while(1) {
		if(jeProst(prost)) {
			prost++;
			break;
		} else prost++;
		
	}
		
		return prost - 1;
	
	
}
