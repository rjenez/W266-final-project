#include <stdio.h>

int daj_prost(){
	static int x=1;
	int prost,i;
	prost=1;
	x++; /* mora se povecavati promjenjiva x zbog toga sto promjenjiva static ce imati istu vrijednost ukoliko je ne povecamo */
	do
	{	
		for(i=2;i<x;i++)
		{
			if(x%i==0)
			{
				prost=0;
				break;
			}
			else prost=1;
		}	
		x++;
	}while(prost==0);
	x--;
	return x;
}
