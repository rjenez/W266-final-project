#include <stdio.h>



int daj_prost() {
	int i,k,prost=1;
	static int b=2;
		do{
			for(i=2; i<b; i++){
				if(b%i==0){
					prost=0;
					break;
					
				}
			}
				if(i==b)
					prost=1;
				if(prost==0)
					b++;
				}
		while(prost==0);
		k=b++;
		return k;
}
