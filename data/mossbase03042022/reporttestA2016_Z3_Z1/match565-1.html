#include <stdio.h>
/*trazenje prostog broja uz ocuvanje njegove vrijednosti*/
int daj_prost(){
	static int broj=1;
	int i;
	broj++;
	do {
	for (i=2;i<broj;i++){
		if (broj%i==0) {
		broj++;
		break;
		}
	}
	} while (broj%i!=0);
	
	return broj;
	
}

}
