#include <stdio.h>
int daj_prost () {


	int i,k;
	do {
		k=0;
		/*Koristimo staticku promjenljivu jer zelimo zadrzati njenu vrijednost*/
		static int n=2;
		for(i=2;i<n;i++) {
			if(n%i==0) k++;
		}
		/*Prost broj zadovoljava uslov*/
		if(k==0) {
			n++;
			return n-1;
		}
		/*Ako broj nije prost, promjenljiva se svejedno mora povecati*/
	n++;
	} while (k!=0);
}
