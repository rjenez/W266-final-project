#include <stdio.h>
int daj_prost(){
	int i,prost=0;
static int n=1;
    n++;


	for(i=2;i<n;i++)
		if(n%i==0){
			prost=1;
			break;
		}
		if(prost==0)
		return n;
		else
			return daj_prost();
		
}
