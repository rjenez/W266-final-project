#include <stdio.h>
int Prost(int n){
	int i;
	for(i=2; i<n; i++){
		if(n%i==0) return 0;
	}
	return 1;
}
int daj_prost(){
	static int n=2;
	int x=n,i;
	for(i=0;i<n;i++){
		if(Prost(x)==1){
			n++;
			return x;
		} else {
			n++; 
			x++;
		}
	}
}
