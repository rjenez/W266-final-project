#include <stdio.h>

int daj_prost ()
{
	static int x=2;
	int i,prost=0;
	while(prost!=1)
	{
		prost=1;
		for(i=2;i<x;i++)
		{
			if(x%i==0)
			{
				prost=0;
				break;
			}
		}
		if(prost==0) x++;
	}
	return x++;
	
}
