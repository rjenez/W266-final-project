		for (j=2; j<k; j++) {
			
			if (k%j==0) {
				prost=0;
			}
		}
		if (prost==0) {
			prost=1;
			k++; 
			continue;
		}
		else if (prost==1) {
			return k;
		}

}
return 0;
}
