#include <stdio.h>
#include <math.h>

int prostbr(int br) {
	int i, j=sqrt(br);
	if(br==2) return 1;
	else if (br<2 || !(br%2)) return 0;
	for (i=3;i<=j;i+=2) {
		if(br%i==0) return 0;
	}
	return 1;
}


int daj_prost() {
	static int prosti=1;
	for(;;) {
		if(prostbr(prosti)==0) prosti++;


		else break;
	}
	prosti++;
	return prosti-1;
}


int main() {
	int a, b, i, suma=0, br;
