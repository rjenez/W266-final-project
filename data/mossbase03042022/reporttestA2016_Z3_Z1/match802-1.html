#include <stdio.h>
/*postavljamo staticku promjenjivu jer je potreban sljedeci prosti broj svaki iduci put*/


int daj_prost () {
	int i;
	static int prost_broj=1;
	prost_broj++;
	/*racunamo prost broj*/
	for(i=2;i<prost_broj;i++){
		if(prost_broj%i==0){
			prost_broj++;
			i=2;
		}
	}
	return prost_broj;
}
