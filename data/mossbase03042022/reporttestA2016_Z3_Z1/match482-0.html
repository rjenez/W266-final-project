#include <stdio.h>

int daj_prost()
{
	
	int i = 1, brojac = 0;
	static int pocetak = 2;                            
	
	for ( i = 1; i <= pocetak; i++ ) {
		if ( pocetak % i == 0 ) {                                      /*Trazenje prostih brojeva*/
			brojac++;                                  
		}
		if ( brojac > 2 ) {                           
			pocetak++;                                
			brojac = 0;
			i = 0;
		}
	}                
		pocetak++;                                     
	
  return (pocetak - 1);                              
}
