#include <stdio.h>

int daj_prost() {


    int m=0, n;
    /*ispitivanje da li je broj prost*/   
    do {
        m=0;
        static int p=2;
        for (n=2; n<p; n++)
        {
            if (p%n==0) 
            {
                m++;
            }
            
        }
            if (m==0)
            {
                p++;
                return p-1;
            }
           p++; 
            
        }
    
        while (m!=0);
}
