int daj_prost()
{

	for(i=curr+1;;i++)
	{
		 korijen = sqrt(i)+1;
		 prost = 1;
		for(j=2;j<korijen;j++)
		{
			if(i%j==0)
			{
				prost = 0;
				break;
			}
		}
		if(prost)
		{
			curr = i;
			break;
		}
	}
	return curr;
}
