#include <stdio.h>

int daj_prost()
{
    static int prost=1; /* prema preporuci profesora koristi se staticka varijabla */
    int nadjen=0, i;

    while(nadjen != 1) 
    {
        prost++;
        nadjen=1;
        for(i=2; i<prost; i++)
        {
            if(prost%i==0)
            {
                nadjen=0;
            }
        }
    }
    
    return prost;	
}
