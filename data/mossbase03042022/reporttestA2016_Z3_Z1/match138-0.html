#include <stdio.h>


int daj_prost()
{
	static int n=1;
	int prost=1;
	int i;
	
	while(1==1)
	{   
		prost=1;
		n++;


		for(i=2;i<n;i++)
			if(n%i==0) prost=0;
		if(prost==1) break;
	}
	return n;
}
