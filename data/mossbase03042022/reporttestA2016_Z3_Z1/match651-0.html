#include <stdio.h>

int daj_prost(){
	int i;
	static int n=1;
	n++;
	if(n==2) return n;
	
	/*printf("usao i n je %d",n);*/
	for(;;){
		int nasao=1;
		if(n==2) return n;


		for(i=2;i<=sqrt(n);i++){
			if(n%i==0) {
				nasao=0;
				break;
			}
		}
		if(nasao==1) return n;
		else n++;
	}
}
