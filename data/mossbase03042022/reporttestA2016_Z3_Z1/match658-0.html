#include <stdio.h>

int daj_prost(){
	//Funkcija koja putem static int odrzava u memoriji nadjeni prosti broj
	static int prost=1;
	int i,k=0;
	for(;;){
		prost++;
		k=0;
		for(i=prost+1; i>1; i--){
			if(prost%i==0){
				k++;
			}
		}
		if(k==1){
			break;
		}
	
		
	}


	return prost;
}
