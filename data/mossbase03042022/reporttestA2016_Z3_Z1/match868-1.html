int daj_prost()
{
	static int i=2;
	int j, prost;
	int broj=0, br=0;
	
	/*Ispitivanje da li je broj prost*/
	
	do{
		br=0;
		for(j=2; j<i; j++)
		{
			if(i%j==0) {
				br++;
			}
			
		}
		
	if(br<1) prost=1;
	else prost=0;
	
	if(prost==1) /*Ukoliko je broj prost funkcija vraca vrijednost*/
		break;
	
	else i=i+1; /*Ukoliko broj nije prost petlja se opet izvrsava do sljedeceg prostog broja*/

	}while(prost==0);
	
	broj=i;
	i=i+1;
	
	

	return broj;
	

}
