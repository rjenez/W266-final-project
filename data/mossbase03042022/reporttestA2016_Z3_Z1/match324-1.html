#include <stdio.h>

int isPrime(int number) {
    int i;


    if(number == 0 || number == 1) return 0;
    for (i = 2; i < number; i++) {
        if (number % i == 0) return 0;
    }
    return 1;
}

int daj_prost() {
	static int broj = 0;
	static int oldValue;
	oldValue = broj;
	if(broj - oldValue == 0) broj++;

	while(isPrime(broj) == 0)
		broj++;
		
	return broj;
}
