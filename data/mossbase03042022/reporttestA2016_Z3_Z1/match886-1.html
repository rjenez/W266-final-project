#include <stdio.h>

int daj_prost() {
	static int n=2;
	int i, prost=0, brojac, broj;
	while(prost==0) {
		brojac=0;
		for(i=2; i<n; i++) {
			if(n%i==0) {
				brojac++;
				break;
			}
		}
		if(brojac==0) { 
			broj=n++;
			break;
		}
		n++;
	}


	return broj;
}
