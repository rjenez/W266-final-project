#include <stdio.h>



int daj_prost()
{

	static int br=1;
	int i,prost;

	br++;


	do {

		prost=1;
		for(i=2; i<br; i++) {
			if(br%i==0) {
				prost=0;
				br++;
				break;
			}
		}

	} while(prost==0);


	return br;
}
