#include <stdio.h>
/*Funkcijom "prost" provjerava se jesu li brojevi prosti*/
int prost (int x) {
	int i;
	for(i=2; i<x; i++) {
		if(x%i==0) return 0;
	}
	return 1;
}
/*Funkcija "daj_prost" uzima te proste brojeve*/
int daj_prost () {
	static int t=1;
	do {
		t++;
	} while (prost(t)==0);
	return t;
}
