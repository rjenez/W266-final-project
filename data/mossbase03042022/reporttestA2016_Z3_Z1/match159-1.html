int daj_prost(){
	static int n=1;
	int i,prost;
	do{
		n++;
		prost=0;
		if (n==2)
		break;
		else {
			for(i=2; i<n; i++){
				if (n%i==0){
					prost=1;
					break;
				}
			}
		}
	}while(prost==1); // Napravili smo funkciju da li je prost broj.
	return n;
}
