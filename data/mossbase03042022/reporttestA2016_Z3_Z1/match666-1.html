#include <stdio.h>
int daj_prost() {
	static int n=3, prvi=1;
	if (prvi) {
		prvi=0;
		return 2;
	}
	while (1) {
		int i, prost=1;
		for (i=2; i<n; i++) {
			if (n%i==0) {
				prost=0;
			}
			if (prost==0) {
				n++;
				break;
			}
		}
		if (prost) {
			prost=n;
			n++;
			return prost;
		}
	}
	
}
