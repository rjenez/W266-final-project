int daj_prost()
{
	static int prost_broj=1;

		prost_broj++;
		if(prost_broj==2)
		{
			return prost_broj;
		}

		for(i=2;i<prost_broj;i++)
		{
			if(prost_broj%i==0)
			{
				prost=0;

		}
		if(prost==1)

		}

	}
}
