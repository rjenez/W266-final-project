#include <stdio.h>
#include <math.h>

int daj_prost()
{
	static int x = 2; 
	int broj_koji_saljemo;
	int prost;
	do
	{
		int i;
		prost = 1;
		for(i = 2; i*i <= x; ++i)
			{
				if(x%i==0)
			{
				prost=0;
				break;
			}
		
		}
		broj_koji_saljemo = x++;
	}
	while(!prost);
	return broj_koji_saljemo;
}
