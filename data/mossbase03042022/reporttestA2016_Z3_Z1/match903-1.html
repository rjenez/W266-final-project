#include <stdio.h>

#define TRUE 1
#define FALSE 0
int isprime(int n) {
	int i, root;
	root=(int)sqrt(n);
	if(n==2) return TRUE;
	for(i=3; i<=root; i+=2) {
		if((n%i)==0) {
			return FALSE;
		}
	}
	return TRUE;
}
int daj_prost() {
	int n;
	static int last_prime=0;
	if(!last_prime) {
		last_prime=1;
		return 2;
	}
	for(n=last_prime+2; !isprime(n); n+=2);
	last_prime=n;
	return n;
}
