#include <stdio.h>

int daj_prost(){
	int i, prost=1;
	static int n=2;
	if(n==2){
		n++;
		return n-1;
	}
	else{
		for(i=2; i<n; i++){
			if(n%i==0){
				prost=0;
				break;
			}
		}
		while(prost==0){
				n++;
				prost=1;
				for(i=2; i<n; i++){
					if(n%i==0){
						prost=0;
						break;
					}
				}
		}
		n++;
		return n-1;
	}
}
