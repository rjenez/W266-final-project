#include <stdio.h>

int daj_prost(){
	static int PROST=2;
	int i, br;


	for(;;PROST++){
	br=1; //ispitivanje da li su brojevi prosti ili slozeni
	for(i=2; i<PROST; i++){
		if(PROST%i==0){
			br=0;
		}
	}
	if(br==1) return PROST++;
	}

	
}
