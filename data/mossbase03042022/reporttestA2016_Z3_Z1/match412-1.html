#include <stdlib.h>

int prost(int n){
    int i;


    for(i=2;i<n;i++){
        if(n%i==0) return 0;
    }
    return 1;
}
int daj_prost(){
    static int var=2;
    while(1){
        if(prost(var)==1) return var++;
        else var++;
    }
}
