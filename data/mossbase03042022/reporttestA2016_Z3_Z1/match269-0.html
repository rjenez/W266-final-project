#include <stdio.h>

/* Funkcija DAJ PROST */

int daj_prost() {
	static int x=1;
	int m, n=0;
	do{
		int i, brojac=0;
		if(x==1){
			m=0;
		}
		else{
			for(i=2;i<=x/2;i++){
				if(x%i==0){
					brojac++;
				}
			}
			if(brojac==0){
				m=1;
			}
			else{
				m=0;
			}
		}
		x++;
	} while(m!=1);
	n=x-1;


	return n;
}
