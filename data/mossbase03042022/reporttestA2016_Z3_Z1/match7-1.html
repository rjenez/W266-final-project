#include <stdio.h>



int IFPROST(int prost_br){
    int i,n,r;
    n=1;
    i=2;
    for(i=2;i<prost_br;i++){
        if(prost_br%i==0){
            n=0;
            break;
        }
    } 
    if(n==1) r=1;
    else r=0;
    return r;
}

int Daj_prost(){
    static int prost_br=2;
    while(IFPROST(prost_br)==0){
        prost_br++;
    }
    prost_br++;
    return prost_br-1;
}
