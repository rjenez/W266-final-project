#include <stdio.h>

int daj_prost(){
	int k, c, j;


	static int i = 2;
	for (c = i; c < 1000000; c++){ 
		k = 1;
		for (j = 2; j < c; j++){
			if(c % j == 0){
				k = 0; 
			}
		}
		i++;
		
		if (k == 0) 
			continue;
		
		if (k == 1)
			return c;
	}
}
