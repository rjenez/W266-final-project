#include <stdio.h>

int daj_prost (){
	static int prost=2;
	int i, prosti_broj=0;
	for(i=2;i<=prost;i++){
		


		if((prost%i==0)&&(prost==i)){
			prosti_broj=prost;
			break;
		}else if((prost%i==0)&&(prost!=i)){
			prost++;
			i=2;
		}
	}
	prost++;
	return prosti_broj;
}
