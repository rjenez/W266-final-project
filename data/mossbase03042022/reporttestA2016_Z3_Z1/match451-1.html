#include <stdio.h>

int daj_prost () {
	int i, j, prost=1;
	static int a=0, izl=1, slucaj=-1;
	i=++izl;
	while(i>0) {
		prost=1;
		for(j=i-1;j>1;j--) 
		if(i%j==0) prost=0;
		if(prost==1) slucaj++;
		if(a==slucaj) {
			izl=i;
			a++;
			break;
		}
		i++;
	}
	return izl;
}
