#include <stdio.h>

int daj_prost(){
	int i, k, prost=0;
	static int n=1;
	do{
		if (n==2){prost=1; k=n;}
		for (i=2; i<n; i++){
			if (n%i!=0){
				prost=1;
				k=n;
			}else{
				prost=0; break;
			}
		}
		n++;
	}while(prost==0);
	if (prost==1){
		return k;
	}else{return 0;}
}
