#include <stdio.h>



int prost (int n)
{
	int i;
	if (n==0) return 0;
	for (i=2; i<n; i++)
	{
		if (n%i==0) return 0;
	}
	return 1;
}

int daj_prost ()
{
	int i;
	static int broj=1;
	for (i=broj+1; i<2147483646; i++)
	{
		if (prost(i))
		{
			broj=i;
			break;
		}
	}
	return broj;
}
