#include <stdio.h>
#include <stdlib.h>



int main() {
	int n, rezultat = 0, cifra1, cifra2, obrnuta = 0, cifre = 0;
	printf("Unesite broj: ");

	n = abs(n);
	if(n/10 == 0) {
	    printf("%d", rezultat);
	    return 0;
	}
	do {
	    cifra1 = n%10;
	    n = n/10;

	        obrnuta = obrnuta*10 + abs(cifra2 - cifra1);
	        cifre++;
