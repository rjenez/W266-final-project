#include <stdio.h>
#include <math.h>



int main() {
	int n , Nr1, Nr2d, rezultat=0, k=0;

	if(n < 0){
		
		n = fabs(n);
	} else if (n <= 0 && n > 10){
		rezultat = 0;
	}
	
	while(n >= 10){
		
		Nr1 = n % 10;
		
		Nr2d = (n/10) % 10;
