class Knjiga

        pok=nullptr;

        return naslov;

        return zanr;

        return pok;
    };
    void ZaduziKnjigu(Korisnik &k) {
        pok=&k;
    }
    void RazduziKnjigu() {
        pok=nullptr;
    }
    bool DaLiJeZaduzena() {
        if(pok==nullptr) return false;
        return true;
    }
};
