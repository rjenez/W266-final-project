typedef std::vector<std::vector<double>> Matrica;

Matrica H (const Matrica &v){
    double temp;
    Matrica mat(v);
    for(int i=0; i<mat.size(); i++){
        for(int j=0; j<mat[i].size()/2; j++){


            temp = mat[i][j];
            mat[i][j] = mat[i][mat[i].size()-1-j];
            mat[i][mat[i].size()-1-j] = temp;
        }
    }
    return mat;
}
