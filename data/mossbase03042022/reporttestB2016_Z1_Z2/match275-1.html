vector<vector<double> > ver(vector<vector<double> > mat)
{


    vector<vector<double> > pom; pom.resize(mat.size());
    for(int i=0; i<mat.size(); i++) pom[i].resize(mat[i].size());
    for(int i=0; i<mat.size(); i++)
    {
        int a=mat[0].size()-1;
        for(int j=0; j<mat[i].size(); j++)
        {
            pom[i][j]=mat[i][a];
            a--;
        }
    }
    return pom;
    
}
