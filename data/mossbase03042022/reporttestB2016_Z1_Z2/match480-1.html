mat obrlati(mat m, bool t)
{

	{

		{


			int p=m[i].size()-1;
			for(int j=0; j<nova[i].size(); j++)
			{
				nova[i][j]=m[i][p];
				p--;
			}
		}
	}
	else {


		int p=m.size()-1;
		for(int i=0; i<nova.size(); i++)
		{
			for(int j=0; j<nova[i].size(); j++)
			{
				nova[i][j]=m[p][j];
			}
			p--;
		}
	}
	return nova;
}
