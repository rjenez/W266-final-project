Matrica Horiz (Matrica mat)
{

	return hmat;
}

Matrica Verti (Matrica mat)
{

			vmat[i][j] = mat[mat.size()-1-i][j];
	return vmat;
	
}

Matrica HorizVerti (Matrica mat)
