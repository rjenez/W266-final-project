#include <iostream>

typedef std::vector<std::vector<double>> Matrica;

Matrica hfilp (Matrica M){
	if(M[0].size()==0||M[0].size()==1)
	return M;
	int m=M.size();


	int n=M[0].size();
	
	Matrica h(m, std::vector<double>(n));
	
	for ( int i = 0; i<m; i++){
		for ( int j = 0; j<n; j++){
			h [i][j] = M[i][ n - 1 - j ];
		}
	}
	return h;
}
Matrica vflip (Matrica M){
	if(M.size()==0||M.size()==1)
	return M;
	int m=M.size();


	int n=M[0].size();
	
	Matrica v(m, std::vector<double>(n));
	
	for ( int i = 0; i<m; i++){
		for ( int j = 0; j<n; j++){
			v[i][j] = M[m - i - 1][j];
		}
	}
	return v;
}
Matrica hvflip(Matrica M){
	Matrica hv(vflip(M));
	hv = hfilp(hv);

	return hv;
}
