mat HorizontalnoOgledalo(mat m){           //horizontalno ogledalo
	mat h(m.size());
    int l;


    for(int i=0; i<m.size(); i++)
        h[i].resize(m[0].size());
	for(int i=0; i<m.size(); i++){
		l=m[0].size()-1;
		for(int j=0; j<m[i].size(); j++){
		h[i][l]=m[i][j];
		l--;
		}
	}
	return h;
}

mat VertikalnoOgledalo(mat m){            //vertikalno ogledalo

		for(int j=0; j<m[i].size(); j++)

}

mat KombOgledalo(mat m){                  //kombinovano ogledalo
