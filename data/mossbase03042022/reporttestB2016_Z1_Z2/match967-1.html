#include <iostream>

typedef std::vector<std::vector<double>> Matrica;

Matrica horizontalnaMatrica(Matrica m) {
	Matrica hm(m.size(), std::vector<double> (m[0].size()));
	int d=m.size();
	int s=m[0].size();
	
	for(int i=0; i<d; i++) {
		for(int j=0; j<s; j++) {
			hm[i][j]=m[i][s-1-j];
			}
	}
	return hm;
}

Matrica vertikalnaMatrica(Matrica m) {
	Matrica vm(m.size(), std::vector<double> (m[0].size()));
	int d=m.size()-1;
	int s=m[0].size();
	for(int i=d; i>=0; i--) {
		for(int j=0; j<s; j++) {
			vm[d-i][j]=m[i][j];
		}
	}
	return vm;
}

Matrica kombinovanoOgledalo(Matrica m) {
	Matrica hvm(m.size(), std::vector<double> (m[0].size()));
	int d=m.size()-1;
	int s=m[0].size()-1;
	for(int i=d; i>=0; i--) {
		for(int j=s; j>=0; j--) {
			hvm[d-i][s-j]=m[i][j];
		}
	}
	return hvm;
}
