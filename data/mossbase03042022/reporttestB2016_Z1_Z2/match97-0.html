matrica vertikalna (matrica m){
	int s;
	if(m.size()==0){
	  s=0;
	}
	else s=m[0].size();

	matrica n(m.size(), vektor(s));
	int k(0);
	for(int j=m.size()-1; j>=0; j--){
			n[k]=m[j];
			k++;
		}
	
	return n;
	
}
matrica horizontalna(matrica m){
	

	int s;
	if(m.size()==0){
	  s=0;
	}
	else s=m[0].size();
	
	matrica n(m.size(), vektor(s));
	
		for (int i=0; i<m.size(); i++){
			int p=(0);
			for(int j=m.at(i).size()-1; j>=0; j--){
				for(int k=p; k<m.at(i).size(); k++){
						n[i][k]=m[i][j];
						p++;
						break;
				}
    		 }

		}
	
		return n;}

matrica OgledaloMatrica(matrica m){

	matrica M (3*m.size(), vektor(3*s));

	for (i=0; i < m.size(); i++){

			M[i][j]=hv[i][j];
		}

		matrica v(vertikalna(m));
	for (i=0; i < m.size(); i++){

	}
	
	for (i=0; i < m.size(); i++){

			M[i][j]=hv[i][j-k];
		}

	for (l=i; l<2*m.size(); l++){

			M[l][j]=h[l-i][j-k];
