typedef std::vector<std::vector<double>> mat;

mat h(mat x)
{
	mat pomocna = x;
	mat rez;


	rez.resize(x.size());
	for(int i(0); i < x.size() ; i++)
	{
		for(int j(0); j < x[0].size(); j++)
		{
			rez[i].push_back(pomocna[i][x[0].size()-j-1]);


		}
	}
	return rez;
}

mat v(mat x)
{
	mat pomocna = x;
	mat rez;


	rez.resize(x.size());
	for(int i(0); i < x.size(); i++)
	{
		for(int j(0); j < x[0].size(); j++)
		{
			rez[i].push_back(pomocna[x.size()-i-1][j]);
		}
	}
	return rez;
}

mat hv(mat x)
{
	mat rez,temp;
	temp = h(x);
	rez = v(temp);
	return rez;
}
