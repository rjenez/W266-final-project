typedef std::vector<std::vector<double>> Matrica;


int BrojRedova (Matrica m)
{
	return m.size();
}

int BrojKolona (Matrica m)
{
	if (m.size() == 0)
	{
		return 0;
	}
	return m[0].size();
}


Matrica Horizontalna (Matrica m) //obrnemo redoslijed kolona u matrici
{
	int r = BrojRedova(m);
	int k = BrojKolona(m);
	double temp;
	
	for (int i = 0; i < r; i++)
	{


		for (int j = 0; j < k/2; j++)
		{
			temp = m[i][j];
			m[i][j] = m[i][k-j-1];
			m[i][k-j-1] = temp;
		}
	}
	return m;
}

Matrica Vertikalna (Matrica m) //obrnemo redoslijed redova u matrici
{
	int r = BrojRedova(m);
	int k = BrojKolona(m);
	double temp;
	
	for (int i = 0; i < r/2; i++)
	{
		for (int j = 0; j < k; j++)
		{
			temp = m[i][j];
			m[i][j] = m[r-i-1][j];
			m[r-i-1][j] = temp;
		}
	}
	return m;
}
