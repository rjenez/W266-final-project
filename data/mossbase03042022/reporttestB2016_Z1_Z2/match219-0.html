Matrica HorizontalnoOgledalo(Matrica a){
    double pomocna;
    for(int i=0;i<a.size();i++){
        for(int j=0;j<a[i].size()/2;j++){
            pomocna=a[i][j];
            a[i][j]=a[i][a[i].size()-j-1];
            a[i][a[i].size()-j-1]=pomocna;
        }
    }
    return a;
}
