#include <stdio.h>

double stepenovanje(int baza,int stepen)
{
    int i,rezultat=1;
    for(i =0; i < stepen; i++)
    {
        rezultat *= baza;
    }
    return rezultat;
}

int izbacicifru(int broj, int cifra)
{
    int broj_cifara=0,temp,rez=0,a;
    if(broj >= 0) temp = broj;
    else if(broj < 0) temp =-1*broj;
    while(temp != 0)
    {
            a = temp % 10;
            if(a != cifra)
            {
                rez += (temp % 10) * stepenovanje(10,broj_cifara);
                broj_cifara++;
            }
            temp /= 10;
    }
    if(broj < 0) rez*=-1;
    
    return rez;
}
