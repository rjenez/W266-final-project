int provjera(int cifre[], int vel){
    int i, j;
    for(i=0; i<vel; i++){
        if(cifre[i]<0 || cifre[i]>9) return 0;
        for(j=i+1; j<vel; j++){
            if(cifre[i]==cifre[j]) return 0;
        }
    }
    
    return 1;
}
