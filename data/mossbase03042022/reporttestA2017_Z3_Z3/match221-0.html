int da_li_ima_duplih (int *niz, int DUZINA)
{
    int i=0,j=0;
    for (i=0; i<DUZINA-1; i++) {
        for (j=i+1; j<DUZINA; j++) {
            if (niz[i]==niz[j] || niz[i]>9 || niz[i]<0) {
                return 1;
            }
        }
    }

}
