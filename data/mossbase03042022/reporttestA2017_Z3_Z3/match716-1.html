#include <stdio.h>

int izbaci_cifre(int *niz, int  vel, int *cifre,int vel1) {
	
		
	int i,j,c,suma=0;
	int k=1, neg=0,n;
	
	for(i=0; i<vel1; i++) 
		if(cifre[i]<0 || cifre[i]>9) return 0;
	
	for(i=0; i<vel1; i++) 
		for(j=i+1; j<vel1; j++) 
			if(cifre[i]==cifre[j]) return 0;
	
	


	for(i=0; i<vel; i++) {
		if(niz[i]<0) neg=1;
		
		n=fabs(niz[i]);
		
		while(n!=0) {
			c=n%10;
			for(j=0; j<vel1; j++) 
				if(c==cifre[j]) c=0;
				
			suma+=c*k;
			n/=10;
			if(c!=0) k*=10;
		}
	
		if(neg==1) suma*=-1;
		niz[i]=suma;
		suma=0;
		neg=0;
		k=1;
	}
	
	return 1;
	
}
