int vrati_broj (int n, int cifra){ /* f-ja koja izbacuje iz broja n cifru koja je proslijedjena */ 
	int broj = n, novi_br = 0, i=1, c;
	while (broj != 0){
		c = broj % 10;
		if (abs(c) != cifra){
		    novi_br += i*c;
		    i*=10;
		}


		broj /= 10;
	}
    return novi_br;
}
