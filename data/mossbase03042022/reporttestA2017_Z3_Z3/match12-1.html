int izbaci_cifre(int nizbr[],int a,int niz2[],int b)
{

	int i,j,k,brojac=0,brojac1=0,temp,noviBroj,temp1,cifra,l=0,q,m;
	
	double h,d;
	int brojac_c[100]= {0};
	int istina=1;
	int noviNiz[100]= {0};

	for(i=0; i<b; i++)
		for(j=i+1; j<b; j++)
			if(niz2[i]==niz2[j])
				brojac++;

	for(i=0; i<b; i++)
		if(niz2[i]<0 || niz2[i]>9)
			brojac1++;

	if(brojac !=0 || brojac1 !=0)
		return 0;



 else{

	for(i=0; i<a; i++) {
		temp=fabs(nizbr[i]);
		do {

			temp=temp/10;
			brojac_c[i]++;

		} while(temp!=0);

		/*treba ostat jedna cifra pri dijeljenju*/
		brojac_c[i]--;
	}


	for(k=0; k<a; k++) {
		temp1=fabs(nizbr[k]);
		noviBroj=0;

		do {


			cifra=temp1/(pow(10,brojac_c[k]));
			for(j=0; j<b; j++) {
				if(cifra!=niz2[j])
					istina=0;
				else {
					istina=1;
					break;

				}
			}
			/*pretpostavka je da se broj nalazi u drugom nizu */
			if(istina==0)
				noviBroj=noviBroj*10+cifra;


			d=pow(10,brojac_c[k]);
			h=cifra*d;

			temp1 -= h;
			brojac_c[k]--;

		noviNiz[l]+=noviBroj;
		l++;
	}

	for(q=0; q<a; q++) {
		if(nizbr[q]<0) {
			noviNiz[q]*=-1;
		} else noviNiz[q]=noviNiz[q];

	}
  for(k=0;k<a;k++)
  nizbr[k]=noviNiz[k];
  
  return 1;
 }
