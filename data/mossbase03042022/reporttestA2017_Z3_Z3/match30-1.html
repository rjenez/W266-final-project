int brojCifri(int broj)
{
    int duzina = 0;
    broj = abs(broj);
    
    do
    {
        broj /= 10;
        duzina++;
        
    }while(broj > 0);
    
    return duzina;
}

int ponavljanje(int niz[], int n)

}
