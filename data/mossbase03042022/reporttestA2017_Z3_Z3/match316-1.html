#include <stdio.h>
#include <math.h>



int sadrzicifru(int n, int a)
{
	while(n!=0)
	{
		if(n%10==a) return 1;
		n/=10;
	}
	return 0;
}

int izbacicifru(int n, int a)
{
	int suma=0, k=1;
	while(n!=0)
	{
		if(n%10!=a) {
			suma+=(n%10)*k;
			k*=10;
		}
		n/=10;
	}
	return suma;
}
