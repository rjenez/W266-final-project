int ima_li(int broj, int niz[], int vel){

	int tmp=abs(clan_niza);
	int uredjen=0, pom=0; 
	int krajnji =0; 
	while(tmp!=0 ){
		pom = tmp %10; 
		tmp = tmp /10; 
		if((ima_li(pom, nizcifara, vel)) == 1 ) continue; 
		uredjen = uredjen*10 + pom; 
	}
	
	while(uredjen != 0){
	int pom; 
	pom = uredjen %10; 
	uredjen = uredjen / 10; 
	krajnji = krajnji * 10 + pom; 
	}

	return krajnji;
}
