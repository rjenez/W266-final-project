		for(br=i+1; br<vel2; br++) {

			}
		}}

		for(j=0; j<vel1; j++) {

			pom_broj = niz1[j];

				negativan=1;
				pom_broj *=-1;

				while(pom_broj>0) {
					isparcano = pom_broj%10;

					if(isparcano == niz2[k])
					{
						postoji_broj = 1;
					}
				}

				pom_broj/=10;

			}
			if(negativan==1) novi_broj*=-1;
			niz1[j] = novi_broj;
			novi_broj = 0;

		}
		return 1;
	}
