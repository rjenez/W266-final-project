    int izbaciti_iz_niza(int *niz, int vel, int x){
        int i,l;
        for(i=0;i<vel;i++){
            if(niz[i]==x){
                for(l=i;l<vel-1;l++) niz[l]=niz[l+1];
                vel--;
                i--;
            }
        }
        
        return vel;
    }

    int izbaciti(int x, int *cifre, int vel){
        int minus=0;
        int i;
        int k=0;
        int pomniz[10];
        if(x<0) {
            minus=1;
            x=-x;
        }
        if(x==0) return 0;
        
        while(x!=0){
            pomniz[k]=x%10;
            x/=10;
            k++;
        }
        
        for(i=0;i<vel;i++){
            k=izbaciti_iz_niza(pomniz,k,cifre[i]);
        }
        
        for(i=k-1;i>=0;i--){
            x+=pomniz[i]*pow(10,i);
        }
        
        
        if(minus) return -x;
        
        return x;
    }
