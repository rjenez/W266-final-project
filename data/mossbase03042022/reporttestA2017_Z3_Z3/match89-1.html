int provjeri_niz(int* niz, int duzina)
{
	int i,j;
	for(i=0; i<duzina; i++)
	{
		if(niz[i]>9||niz[i]<0)
			return 0;

		for(j=i+1; j<duzina; j++)
		{
			if(niz[i]==niz[j])
				return 0;
		}

		for(j=i-1; j>-1; j--)
		{
			if(niz[i]==niz[j])
				return 0;
		}

	}
	return 1;
}
