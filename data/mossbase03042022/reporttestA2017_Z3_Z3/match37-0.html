int ispravno_dva( int *b,  int velb){
	int er1=0,er2=0,i,j;
	for(i=0;i<velb;i++){
		if(b[i]<0 || b[i]>9) er1++;
	}
	for(i=0;i<velb-1;i++){
		for(j=i+1;j<velb;j++){
			if(b[i]==b[j]) er2++;
		}
	}
	if(er1!=0 || er2!=0  ) return 0;
	else return 1;
}
