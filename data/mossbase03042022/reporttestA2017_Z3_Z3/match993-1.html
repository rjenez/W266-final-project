int izmijeni_broj(int x, int cifra) {
	int novi_broj = 0, stepen_10=1,pomocna;
	pomocna=fabs(x);
	while(pomocna!=0) {
		if(pomocna%10!=cifra) {
			novi_broj=novi_broj+(pomocna%10)*stepen_10;
			stepen_10*=10;
		}
		pomocna/=10;
	}
	if(x>0){
		return novi_broj;
	}
	else {
		return (-novi_broj);
	}
}
