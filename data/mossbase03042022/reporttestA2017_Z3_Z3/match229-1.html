int stepen(int x, int n) {
	
	int i, rez = 1;
	
	for(i = 0; i < n; i++) {
		
		rez *= x;
		
	}
	
	return rez;
	
}

int izbacivanjeCifara(int broj, int cifra)
{
	
	int noviBroj = 0;
	int cifraProvjera;
	int i = 0;
	
	broj = abs(broj);
	
	while(broj > 0) {
		
		cifraProvjera = broj % 10;
		
		if(cifraProvjera != cifra){
			
			noviBroj += cifraProvjera * stepen(10, i);
			i++;
		}
		
		broj /= 10;
	}
	
	return noviBroj;
	
}
