int izbaci_cifre(int niz1[], int n, int niz2[], int m)
{

    for (j=0; j<m; j++) {
        for (i=0; i<n; i++) {
            while (niz1[n]>0) {
                R=niz1[n]%10;
                niz1[n]=niz1[n]/10;
                if(niz2[m]==R) continue;
                p=p+R*t;
                t=t*10;
            }
        }
    }
    if (niz2[m]>9 || niz2[m]<0)
        return 0;
    else
        return 1;
}
