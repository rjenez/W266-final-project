int izbaci_broj(int broj, int x)
{ int pomocna = 0;
	int novi = 0;
	int k=1, c;
	if (broj<0)pomocna = 1;
	broj = fabs(broj);
	while(broj!=0){
		c = broj%10;
		if(c!=x){
			novi = novi + c*k;
			k *=10;
		}
		broj /=10;
	}
	if(pomocna == 1)novi = novi *(-1);


	return novi;
}
