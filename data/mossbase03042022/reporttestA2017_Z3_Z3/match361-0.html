int izbaci_cifre(int a[], int b[], int m){

   int i, d;
   int c[100]={0},x=0,temp;

   for(i=0; i<m; i++){
      while(a[i]){
        x=a[i]%10;
        temp=x;
        x=c[i];
        c[i]=temp;
        a[i]/=10;

      }
    if(c[i]==b[i]){
        c[i]=c[i+1];
        m--;
        d=1;
    }
    else if(a[i]<0 || b[i]<0 || b[i]>9)
      d=0;

   for(int j=i+1; j<m; j++){
    if(a[i]==a[j])
        d=0;
   }}

   return d;

}
