        -DuzinaStranice(B, C)*DuzinaStranice(B, C))/(2*DuzinaStranice(A, C)*DuzinaStranice(A,B)));

        c.second=t.second+faktor*(c.second-t.second);
    }

    void Skaliraj(double faktor) { Skaliraj(DajCentar(), faktor); }
