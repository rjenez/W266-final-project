bool Trougao::DaLiSuJednaki(double x, double y) {
    return std::fabs(x - y) <= EPS * (std::fabs(x) + std::fabs(y));

    return t1.first * (t2.second - t3.second) - t2.first * (t1.second - t3.second) + t3.first * (t1.second - t2.second);
