/*Funkcija za unos stringa*/
void unesi(char s[], int n)
{
	char znak;
	int i=0;
	znak=getchar();
	if(znak=='\n')
		znak=getchar();


	while(i<n-1 && znak!='\n') {
		s[i]=znak;
		i++;
		znak=getchar();
	}
	s[i]='\0';
}

char *rijec(char *d)
{
	char *p;
	p=d;

	while((*d>='A' && *d<='Z') || (*d>='a' && *d<='z')) {
		d++;
	}
	*d='\0';


	return p;
}

int duzina(char *s)
{
	int br=0;
