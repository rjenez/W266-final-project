#include <stdio.h>



int uporedi(char *s1, char *s2, int len) {
	int i = 0;
	
	while(i < len) {
		if (*s1 > 90 && *s2 < 91) {
			if (*s1 != *s2 + 32) return 0;
		} else if (*s1 < 91 && *s2 > 91) {
			if (*s1 + 32 != *s2) return 0;
		} else {
			if (*s1 != *s2) return 0;
		}
		s1++;
		s2++;
		i++;
	}
	
	return 1;
}

/* Ispitujue da li je karakter slovo */
int slovo(char c) {
	if ((c > 64 && c < 91) || (c > 96 && c < 123)) return 1;
	return 0;
}
