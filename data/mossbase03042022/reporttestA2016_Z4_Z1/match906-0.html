void unesi(char niz[], int velicina)
{
    int i=0;
    char znak=getchar();
    if (znak=='\n')
        znak=getchar();
    while(i<velicina-1 && znak!='\n')
    {
        niz[i]=znak;
        i++;
        znak=getchar();
    }
    niz[i]='\0';
}
