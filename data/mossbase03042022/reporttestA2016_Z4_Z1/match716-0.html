#include <stdio.h>
int podstring(char* s, char* poc, char* kraj)
{
	//broj ponavljanja rijeci
	char* s1=s;
	char* s2=poc;
	int brojac=1;
	while(*s1 != '\0') {
		s2=poc;
		while((*s1 == *s2 || *s1-32 == *s2 || *s1+32 == *s2) && s2<kraj) {
			s1++;
			s2++;
		}
		if(*s1 == '\0')
			return brojac;


		if(s2==kraj && !((*s1 >= 'A' && *s1 <= 'Z') || (*s1 >= 'a' && *s1 <='z')))
			brojac++;
		s1++;
	}
	return brojac-1;
}
