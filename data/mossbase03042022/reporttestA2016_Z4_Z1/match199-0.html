char* izbaci_najcescu(char* s)
{

	pocetak=s;
	if(*s==NULL)
		return NULL;
	/*Trazi rijec koja se ponavlja najvi≈°e puta ~ pokazivac na tu rijec kod prvog pojavljivanja*/
	while(*s!='\0')

				{
					br2++;
					break;
				}
			}
			temp++;
		}
		
		
		if(br2>br_temp || prvi_prolazi)
		{
			br_temp=br2;
			while(*pok2!='\0')

				{
					p++;
					q++;


					if((s==pocetak || (*(s-1)<'A' || (*(s-1)>'Z' && *(s-1)<'a') || *(s-1)>'z')) && ((*p<'A' || (*p>'Z' && *p<'a') || *p>'z')) && ((*q<'A' || (*q>'Z' && *q<'a') || *q>'z')) && (*(pok2-1)<'A' || (*(pok2-1)>'Z' && *(pok2-1)<'a') || *(pok2-1)>'z'))

			{

		s=niz[i];

			*niz[i]='\0';
