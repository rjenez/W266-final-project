//Funkcija duz mi daje duzinu stringa.
int duz(char* s){
	int l=0;
	while(*s++!=0)
		l++;
	return l;
}
/*Funkcija pitaj mi ispituje da li je dati char slovo ili ne.
Ako je slovo onda vraca 1, u suprotnom vraca 0.*/
int pitaj(char s){
	if(s>='A'&&s<='Z'||s>='a'&&s<='z')
		return 1;
	else 
	return 0;
}
//Ova funkcija salje mi najveci clan u datom nizu.
