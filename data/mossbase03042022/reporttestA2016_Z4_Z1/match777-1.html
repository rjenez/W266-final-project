void unesi(char niz[], int velicina) {
    int i = 0;

int je_li_slovo(char s){
	
	if ((s >= 'A' && s <= 'Z') || (s >= 'a' && s <= 'z'))
		return 1;
		
	return 0;
}

char u_malo (char p) {
	
	if (p >= 'A' && p <= 'Z')
		p += 'a' - 'A';
	
	return p;	
}

int je_li_ista_rijec (char *s1, char *s2) {
	char *p=s1;
	char *q=s2;
	
	while (u_malo(*p) == u_malo(*q)) {
		
		if (!je_li_slovo(*p) && !je_li_slovo(*q))
			return 1;
		p++;
		q++;
	}
	if (!je_li_slovo(*p) && !je_li_slovo(*q))
			return 1;
	
	return 0;
}
