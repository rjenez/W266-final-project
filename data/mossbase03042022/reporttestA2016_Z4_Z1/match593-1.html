short istoSlovo (char a, char b) {
	if(a>='A' && a<='Z')
		a-='A';
	else if(a>='a' && a<='z')
		a-='a';
	if(b>='A' && b<='Z')
		b-='A';
	else if(b>='a' && b<='z')
		b-='a';
	return a==b;
}
