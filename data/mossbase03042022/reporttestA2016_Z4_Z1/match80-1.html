int slovo(char x) {
	return ((x >= 'a' && x <= 'z') || (x >= 'A' && x <= 'Z'));
}

		if (!(*rijec1 == *rijec2 || *rijec1-32 == *rijec2 || *rijec1 == *rijec2-32)) 
			return 0;
		rijec1++;
		rijec2++;
	}
	if (*rijec1 == 0 && *rijec2 == 0)
		return 1;
	else

}
