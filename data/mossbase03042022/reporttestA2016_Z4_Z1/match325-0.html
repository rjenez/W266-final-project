void unesi( char *s, int vel){
    char znak=getchar();
    if (znak == '\n') znak = getchar();
    int i=0;
    while(i<vel && znak!='\n'){
        *(s+i)=znak;
        i++;
        znak=getchar();
    }
    *(s+i)='\0';
}
