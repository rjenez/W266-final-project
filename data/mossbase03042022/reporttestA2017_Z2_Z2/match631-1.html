{

    printf("Unesite velicinu niza: ");

    printf("Unesite clanove niza: ");
    for(i=0;i<n;i++)
    {
        scanf("%d",&niz1[i]);

    for(i=n-1;i>=0;i--)
    {
        if(niz1[i]==0)
        {
            pomocniniz[brojac]=0;

        }
    }

            broj++;
        }
        else
        {
            niz2[k+1]=broj;
