    printf("Unesite velicinu niza: ");

    scanf("%d",&n);
    }while(n>100 || n<=0);

    for(i=0;i<n;i++){

        if(niz1[i]<0)i--;
    }
    for(i=0;i<n;i++){
        tmp=niz1[i];
        counter=0;
        if(tmp==0){counter=1;tmp=0;}
        else while(tmp!=0){
            counter++;
            tmp/=10;
        }    
        velNiza2+=counter;
        tmp=niz1[i];
        for(j=velNiza2-1;j>=velNiza2-counter;j--){
                niz2[j]=tmp%10;
                tmp/=10;    
        }
    }    
    k=0;


    for(i=0;i<velNiza2;i++){
        counter=1;
        for(j=i+1;j<velNiza2;j++){
            if(niz2[i]==niz2[j])counter++; 
            else break;
        }
        niz3[k]=niz2[i];
        k++;
        niz3[k]=counter;

        i=j-1;
    }
