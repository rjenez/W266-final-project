		b[i]=-1;

	for(i=0; i<d; i++){

	}

		l=j;
		do{
			b1[j]=a[i]%10;
			j++;
			a[i]=a[i]/10;
		}while(a[i]>0);

			b[m]=b1[k];
		}

					break;

			}
