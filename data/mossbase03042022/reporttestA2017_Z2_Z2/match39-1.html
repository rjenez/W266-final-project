	for (i=0; i<10; i++)
	brojaci[i]=1;
	
	do {

    while (n<1 || n>100);
	
	printf ("Unesite clanove niza: ");
	for (i=0; i<n; i++) {
	scanf ("%d", &niz[i]);
	if (niz[i]<0) i--; }
	
	
	/* formiranje niza neprekidnih cifara */
	for (i=0; i<n; i++) {
		
	if (niz[i]==0) {  /* ako je broj = 0 */
		niz2[j]=0;
		j++; n2++; }
		
	else {
		a=niz[i];
		l=0;
	do {
		a/=10;
		l++;
		} while (a>0);
	l-=1;
	a=niz[i];
	do {
		eksp=pow(10,l);
		cifra=a/eksp;
		niz2[j]=cifra;
		j++; n2++;

		l--; }
	while (l>=0);
		}
	}

	j=0;

			brojaci[niz2[i]]++;

		}
		}

		for (k=0; k<10; k++)  /* resetovanje cifara brojaca */
		brojaci[k]=1;
		j++;
		l+=2;
	}

	printf ("Finalni niz glasi:\n");

	if (i==l-1) printf ("%d", C[i]);
