template<typename tip>
bool ProvjeraIspravnosti(tip &ref)
{
	int vel = ref.size();
	if(vel == 0) return false;
	for(int i = 0; i < ref.size(); i++) {
		if(ref.at(i).size() != vel) return false;
		for(int j = 0; j < ref.at(i).size(); j++) {
			if(ref.at(i).at(j).size() != vel) return false;
		}
	}
	return true;
}
