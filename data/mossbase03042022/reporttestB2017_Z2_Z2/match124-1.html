template <typename Tip2D>
Tip2D GD(Tip2D &c){
	int n(c.size());
	Tip2D x(n);
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			if(j>=i && j<n-i) x.at(i).push_back(c.at(i).at(j));
			else if(j>=n-1-i && j<=i) x.at(i).push_back(c.at(i).at(j));
		}
	}
	return x;
}
template <typename Tip2D>
Tip2D DG(Tip2D &c){
	int n(c.size());
	Tip2D x(n);
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			if(j>=i && j<n-i) x.at(i).push_back(c.at(n-1-i).at(j));
			else if(j>=n-1-i && j<=i) x.at(i).push_back(c.at(n-1-i).at(j));
		}
	}
	return x;
}
template <typename Tip2D>
Tip2D LD(Tip2D &c){
	int n(c.size());
	Tip2D x(n);
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			x.at(i).push_back(c.at(j).at(i));
		}
	}
	Tip2D y(GD(x));
	return y;
}
template <typename Tip2D>
Tip2D DL(Tip2D &c){
	int n(c.size());

	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			x.at(i).push_back(c.at(j).at(i));
		}
	}
	Tip2D y(DG(x));
	return y;
}
