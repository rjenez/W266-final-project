int JeLiSlovo (char znak)
{
	return znak != ' ';
	//return ((znak >= 'A' && znak <= 'Z') || (znak >= 'a' && znak <= 'z'));
}
// potrebna duzina broja da znamo za koliko treba prosiriti nas string

int dajDuzinuBroja(int x) {
	int brojCifara = 0;
	while (x) {
		brojCifara++;
		x /= 10;
	}
	return brojCifara;
}
