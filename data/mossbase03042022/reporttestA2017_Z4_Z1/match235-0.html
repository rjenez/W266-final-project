#include <stdio.h>



int broj_znakova(int n){
   int znakova=0;
   if(n<0){
      znakova++;
      n=n*-1;
   }
   while(n!=0){
      n=n/10;
      znakova++;
      
   }
   return znakova;
}

char*napravi_mjesto(char*string, int duzina){
   char*pocetni,*kraj;
   pocetni=string;
   kraj=string;
   while(*kraj!='\0')kraj++;
   kraj = kraj + duzina+1;
   *kraj='\0';
   kraj--;
   while (kraj > pocetni){
      *kraj=*(kraj-duzina-1);
      kraj--;
   }
   return string;
}
