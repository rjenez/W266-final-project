#include <stdio.h>
void unesi (char niz[], int velicina){	/*Funkcija za unos teksta*/
	char znak=getchar();
	if(znak=='\n') znak=getchar();
	int i=0;
	while (i<velicina && znak!='\n'){


		niz[i]=znak;
		i++;
		znak=getchar();
	}
	niz[i]='\0';
}
