#include <stdio.h>



int broj_cifara(int broj){
	int brojac=0;
	if(broj==0) return 1;
	if(broj<0) broj*=-1;
	while(broj!=0){
		brojac++;
		broj/=10;
	}
	return brojac;
}

int obrni(int broj){
	int c=0;
	while(broj!=0){
		c=c*10+broj%10;
		broj/=10;
	}
	return c;
}

int nula(int broj){
    if(broj%10==0) return 1;
    else return 0;
}
