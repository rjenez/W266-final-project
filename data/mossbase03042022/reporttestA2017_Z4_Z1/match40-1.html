int DaLiImaNula(int pocBr,int broj)

	}

	}

int duzina(int br)
{
	int suma=0;
	if(br<0) suma++;
	while(br!=0) {
		suma++;
		br/=10;
	}
	return suma;
}

int okreni( int br)
{
	int broj=0;
	while(br!=0) {
		broj=(broj+(br%10))*10;
		br/=10;
	}
	broj/=10;
	return broj;
}
