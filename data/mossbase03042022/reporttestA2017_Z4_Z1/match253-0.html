void pomjeri(char* tekst,int broj)
{
	char *p=tekst;
	if(*tekst==' ') broj--;
	while(*p!='\0')
		p++;
	*(p+broj)='\0';
	while(p>=tekst) {
		*(p+broj)=*p;
		p--;
	}
}
