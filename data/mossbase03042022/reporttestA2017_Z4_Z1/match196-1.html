char* razdvoji_cifre(char* rez, int broj)
{
	char *p=rez;
	int zadnja_cifra=0;
	char pom[10];
	char *x=pom;


	*p=' ';
	p++;


	if (broj<0) {
		*p='-';
		p++;
		broj=-broj;
	}


	while(broj>0) {
		zadnja_cifra=broj%10;
		*x=(zadnja_cifra+'0');
		x++;
		broj=broj/10;
	}
	x--;

	while(x>=pom) {
		*p = *x;
		p++;
		x--;
	}

	*p=' ';
	p++;

	*p='\0';


	return rez;
}

int duzina(char* tekst)
{

		d++;

	}
	return d;
}
