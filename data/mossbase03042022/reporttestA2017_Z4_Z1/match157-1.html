#include <stdio.h>



 int duzina_broja(int b) {
    int r;
    
    r = 0;
    if (b <= 0) {
        r++;
        b = -b;
    }
    
    while (b > 0) {
        r++;
        b /= 10;
    }

    return r;
} 

char* upis_broja(char* s, int b) {
    char* t = s;
    int duz = duzina_broja(b);
    
    if (b < 0) {
        *t++ = '-';
        duz--;
        b = -b;
    }
    else if (b == 0)
        *t = '0';
    
    while (b > 0) {
        *(t + duz - 1) = (b % 10) + '0';
        duz--; 
        b /= 10;
    }

    return s;
}

void unesi(char niz[], int velicina) {
    int i;
    
    char znak = getchar();
    if (znak == '\n') znak=getchar();
    
    i = 0;
    while (i < velicina-1 && znak != '\n') {
        niz[i] = znak;
        i++;
        znak = getchar();
    }
    niz[i]='\0';
}
