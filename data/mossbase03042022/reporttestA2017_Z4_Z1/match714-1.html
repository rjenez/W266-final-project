#include <stdio.h>

void unesi(char *s, int duzina)
{
	int znak = getchar();
	if(znak == '\n') znak = getchar();
	int i = 0;


	while(i < duzina - 1 && znak != '\n') {
		s[i] = znak;
		i++;
		znak = getchar();
	}
	s[i] = '\0';
}

int broj_rijeci(const char *s)
{
	int brojac = 0, razmak = 1;
	const char *p = s;



	while(*p != '\0') {
		if(*p == ' ')
			razmak = 1;
		else if(razmak) {
			razmak = 0;
			brojac++;
		}
		p++;
	}
	return brojac;
}
