        pomocni=broj;

            broj_mjesta++;

        broj_mjesta++;

    else if(broj>0){
        pomocni=broj;

            pomocni=pomocni/10;
            broj_mjesta++;
        }
    }
    pomocni=broj;

            if(*kraj=='\0') break;

                kraj++;
            }

                *(kraj+broj_mjesta)=*kraj;
                kraj--;
            }
            tekst+=broj_mjesta;
            while(pomocni!=0){
                if(pomocni<0) pomocni*=-1;

            else tekst+=broj_mjesta+1;

                while(kraj>=tekst){
                    *(kraj+1)=*kraj;
                    kraj--;
                }
                *tekst=' ';
            }
        }
            while(*tekst==' ') tekst++; 
            tekst++;
    }
    return pocetak;
}
