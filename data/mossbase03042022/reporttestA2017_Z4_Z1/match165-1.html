#include <stdio.h>




int PrebrojiCifre(int n)
{    
	int br=0;
	if(n<0){ n*=-1;}
	while(n!=0)
	{
		n/=10;
		br++;
	}
	return br;
}

int ObrniBroj(int n)
{   
	int broj=0;
	if(n<0) n*=-1;
	while(n!=0)
	{
		broj=broj*10+n%10;
		n/=10;
	}
	return broj;
}


char* ProsiriString(char *pom,int n)
{	
	
    char *temp=pom;
	pom++;
    while(*temp!='\0') temp++;
    temp+=n-1;
    
  	
    while(temp!=pom)
    {
    	*temp=*(temp-n);
    	temp--;
    }
  
	return pom;
}

char Umanji(char k)
{
	if(k>='A' && k<='Z') k+=32;
	
	return k;
}
