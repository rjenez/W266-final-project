int daj_broj(int n)
{
    int broj_cifara = 0;

    if (n == 0) return 1;
    else if(n < 0) {
        broj_cifara++;
        n *= -1;
    }
    while(n != 0) {
        broj_cifara++;
        n /= 10;
    }
    return broj_cifara; //zbog razmaka iza -.-
}
