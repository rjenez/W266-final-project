#include <stdio.h>

void unesi(char *tekst, int velicina)
{
	int i=0;
	char c;
	do {
		c=getchar();
		tekst[i]=c;
		i++;
	} while(c!='\n' && i<velicina);
	tekst[i-1]='\0';
}
