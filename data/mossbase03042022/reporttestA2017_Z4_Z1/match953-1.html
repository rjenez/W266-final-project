#include <stdio.h>

void unesi(char niz[], int velicina)
{
	char znak = getchar();
	if (znak == '\n') znak = getchar();
	int i = 0;


	while (i < velicina-1 && znak != '\n') {
		niz[i] = znak;
		i++;
		znak = getchar();
	}
	niz[i] = '\0';
}


int duzina(char *s)
{
	int br=0;
	while(*s!='\0') {
		br++;
		s++;
	}
	return br;
}

int aps(int n)
{
	/* vraca apsolutnu vrijednost parametra*/
	if(n<0)
		n-=2*n;
	return n;
}

int cifra(int n)
{
	/*vraca broj cifara parametra*/
	int i,br=n;
	for(i=0; br!=0; i++)
		br/=10;
	if(n<0)
		i++;
	return i;
}

	if(*tekst==' ') {
		while(*(tekst+i)==' ')

			for(j=i; *(tekst+j)!='\0'; j++)
				if(*(tekst+j)!=' ')
					break;
					/*ako ne, prekida dalju manipulaciju stringa*/

				razmak=1;
				/*pomjeranje znakova za broj cifara udesno*/
			for(j=duz+m+razmak; j>i+m-1-razmak; j--)
				*(tekst+j)=*(tekst+j-m-razmak);
