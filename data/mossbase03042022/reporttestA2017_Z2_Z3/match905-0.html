// An Inplace function to rotate a N x N matrix
// by 90 degrees in anti-clockwise direction
void okreni_redove(int mat[][N], int broj_redova, int broj_kolona) {
    int i = 0;
    for (; i < broj_redova; ++i) {
        int j = 0;
        int k = broj_kolona - 1;
        for (; j < k; ++j) {
            int tmp = mat[i][j];
            mat[i][j] = mat[i][k];
            mat[i][k] = tmp;
            --k;
        }
    }
}




void okreni_kolone(int mat[][N], int broj_redova, int broj_kolona) {
    int i = 0;
    for (; i < broj_kolona; ++i) {
        int j = 0;
        int k = broj_redova - 1;
        for (; j < k; ++j) {
            int tmp = mat[j][i];
            mat[j][i] = mat[k][i];
            mat[k][i] = tmp;
            --k;
        }
    }
}
