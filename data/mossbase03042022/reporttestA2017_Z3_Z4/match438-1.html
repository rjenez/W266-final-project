#include <stdio.h>
#include <stdlib.h>




int max(int matrica[100][100], int visina, int sirina){
	int i, j, p, q, broj, maxbroj=matrica[0][0], brojac=0, maxbrojac=0;
	for(i=0; i<visina; i++){
		for(j=0; j<sirina; j++){
			broj=matrica[i][j];
			for(p=0; p<visina; p++){
				for(q=0; q<sirina; q++){
					if(matrica[p][q]==broj) brojac++;
				}
			}
			if(brojac==maxbrojac){
				if(broj<maxbroj) maxbroj=broj;
				brojac=0;
			}
			if(brojac>maxbrojac){
				maxbrojac=brojac;
				maxbroj=broj;
				brojac=0;
			}
			brojac=0;
		}
	}
	return maxbroj;
}

int izbaci_kolone(int matrica[100][100], int visina, int sirina, int n){
	int i, j, p, q;
	for(i=0; i<visina; i++){
		for(j=0; j<sirina; j++){
			if(matrica[i][j]==n){
				for(p=0; p<visina; p++){
					for(q=j; q<sirina-1; q++){


						matrica[p][q]=matrica[p][q+1];
					}
				}
				sirina--;
				j--;
			}
		}
	}
	return sirina;
}
