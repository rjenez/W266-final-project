#include <stdio.h>


int max(int matrica[100][100],int visina,int sirina){
	int i,j,k,l,br,br1,br11=1;
	br1=matrica[0][0];


	for(i=0;i<visina;i++){
		for(j=0;j<sirina;j++){
			br=1;
			for(k=i;k<visina;k++){
				for(l=0;l<sirina;l++){
					if(k!=i || l>j){
						if(matrica[i][j]==matrica[k][l]){
							br++;
						}
					}
				}
			}
			if(br>br11){
				br11=br;
				br1=matrica[i][j];
			}
			if(br==br11){
				if(matrica[i][j]<br1) br1=matrica[i][j];
			}
		}
	}
	return br1;
}
int izbaci_kolone(int matrica[100][100],int visina,int sirina,int n){
	int i,j,k,l;
	for(i=0;i<visina;i++){


		for(j=0;j<sirina;j++){
			if(matrica[i][j]==n){
				for(k=i;k<visina;k++){
					for(l=j;l<sirina-1;l++){
						matrica[k][l]=matrica[k][l+1];
					}
				}
				sirina--;
				i=0;
				j--;
			}
		}
	}
	return sirina;
}
