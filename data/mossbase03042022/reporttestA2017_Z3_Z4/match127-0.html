#include <stdio.h>



void sortiraj(int *niz, int n){
	int i, j,temp;
	for(i = 0; i < (n-1); i++){
		for(j = 0; j < (n-1); j++){
			if(*(niz + j) > *(niz + j + 1)){
				temp = *(niz + j + 1);
				*(niz + j + 1) = *(niz + j);
				*(niz + j) = temp;
			}
		}
	}
}

int izbaci_kolone(int niz[100][100], int visina, int sirina, int x){
	int i, j, k, l;
	for(i = 0; i < visina; i++){


		for(j = 0; j < sirina; j++){
			/*BRISANJE KOLONE*/
			/*U SUŠTINI NIJE BRISANJE NEGO PREPISIVANJE DRUGIH ČLANOVA PREKO NJE I SMANJIVANJE VELIČINE*/
			if(niz[i][j] == x){
				
				for(k = j; k < sirina - 1; k++){
					for(l = 0; l < visina; l++){
						niz[l][k] = niz[l][k + 1];
					}
				}
				sirina--;
				//i--;
				j--;
			}
		}
	}
	return sirina;
}

	for(i = 0; i < n - 1; i++){

				i++;
