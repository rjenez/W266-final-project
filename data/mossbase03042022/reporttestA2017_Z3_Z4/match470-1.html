int max (int matrica[100][100], int visina, int sirina)

{
    int kolona[100],z;
    int min;


    int i,jednaki;
    for (i=0; i<visina; i++)
        kolona[i] = sirina-1;

    min=0;


    while (kolona[min] >= 0) {
        for (i=0; i<visina; i++) {
            if (matrica[i][kolona[i]] < matrica[min][kolona[min]] )
                min = i;
        }


        jednaki=0;


        for (i=0; i<visina; i++) {

            if (matrica[i][kolona[i]] > matrica[min][kolona[min]]) {
                if (kolona[i] == 0) {
                    z=matrica[min][kolona[min]];
                    return z;
                }

                kolona[i]--;
            } else {
                jednaki++;
            }
        }

        if (jednaki==visina) {
            z=matrica[min][kolona[min]];
            return z;
        }


    }

}
