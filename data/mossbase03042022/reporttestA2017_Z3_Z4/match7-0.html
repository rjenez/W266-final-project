#include <stdio.h>


void sort(int niz[],int k)
{
	int i,j,pom;
	for(i=0; i<k; i++) {
		for(j=i; j<k; j++) {
			if(niz[j]<niz[i]) {
				pom=niz[i];
				niz[i]=niz[j];
				niz[j]=pom;
			}
		}
	}
}
int max(int matrica[100][100],int visina,int sirina)
{
	int i,j,k=0,niz[100];
	int maxi=0,c=1,broj=0;
	for(i=0; i<visina; i++) {
		for(j=0; j<sirina; j++) {
			niz[k]=matrica[i][j];
			k++;
		}
	}
	sort(niz,k);
	for(i=0; i<k; i++) {
		for(j=i+1; j<k; j++) {
			if(niz[j]==niz[i])
				c++;
			else {
				break;
			}
		}
		if(c>maxi) {
			maxi=c;
			broj=niz[i];
		}
		c=1;
	}
	return broj;
}

				}
			}
		}
	}
	for(i=0; i<visina; i++) {

			for(b=0; b<k; b++) {
				if(j==niz1[b]) {
					p=0;
				}
			}
