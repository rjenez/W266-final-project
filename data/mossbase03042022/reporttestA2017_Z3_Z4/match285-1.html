#include <stdio.h>

int max(int matrica[100][100],int visina,int sirina)
{
	int br,brojac=0,temp,i,j,ponavljanje=0,q,m;
	int max=0;
	int prvi=matrica[0][0];


	for(i=0; i<visina; i++) {

		for(j=0; j<sirina; j++) {

			br=matrica[i][j];

			for(q=0; q<visina; q++)
				for(m=0; m<sirina; m++) {
					ponavljanje=matrica[q][m];
					if(br==ponavljanje)
						brojac++;
				}

			if(brojac>=max) {
				temp=prvi;
				prvi=br;
				if(brojac==max && br>temp)
					prvi=temp;

				max=brojac;



			}
			brojac=0;


		}
	}

	return prvi;

}
int izbaci_kolone (int matrica[100][100],int visina,int sirina,int N)
{
	int v,s,i,k,j,w;


	v=visina;
	s=sirina;

	for(i=0; i<v; i++)
		for(j=0; j<s; j++) {
			if(matrica[i][j]==N) {
				for(k=j; k<s; k++)
					for(w=0; w<v; w++)
						matrica[w][k]=matrica[w][k+1];
				s--;
			}
		}




	return s;

}
