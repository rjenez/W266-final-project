#include <stdio.h>


int max(int matrica[100][100], int visina, int sirina){
	int i,j,k,n,broj1,broj2,brojac=0,max=0,max1=matrica[0][0],temp;
	for(i=0; i<visina; i++) {
		for(j=0; j<sirina; j++) {
			broj1=matrica[i][j];
			for(k=0; k<visina; k++) {
				for(n=0; n<sirina; n++) {
					broj2=matrica[k][n];
					if(broj1==broj2) {
						brojac++;}}}
			if(brojac>=max) {
				temp=max1;
				max1=broj1;
				if(brojac==max && broj1>temp)
					max1=temp;
				max=brojac;}
			brojac=0;}}
	return max1;
}
int izbaci_kolone (int matrica[100][100], int visina, int sirina, int N){
	int i,j,k,n,s;
	s=sirina;
	for (i=0; i<visina; i++)
		for (j=0; j<s; j++) {
			if (N==matrica[i][j]) {
				for (n=j; n<sirina; n++)
					for (k=0; k<visina; k++)
						matrica[k][n]=matrica[k][n+1];
				s--;}}
	return s;
}
