int izbaci_kolone(int matrica[100][100], int visina, int sirina)
{
	int elemenat=0;
	int brojac=0,i, j;
	for(i=0; i<visina; i++) {
		for(j=0; j<sirina; j++) {
			if (matrica[i][j]==elemenat)
				brojac=elemenat;
				
			matrica[i][j]=matrica[i-1][j];
		}
	}
	return brojac;
}



int max(int matrica[100][100], int visina, int sirina)
{

	max=matrica[0][0];

	for(i=0; i<visina; i++)

		for(j=0; j<sirina; j++)
		{
			brojac=0;

			for(k=0; k<visina; k++)
			{
				for(g=0; g<sirina; g++)
				{
					if(matrica[i][j]==matrica[k][g]) 
					brojac++;
				}
			}
			if(brojac>brojac1 || (brojac==brojac1 && broj<=max))
			{
				brojac1=brojac;
				max=broj;
			}
		}
	}
	
	
	return max;
}
