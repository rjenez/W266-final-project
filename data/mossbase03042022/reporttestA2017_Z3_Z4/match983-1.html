int  UkloniDuple(int *niz,int size){

    for(i=0;i<size;i++){

            if(niz[j]==niz[i]){
                for(k=j;k<size-1;k++){
                    niz[k]=niz[k+1];
                }
                size--;

            j++;
        }
    }
 return  size;
}

void BubbleSort(int a[], int array_size)
 {
 int i, j, temp;
 for (i = 0; i < (array_size - 1); ++i)
 {
      for (j = 0; j < array_size - 1 - i; ++j )
      {
           if (a[j] > a[j+1])
           {
                temp = a[j+1];
                a[j+1] = a[j];
                a[j] = temp;
           }
      }
 }
 }
