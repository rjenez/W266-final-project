#include <stdio.h>
#include <math.h>



int cifre(int niz[],int vel,int niz1[100])
{
    int i,j,ima,temp,vel1=0;
    for(i=0; i<vel; i++) {
        temp=niz[i];
        while(temp!=0) {
            ima=0;
            for(j=0; j<vel1; j++) {
                if(temp%10==niz1[j]) {
                    ima=1;
                    break;
                }
            }
            if(ima==0) {
                niz1[vel1]=temp%10;
                vel1++;
            }

            temp/=10;
        }
    }
    return vel1;
}
