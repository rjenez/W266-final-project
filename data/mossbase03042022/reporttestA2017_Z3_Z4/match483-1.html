    int i,j;
    for(i=0;i<visina;i++){
        for(j=0;j<sirina;j++){
            if(matrica[i][j]==N){
                int p,d;
                for(p=0;p<visina;p++){
                    for(d=0;d<sirina;d++){
                        matrica[p][d]=matrica[p][d+1];
                    }
                }sirina--;
            }
        }
    } return sirina;
}
