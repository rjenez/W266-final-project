#include <stdio.h>

int max(int matrica[100][100],int visina,int sirina){
	int i,j,brojac[101]={0},maks;
	for(i=0;i<visina;i++){
		for(j=0;j<=100;j++){
			if(matrica[i]==j)brojac[j]++;
		}
		
	}
	
	maks=0;
	for(i=0;i<=100;i++){
		if(brojac[i]>brojac[maks]) maks=i;
	}
	
	
	return maks;
}
