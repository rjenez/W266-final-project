#include <stdio.h>

int max(int matrica[100][100], int visina, int sirina) {

	int br[1000]={0};
	int max=0; int i,j;
	int maxi=matrica[0][0];
	int broj;
	
	for(i=0; i<visina; i++)
	{
		for(j=0;j<sirina; j++) {
				br[matrica[i][j]]++;
				if(matrica[i][j]>maxi)maxi=matrica[i][j];
		}
	}
	
	broj=maxi;
	for(i=0;i<=maxi;i++) 
		 if(br[i]>max  && br[i]!=0)
		 { 
		 	max=br[i]; 
		 	broj=i;
		 	
		 }
		
		
	return broj;

}

int izbaci_kolone(int mat[100][100],int visina,int sirina, int N) {
	
	int i,j,p,q;
	
	for(i=0; i<visina; i++)  {
		for(j=0; j<sirina; j++) {
				if(mat[i][j]==N) {
					for(p=0;p<visina;p++)
						for(q=j;q<sirina-1;q++)


							mat[p][q]=mat[p][q+1];
					
					sirina--;
					j--;
					
				}
					
		}
	}
	return sirina;	
}
