#include <stdio.h>

int max (int mat[100][100], int visina, int sirina)
{
    int i, j, k, l, brojac1 = 0, brojac2 = 0;
    int max = mat[0][0];



    for(i = 0; i < visina; i++) {
        for(j = 0; j < sirina; j++) {
            brojac1 = 0;
            for(k = 0; k < visina; k++) {
                for(l = 0; l < sirina; l++) {
                    if(mat[i][j] == mat[k][l])
                        brojac1++;
                }
            }
            if(brojac1 == brojac2) {
                if(mat[i][j] < max)
                    max = mat[i][j];
            } else if(brojac1 > brojac2) {
                max = mat[i][j];
                brojac2 = brojac1;
            }
        }
    }
    return max;
}

int izbaci_kolone(int mat[100][100], int visina, int sirina, int N)
{
    int i, j, k, l, tmp;
    for(i = 0; i < visina; i++) {
        for( j = 0; j < sirina; j++) {


            if(mat[i][j] == N) {
                for(k = 0; k < visina; k++) {
                    for(l = j; l < sirina - 1; l++) {
                        tmp = mat[k][l];
                        mat[k][l] = mat[k][l + 1];
                        mat[k][l + 1] = tmp;
                    }
                }
                sirina--;
                j--;
            }
        }
    }
    return sirina;
}
