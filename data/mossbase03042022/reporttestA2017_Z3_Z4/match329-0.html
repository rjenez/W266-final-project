#include <stdio.h>



int max(int mat[100][100], int visina, int sirina){
    int dimenz, i, j;
    int el;
    int counter=0;
    int niz[10000];
    int tempel, tempcount;
    dimenz=sirina*visina;
    
    /* Matrix to array */
    
    for (i=0; i<visina; i++){
        for(j=0; j<sirina; j++){
            niz[sirina*i+j]=mat[i][j];
        }
    }
    
    for (i=0; i<dimenz; i++){
        tempel=niz[i];
        tempcount=0;
        for(j=0; j<dimenz; j++){
            if(niz[j]==tempel)
                tempcount++;}
                
                if(tempcount>counter){
                    el=tempel;
                    counter=tempcount;
                }
                else if(tempcount==counter){
                    if(el>tempel)
                    el=tempel;}
        }
        return el;
    }
int izbaci_kolone (int mat[100][100],int visina, int sirina, int max){
    int j, i, a, b;
    for(i=0; i<visina; i++){


        for(j=0; j<sirina; j++){
            if(mat[i][j]==max){
                for(a=j; a<sirina; a++){
                    for(b=i; b<visina; b++){
                        mat[b][a]=mat[b][a+1];
                    }
                }
                sirina=sirina-1;
                j=j-1;
            }
        }
    }
    return sirina;
}
