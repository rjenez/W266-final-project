#include <stdio.h>
int izbaci_kolone(int matrica[100][100], int visina, int sirina, int N)
{
	int i, j, a, b;
	for(i=0; i<visina; i++) {


		for(j=0; j<sirina; j++) {
			if(matrica[i][j]==N) {
				for(a=j; a<sirina; a++)
					for(b=0; b<visina; b++)
						matrica[b][a]=matrica[b][a+1];
				sirina--;
			}
		}
	}
	return sirina;
}


int max(int matrica[100][100], int visina, int sirina)
{
	int i,j,a,b;
	int max = 0, brojac = 0, poredi, temp, broj=0, ponoviti=1;
	int maks=matrica[0][0];



	for(i=0; i<visina; i++) {
		for(j=0; j<sirina; j++) {
			broj=matrica[i][j];
			for(a=0; a<visina; a++) {
				for(b=0; b<sirina; b++) {

					poredi=matrica[a][b];
					if(broj==poredi) {
						brojac++;
					}
				}
			}
			if(brojac>=max) {
				temp=maks;
				maks=broj;
				if(brojac==max && broj>temp)
					maks=temp;
				max=brojac;
			}
			brojac=0;
		}
	}
	return maks;
}
