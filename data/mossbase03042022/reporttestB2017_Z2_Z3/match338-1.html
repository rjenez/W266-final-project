bool DaLiJePeriodicno(std::vector<int>a)
{
	bool periodican(false);

	
		for(int j=1;j<a.size();j++)
		{
			if(a[j]==a[0])
			{
				int k=j;
				while(a[k]==a[k-j] && k!=a.size()-1)
				k++;
				if(k==a.size()-1)
				periodican=true;
			}
			
		}
	return periodican;
}
