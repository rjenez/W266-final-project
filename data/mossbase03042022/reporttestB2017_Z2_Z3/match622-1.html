bool TestPerioda (Vektor v, int p){
	
	if(p<=0)return false;
	
	if(int(v.size())-p<=0)return false;
	
	for(int i=0;i<v.size()-p;i++){
		if(v.at(i+p)!=v.at(i))return false;
	}
	
	return true;
}

	for(int i=0;i<int(v.size());i++){
		if(TestPerioda(v,i)) return i;
	}
	
	return 0;
}
