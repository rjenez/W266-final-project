template <typename IntTip>
int Periodicnost(const std::vector<IntTip> &v)
{
	int p(0), period(0);
	for(;;) {
		p++;
		if(p>=v.size()) break;
		bool b(true);
		for(int i=0; i<v.size()-p; i++) {
			if(v.at(i)!=v.at(i+p)) {
				b=false;
				break;
			}
		}
		if(b) {
			period=p;
			break;
		}
	}
	return period;
}
