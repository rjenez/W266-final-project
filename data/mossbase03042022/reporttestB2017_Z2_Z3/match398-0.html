template<typename tip1>
bool TestPerioda(const std::vector<tip1> &v, int broj) {
	int n; 
	if(v.size()==broj || broj<=1) return false; 
	for(int i=0; i<v.size(); i++)
	{
		n=i+broj; 
		if(n<v.size()) {
			if(v[i]!=v[n]) return false; 
		}
	}
	
	return true;
}
