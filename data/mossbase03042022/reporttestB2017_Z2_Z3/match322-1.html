bool JelPeriodicna(vector<int>v,int &period)
{    
    
	for(int i=0;i<v.size();i++)
	{	
		int temp=1;
		for(int j=i+1;j<v.size();j++)
			if(v[i]!=v[j]) temp++;
			else {period=temp;break;}
		
		for(int j=i;j<v.size();j+=period)
			if(v[i]!=v[j])return false;

	}
	
	return true;
}
