template<typename Tip>
int periodican(const std::vector<Tip>&v){
	for(int i(1);i<=v.size()-1;i++){
		bool periodican=true;
		
		for(int j(0);j<v.size()-1;j++){
			for(int k(j+i);k<v.size();k+=i){
			if(v.at(j)!=v.at(k))
			periodican=false;
			}
		}
		if(periodican==true) return i;
	}
	return-1;
}
