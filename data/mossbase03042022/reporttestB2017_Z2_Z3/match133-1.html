bool Periodican(const std::vector<int> &v, int per) {
	for (int i=0; i<v.size()-per; i++) {
		if (v.at(i)!=v.at(i+per)) return false;
	}
	return true;
}

int max(const std::vector<int> &v) {
	int maks(v.at(0));
	for (int i=1; i<v.size(); i++) {
		if (v.at(i)>maks) maks=v.at(i);
	}
	return maks;
}
