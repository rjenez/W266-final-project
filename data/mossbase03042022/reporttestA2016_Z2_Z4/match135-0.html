#define E 0.00001

	double a[100][100];
	do{
	printf("\nUnesite M i N: ");

			scanf("%lf",&a[i][j]);
		}
	}



	for(i=0;i<m-1;i++){

	    	p=0;
	    	
	    }
		else{p=1;goto mm;}
		}}
	    	for(i=0;i<m-1;i++){
		
	    if(fabs(a[i][n-1]-a[i+1][0])<E){
	    	w=0;
		}else{w=1;goto mm;}}

}
