long double mat[broj_el][broj_el];

	{
		printf("Unesite M i N: ");

		}
	}

	for(i=0;i<M-1;i++)
	{


		for(j=0;j<N-1;j++)
		{
			if(mat[i][j]!=mat[i+1][j+1])
			
			{	teplicova=0; break; }
			
			
		}
	}
	if(teplicova)
	{


		for(i=0;i<M-1;i++)
		{
			for(j=0;j<N-1;j++)
			{
				if(mat[i][N-1]!=mat[i+1][0])
