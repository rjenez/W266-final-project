	double mat[100][100];
	do

	printf ("Unesite M i N: ");

	while (m<1 || n<1 || m>100 || n>100)

			scanf ("%lf", &mat[i][j]);

			}

			{
				if (mat[i][j]!=mat[i-1][j-1])
				{
					cirkularna=0;
				}
			}
		}			
	
	for (i=1; i<m; i++)
