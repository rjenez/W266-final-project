int BrojProstihFaktora(int broj)
{
	int broj_prostih_faktora(0);
	for(int i = 2; i < broj/i; i++) {
		while(broj % i == 0) {
			broj_prostih_faktora++;
			broj = broj / 2;
		}
	}
	if(broj > 1) {
		broj_prostih_faktora++;
	}
	return broj_prostih_faktora;
}
