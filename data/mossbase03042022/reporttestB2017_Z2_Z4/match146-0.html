bool jeProst(int n) {
	
	if(n<=1)
		return false;
	for(int i=2;i<=sqrt(n);i++) {
		if(n%i==0) 
			return false;
	}
	return true;
}

int brojProstihFaktora(int x) {
	int brojac=0;
	while(true) {
		for(int i=2;i<=x;i++) {
			if(x%i==0 && jeProst(i)) {
				brojac++;
				x/=i;
				break;
			}
		}
		if(x<=1)
			break;
	}
	return brojac;
}
bool viseProstihFaktora(int x,int y) {
	if(brojProstihFaktora(x)==brojProstihFaktora(y))
		return x>y;
	return brojProstihFaktora(x)>brojProstihFaktora(y);
}
