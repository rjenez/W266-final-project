template <typename ItTip>
int BrojElemenata(ItTip it1, ItTip it2) {
    int br_el(0);

bool DaLiJeProst(int a) {
    if (a < 2) return false;
    for (int i = 2; i <= int(sqrt(a)); i++)
        if (a % i == 0) return false;
    return true;
}

        if (DaLiJeProst(i) && x % i == 0) {

        }
    return br;
}
