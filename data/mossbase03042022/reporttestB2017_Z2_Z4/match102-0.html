bool DaLiJeProst(int x){

}

int broj_prostih(int x){
	int brojac = 0;
	if(x == 1 || x == 0) return 1;
	x = abs(x);
	
	int i(2);
	while(x){
		if(i > x) return brojac;
		if(DaLiJeProst(i) && x % i == 0){
			brojac++;
			x /= i;
			i = 1;
		}
	
		i++;
	}
	return brojac;
}
