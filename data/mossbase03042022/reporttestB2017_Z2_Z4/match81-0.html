bool JeLiProst(int x) {
	if(x<0) x*=-1;
	if(x==0 || x==1) return false;
	if(x>2) {
		for(int i=2; i<=std::sqrt(x); i++)
		if(x%i==0) return false;
	}
	return true;
}

	int tmp1(std::abs(x));
