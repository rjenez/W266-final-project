int prost(int n){
	for(int i=2;i<=sqrt(n);i++){
		if(n%i==0) return 0;
	}
	if(!n) return 1;
	return n;
}
int broj_prostih_faktora(int n){
	int brojac(0);
	for(int i=2;i<=n;i++){
		if(n%i==0 && prost(i)){
			n/=i;
			i--;
			brojac++;
		} 
	}
	return brojac;
}
