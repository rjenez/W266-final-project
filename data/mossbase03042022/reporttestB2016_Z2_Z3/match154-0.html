int SumaDjelilaca (long long int x) {
	int suma (0);
	for (int i=1; i<=abs(x); i++) {
		if (x%i==0) suma+=i;
	}
	//std::cout << suma;
	return suma;
}

int BrojProstihFaktora (long long int  x) {
	int brfakt (0);
	for (int i=2; i<=abs(x); i++) {
