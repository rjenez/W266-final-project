int SumaDjelilaca (long long int n){
	int suma(0);
	if(n<0)n=n*(-1);

		if(n%i==0) suma=suma+i;
	}
	suma=suma+n;
	return suma;
}

int SumaCifara(long long int n){

			}
		}
		if(ponavljanje){
			i--;
			continue;
		}
		kont.push_back(t);
	}

}
