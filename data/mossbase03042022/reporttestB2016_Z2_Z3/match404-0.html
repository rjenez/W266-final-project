int SumaDjelilaca (long long int x) {
	int suma(0);
	for(int i=1; i<=std::abs(x); i++) {
		if(x%i==0) suma+=i;
	}
	return suma;
}

int SumaCifara(long long int x) {
	int suma(0);
	for(;;) {
		int cifra;
		cifra=std::abs(x%10);
		suma+=cifra;
		x/=10;
		if(x==0) break;
	}
	return suma;
}
