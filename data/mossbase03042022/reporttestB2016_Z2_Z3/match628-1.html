int SumaCifara(long long int n) {
    int suma(0);
    while(n!=0) {
        suma+=n%10;
        n/=10;
    }
    return suma;
}

int SumaDjelilaca(long long int n) {
    int suma(0);
    if(n<0) n=-n;
    for(int i(1); i<=n; i++)
        if(n%i==0)
            suma+=i;
    return suma;
}

int BrojProstihFaktora(long long int n) {
    int broj(0);
    for(int i(2); i<=n; i++)
        if(n%i==0) {
            broj++;
            n/=i;
            i=1;
        }
    return broj;
}

int BrojSavrsenihDjelilaca(long long int n) {
    int broj(0);
    for(int i(n); i>1; i--)
        if(n%i==0) {
            int suma(0);
            for(int j(1); j<i; j++)
                if(i%j==0) {
                    suma+=j;
                }
            if(suma==i) {
                broj++;
                n/=i;
                i=n+1;
            }
        }
    return broj;
}
