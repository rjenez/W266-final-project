int BrojProstihFaktora(long long int n){
	int umnoz(1), brojac(0);
	int broj(n);
	for(int i=2; i<=broj; i++){
		if(broj%i==0){
			umnoz*=i; brojac++;

		}
		if(umnoz==n)break;
	}

}
int SumaDjelilaca(long long int n){
	int sumadjel(0);
	for(int i=1; i<=abs(n); i++){
		if(abs(n)%i==0) sumadjel+=i;
	}
	return sumadjel;
}

}
