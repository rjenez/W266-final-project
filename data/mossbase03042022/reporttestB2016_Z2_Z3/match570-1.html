int SumaDjelilaca(long long int n)
{
	int suma(0);
	if(n < 0)
	{
		for(long long int i = -1; i >= n; i--)
			if(n % i == 0) suma += std::abs(i);
	}
	for(long long int i = 1; i <= n; i++)
		if(n % i  == 0) suma += i;
	return suma;
}
