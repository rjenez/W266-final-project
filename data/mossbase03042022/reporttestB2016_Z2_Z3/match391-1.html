bool JeLiProst(long long int x) {
	double korijen(std::sqrt(std::abs(x)));
	for(int i=2; i<=korijen; i++)
		if(x%i == 0)
			return false;
	return true;
} 

int SumaDjelilaca(long long int x) {
	int suma(0);
	if(x>0)
		for(int i=1; i<=x; i++)
			if(x%i == 0)
				suma+=i;
	if(x<0)
		for(int i=x; i<=-1; i++)
			if(x%i == 0) 
				suma+=-i;
	return suma;
}
