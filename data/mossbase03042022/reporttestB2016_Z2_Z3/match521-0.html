int SumaCifara(long long int n) {
	if(n<0) n=abs(n);
	int suma(0);
	while(n!=0) {
		suma+=n%10;
		n/=10;
	}
	return suma;
}
bool savrsen(long long int n) {
	int suma=0;
	
	for(int i(1);i<n;i++) {
		if(n%i==0) {
			suma+=i;
		}
	}
	if(suma==n) return true;
	return false;
}

int BrojProstihFaktora(long long int n) {
	int br(0);
	if(n<0) n=abs(n);
	for(int i(2);i<=n;i++) {
		if(n%i==0) {
			while(n%i==0) {
			br++;
			n/=i;
			}
	}
	}
	return br;
}
