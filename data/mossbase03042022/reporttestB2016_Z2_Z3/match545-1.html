int SumaDjelilaca(long long int g){
	if(g<0)g=-g;
	if(g==0)return 0;
	if(g==1)return 1;
	long long int suma=g+1;
	long long int a=std::sqrt(g);
	for(long long int i=2;i<=a;i++){
		if(g%i==0){
			suma+=i;
			suma+=g/i;
		}
	}
	if(a*a==g)suma-=a;
	return suma;
}
