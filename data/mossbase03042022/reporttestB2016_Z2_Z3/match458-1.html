bool jeLiSavrsen(long long x)
{
	if(x<1) return false;
	long long suma = 0;
	for(int i=1;;i++)
	{
		if(x/i<=i) break;
		if(x%i == 0) suma+=i;
	   	if(x%i == 0) suma+=x/i;
	}
	return suma == 2*x;
}
