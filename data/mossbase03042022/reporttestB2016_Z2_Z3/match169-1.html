int PascalovTrougao (int m, int n ){
    Mat a;
    a.resize(m);
    for(int i=0;i<m;i++){
        a[i].resize(i+1);
    }
    for(int i=0; i<m; i++){
        for(int j=0; j<=i; j++){
            if(j==0){
                a[i][j]=1;
            }    
            else 
                if(j==i){
                    a[i][j]=1;
                }    
            else{
                a[i][j]=a[i-1][j-1]+a[i-1][j];
            }    
        }
    }
    return a[m-1][n-1];
}
