int SumaDjelilaca(long long int broj) {

}
int BrojProstihFaktora(long long int broj) {
	std::vector<int> v;
	int k=2;
	while(broj>1) {
		if(broj%k==0) {
			v.push_back(k);
			broj/=k;
		}
		else k++;
	}
	return v.size();
}
