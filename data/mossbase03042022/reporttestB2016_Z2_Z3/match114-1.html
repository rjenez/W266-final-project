template <typename Container>
bool Sortiranje(Container v1, Container v2){
    if (v1[2]==v2[2]){
        if (v1[0]==v2[0]){
            return v1[1]<v2[1];
        }
        return v1[0]<v2[0];
    }
    return v1[2]<v2[2];
}
