	}
	std::sort(mat.begin(),mat.end(),[](std::vector<decltype(mat[0][0]+mat[0][0])> a,std::vector<decltype(mat[0][0]+mat[0][0])> b){
		if(a[2]<b[2]) return true;
		else if(a[2]==b[2]){
			if(a[0]<b[0]) return true;
			else if(a[0]==b[0]){
				if(a[1]<b[1]) return true;
			}
		}
		return false;
	}
	);

}
