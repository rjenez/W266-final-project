template<typename T1,typename T2>
bool imal(T1 p1,T1 p2,T2 n){
    while(p1!=p2){
        if(*p1==n)return true;
        p1++;
    }
    return false;
}
template<typename T1,typename T2,typename T3>
bool ImalPoFunkciji(T1 p1,T1 p2,T2 n,T2 fun(T3)){
    while(p1!=p2){
        if(fun(*p1)==fun(n))return true;
        p1++;
    }
    return false;
}
