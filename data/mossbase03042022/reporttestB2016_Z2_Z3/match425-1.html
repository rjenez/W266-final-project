int SumaDjelilaca (long long int n) {
	int suma(0);

	}
	return suma;
}


int BrojProstihFaktora (long long int n) {
	int broj(0), prvi(2);
	n=abs(n);
	while(n>1) {
		if(n%prvi==0) {
			broj++;
			n/=prvi;
		} else prvi++;
	}
	return broj;
}
