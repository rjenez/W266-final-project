   	if(mine.at(i).at(0)<0|| n<=mine.at(i).at(0) || mine.at(i).at(1)<0 || n<=mine.at(i).at(1)) 
   	             throw std::domain_error("Ilegalne pozicije mina");

   }
   return ploca;
}
