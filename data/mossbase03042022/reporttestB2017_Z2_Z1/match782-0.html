	Tabla polja;
	int novi_x,novi_y;
	for(;;) {
		try {
			std::cout<<"Unesite komandu: ";
			bool ispravna(UnosKomande(komanda,smjer,x,y,greska));
			if(ispravna) {
				IzvrsiKomandu(komanda,polja,x,y,smjer,novi_x,novi_y);
			}

		}
		catch(std::runtime_error) {
			std::cout<<"Dovidjenja!"<<std::endl;
			return 0;
		}
	}
	return 0;
