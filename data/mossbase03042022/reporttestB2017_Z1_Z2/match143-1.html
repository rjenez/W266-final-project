int UdaljenostOdKraja(int i, int j, int max){
	if(max<=1) return 0;
	if(j<=max/2){
		if(i<j) return i;

		return j;
	}
	int pom(max-j);
	if(i<pom) return i;
	if(pom<max-i) return pom;
	return max-i;
}
