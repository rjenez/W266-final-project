typedef std::vector<std::vector<double>> Matrica;

Matrica KreirajMatricu(int broj_redova, int broj_kolona)

}

int BrojRedova(Matrica a)

}

int BrojKolona(Matrica a)
{
	return a.at(0).size();
}
