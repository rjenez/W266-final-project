bool DaLiJePlanina(std::vector<std::vector<double>> matrica, int a, int b, int vel)
{
	double vs = matrica.at(a).at(b);
	int h;
	
	for(h=1; h<=vel/2; h++)
	{
		double ms(0);
		
		for(int i=a-h; i<=a+h; i++)
		{
			for(int j=b-h; j<=b+h; j++)
			{
				if(j!=b-h && j!=b+h) 
				{
					continue;
				}
				ms = ms + matrica.at(i).at(j);			
			}
		}
		if(ms>vs) 
		{
			return false;
		}
		
		ms=vs;
	}
	return true;
}
