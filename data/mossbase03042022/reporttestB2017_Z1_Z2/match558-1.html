double SumaKoncentricnogKvadrata (Matrica m, int x, int y, int k)
{
	double suma(0);
	if(x-k<0 || y-k<0) return 0;
	for(int i=0; i<2*k+1; i++) {
		for(int j=0; j<2*k+1; j++) {
			if(i==0 || i==2*k)
				suma+=m.at(i+x-k).at(j+y-k);
			else if(j==0 || j==2*k)
				suma+=m.at(i+x-k).at(j+y-k);
		}
	}



	return suma;
}
