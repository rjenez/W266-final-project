double IzracunajSumu(vector<vector<double>> mat,int p,int q,int br)
{
    double suma=0;
    if(p-br>0 && p+br<mat.size() && q-br>0 && q+br<mat[0].size())
        for(int i=p-br;i<=p+br;i++)
        {
            for(int j=q-br;j<=q+br;j++)
            {
                if(i!=p || j!=q)
                    suma+=mat[i][j];
            }
        }
    else return 0;

    return suma;

}
