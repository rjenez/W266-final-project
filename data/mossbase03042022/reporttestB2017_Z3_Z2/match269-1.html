bool validno(char a)
{

}

vector<string> sep(string s)
{
    vector<string> v;
    for(int i=0; i<s.size(); i++)
    {
        string s1;
        while(validno(s[i])) {s1.push_back(s[i]); i++;}
        if(s1.size()!=0) v.push_back(s1);
    }
    return v;
}
