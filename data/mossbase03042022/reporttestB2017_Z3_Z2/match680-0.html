bool DaLiJeSlovoCifra (char s)

    int i=0;
    while (i!=s.size())
    {
        while (i!=s.size() && DaLiJeSlovoCifra(s[i])) i++; 
        
        int j=i;
        
        while (j!=s.size() && !DaLiJeSlovoCifra(s[j])) j++;

}
