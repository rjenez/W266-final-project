std::string PodString(int poz, std::string rec)
{
    std::string temp;
    for(int i(poz); i < rec.size(); i++) {
        while(rec[i] >= 'a' && rec[i] <= 'z' || rec[i] >= 'A' && rec[i] <= 'Z' || rec[i] >= '0' && rec[i] <= '9') {
            temp.push_back(rec[i]);
            i++;
        }
        if(!(rec[i] >= 'a' && rec[i] <= 'z' || rec[i] >= 'A' && rec[i] <= 'Z' || rec[i] >= '0' && rec[i] <= '9')) break;
    }
    return temp;
}
