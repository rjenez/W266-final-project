void unesi(char niz[], int velicina)
{
	char znak = getchar();
	int i;
	if(znak != '\n') znak = getchar();


	i = 0;
	while(i < velicina - 1 && znak != '\n') {
		niz[i] = znak;
		i++;
		znak = getchar();
	}
	niz[i] = '\0';

}

int vrati_duzinu(char *s)
{
	int duzina = 0;
	char *pom = s;
	while(*pom != '\0') {
		duzina++;
		pom++;
	}
	return duzina;
}
