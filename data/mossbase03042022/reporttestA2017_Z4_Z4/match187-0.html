int ukrstanje(char *poc, char *kraj){
	int br1 = 0, br2 = 0;
	while(poc <= kraj){
		if(*poc == '<'){
			if(*(poc+1) == '/') br2++;
			else br1++;
		}
		poc++;
	}
	if(br1 == br2) return br1;
	else return 0;
}
