int jeSlovo(char c){
    if((c >= 'A' && c <= 'Z') ||(c >= 'a' && c <= 'z'))
    return 1;
    return 0;
}
int vecSeNalazi(char *x, struct rijec *niz, int size){
    //vraca indeks x u nizu niz ako se nalazi, -1 ako se ne nalazi
    int i;
    for(i = 0; i < size; i++){
        if(strcmp(x, niz[i].sadrzaj) == 0){
         return i;   
        }
    }
    return -1;
}
