void unesi(char niz[]) 
{
	int i;
	char znak = getchar();
	if(znak == '\n') 
	  znak = getchar();
	i = 0;
	while(i < 19 && znak != '\n') 
	{
		niz[i] = znak;
		i++;
		znak = getchar();
	}
	niz[i] = '\0';
}
