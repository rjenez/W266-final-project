#include <stdio.h>
#include <ctype.h>
#include <string.h>
void unesi(char niz[], int velicina)
{
    char znak = getchar();
    if (znak == '\n') znak=getchar();
    int i = 0;
    while (i < velicina-1 && znak != '\n')
    {
        niz[i] = znak;
        i++;
        znak = getchar();
    }
    niz[i]='\0';
}

/* Funkcija dekompresuje datoteku kompresovanu funkcijom "kompresuj"*/
void dekompresuj(char *ul,char *iz)
{

	FILE* ulaz = fopen(ul,"r"), *izlaz = fopen(iz,"w");
	char mat[11][11];
	for(i=0;i<10;i++)
	{
		aux = fgetc(ulaz);
		if(aux=='\n') break;
		mat[i][0] = aux;
		for(j=1;;j++)
		{
			aux = fgetc(ulaz);
			if(aux=='\n') break;
			mat[i][j] = aux;
		}
		mat[i][j] = 0;
	}
	while(aux!=EOF)
	{
		aux = fgetc(ulaz);
		if(19<aux && aux<30)
		{
			fprintf(izlaz,"%s",mat[aux-20]);
		}
		else fputc(aux,izlaz);
	}
	printf("Datoteka dekompresovana.\n");
	fclose(ulaz);
	fclose(izlaz);
}
