#include <stdio.h>
#include <string.h>

int isLetter(char c) {
	if((c >= 65 && c <= 90) || (c >= 97 && c <= 122)) return 1;
	return 0;
}

int checkFile(FILE *f) {
	if(f == NULL) {
		printf("Datoteka ne postoji.\n");
		return 0;
	}
	return 1;
}

void oslobodiResurse(FILE *ulaz, FILE *izlaz) {
	fclose(ulaz);
	fclose(izlaz);
}

char loadWord(char *rijec, FILE *file) {
	char c;
	while(1){
		c = getc(file);
		if(!isLetter(c))
			break;
		*rijec = c;
		rijec++;
	}
	*rijec = '\0';
	return c;
}
