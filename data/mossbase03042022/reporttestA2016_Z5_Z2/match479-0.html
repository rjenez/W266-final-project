int je_slovo(char c){
	
	if((c >= 'a' && c <= 'z' ) || (c >= 'A' && c <= 'Z')) return 1;
	return 0;
}

			continue;
		}
		if(je_slovo(c)){
			trenutna_duzina = 0;
			while(je_slovo(c) && trenutna_duzina <= 10){

				trenutna_duzina++;

				while(c != EOF && je_slovo(c)){
					c = fgetc(ulaz);
