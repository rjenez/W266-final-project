bool Broj(std::string s){
    for(int i=0;i<s.length();i++){
        if(!std::isdigit(s[i]))return false;
        else{
            while(std::isdigit(s[i])&&i!=s.length()-1)i++;
            if(s[i]!='/'||i==s.length()-1) return false;
