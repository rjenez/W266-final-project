int VratiVelicinuStringa(string s){
    int brojac(0);


    if (s.length()==0)return brojac=0;
    for(int i=0;i<s.length();i++){
        if((s[i]>='a' && s[i]<='z') || (s[i]>='A' && s[i]<='Z') || (s[i]>='0' && s[i]<='9'))brojac++;


    }
    return brojac;
}
