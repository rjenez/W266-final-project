int broj_slova(string s){
    int i=0,brojac=0;
    while(i!=s.size()){
        brojac=0;
        if((s[i]>='a'&&s[i]<='z')||(s[i]>='A'&&s[i]<='Z')||(s[i]>='0'&&s[i]<='9')){
            while((s[i]>='a'&&s[i]<='z')||(s[i]>='A'&&s[i]<='Z')||(s[i]>='0'&&s[i]<='9')){
                brojac++; i++;
            }
        }


        else i++;
    }
    return brojac;
}
