int Prebroji (string rijec)
{
    int broj(0);
    
    for (int i=0; i<rijec.length(); i++)
    {


        if ((rijec[i]>='A'&&rijec[i]<='Z')||(rijec[i]>='a'&&rijec[i]<='z')||(rijec[i]>='0'&&rijec[i]<='9'))
        {
            broj++;
        }
    }
    
    return broj;
}

    n=grupa.size();
    
    prvi=n%k;
    
    for (int i=0; i<k; i++)
    {
        if (i<prvi)

            granice.push_back(n/k);
        }

    auto x=kopija.begin();
    auto y(x);
    while (pravo!=grupa.size())
    {
        vel=Prebroji(*x);
    
        temp.insert(*x);
        y=x;
        if (x==--kopija.end())
        {
            x=kopija.erase(y);
            x++;
        }
        else
        {
            x=kopija.erase(y);
        }
        
  
        brojac++;
        
        if (brojac==granice[g])

            rez.push_back(temp);
            g++;

            {

            }
        }

    }
    
    return rez;
}
