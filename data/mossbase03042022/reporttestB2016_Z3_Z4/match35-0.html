int BrojSlova(string s){
    int brojac(0);
    for (auto i=0; i<s.size(); i++) {
        if (s.at(i)>='A' && s.at(i)<='Z') brojac++;
        else if (s.at(i)>='a' && s.at(i)<='z') brojac++;
        else if(s.at(i)>='0' && s.at(i)<='9') brojac++;
    }
    return brojac;
}
