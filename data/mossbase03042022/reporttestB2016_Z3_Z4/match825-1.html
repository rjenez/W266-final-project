bool SlovoIliCifra(char a){
    return ((a>='a' && a<='z') || (a>='A' && a<='Z') || (a>='0' && a<='9'));
}



int Duzina(std::string ime){
    int brojac(0);
    for(int i=0; i<ime.size(); i++) {
        if(SlovoIliCifra(ime[i]))
            brojac++;
    }
    return brojac;
}

}
