int Duzina(std::string ime){
    int brojac=0;


    for(int i=0;i<ime.size();i++){
        if ( ime[i]<'0' || (ime[i]>'9' && ime[i]<'A') || (ime[i]>'Z' && ime[i]<'a') || ime[i]>'z' )brojac++;
    }
    return ime.length()-brojac;
}
