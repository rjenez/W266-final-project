int duzinaImena(std::string s){
    int brojac=0;
    for(int i=0;i<s.size();i++){
        if((s[i]>='A' && s[i]<='Z') || (s[i]>='a' && s[i]<='z') || (s[i]>='0' && s[i]<='9')) brojac++;
        
    }
    return brojac;
}
