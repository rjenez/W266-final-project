int BrojSlova(std::string ime) {
    int br(0);
    //std::cout<<"U funkciji: "<<ime<<std::endl;
    for(int i(0);i<ime.size();i++) {
        if((ime[i]>='A' && ime[i]<='Z') || (ime[i]>='a' && ime[i]<='z') || (ime[i]>='0' && ime[i]<='9')) br++;
    }
    return br;
}
