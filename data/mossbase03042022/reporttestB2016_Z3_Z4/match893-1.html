int PrebrojiSlova (std::string ime) {
    int broj_slova(0);
    for (int i=0; i<ime.size(); i++) {
        if ((ime[i]>='a' && ime[i]<='z') || (ime[i]>='A' && ime[i]<='Z') || (ime[i]>='0' && ime[i]<='9')) broj_slova++;
    }
    return broj_slova;
}
