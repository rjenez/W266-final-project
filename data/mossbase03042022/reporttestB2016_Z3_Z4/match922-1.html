int DuzinaImena(std::string ime) {
    int brojac(0);
    for(int i = 0; i < ime.length(); i++)
        if((ime[i] >= 'a' && ime[i] <= 'z') || (ime[i] >= 'A' && ime[i] <='Z') || (ime[i] >= '0' && ime[i] <= '9'))
            brojac ++;
    return brojac;
    
}

}
