bool slovo(char a)
{
    return (toupper(a)>='A'&&toupper(a)<='Z' || a>='0'&&a<='9');
}

int slova(string s)
{
    int br=0;
    for(int i=0; i<s.size(); i++)
        if(slovo(s[i])) br++;
    return br;
}
