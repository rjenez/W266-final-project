int Duzina(std::string ime) {
    int n(0);
    for (auto i : ime) {
        if ((i >= 'a' && i <= 'z') || (i >= 'A' && i <= 'Z') || (i >= '0' && i <= '9')) {
            n++;
        }
    }
    return n;
}
