auto Postavi(list<string>::iterator iter,list<string> &l,int n) -> decltype (iter) {
    for(int i=0;i<n-1;i++)
    {
        auto pom=l.end(); pom--;
        auto pom1=l.begin();
        if(iter==pom) iter=pom1; 
        else iter++;
    
    }
    return iter; 
}
