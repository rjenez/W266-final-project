int DuzinaStringa (std::string rijec) {
    int duzina=rijec.length();
    auto it(rijec.begin());
    while(it!=rijec.end()) {
        if(!((*it>='A' && *it<='Z') || (*it>='a' && *it<='z') || (*it>='0' && *it<='9'))) duzina--;
        it++;
    }
    return duzina;
}
