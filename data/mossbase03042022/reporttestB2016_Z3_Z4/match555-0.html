int DajBrojSlova(string s)
{
    int brojac=0;
    auto it(s.begin());
    while(it != s.end()) {
        if((*it >= 'a' && *it <= 'z') || (*it >= 'A' && *it <= 'Z') || (*it >= '0' && *it <= '9'))
            brojac++;
        it++;
    }
    return brojac;
}
