int DajBrojSlova(string s){
    
    int broj_slova = 0;
    for(int i = 0 ; i < s.length() ; i++){
        if((s[i] >= 'a' && s[i] <= 'z') || (s[i] >= 'A' && s[i] <= 'Z') || (s[i] >= '0' && s[i] <= '9') ){


            broj_slova++;
        }
    }
    
    return broj_slova;
}
