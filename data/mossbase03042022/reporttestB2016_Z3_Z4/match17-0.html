std::vector<int>Napuni_velicinom(std::vector<std::string>vs, int timovi){

    ostali_timovi=k-prvi_timovi;  cl_ost=n/k;
    for(;;){
        if(prvi_timovi>0){
            while(prvi_timovi>0){
                v.push_back(cl_pt);
                prvi_timovi--;
            }
        }else{
            while(ostali_timovi>0){
             v.push_back(cl_ost);
             ostali_timovi--;
            }
        }
        if(prvi_timovi==0 && ostali_timovi==0) break;
    }
    return v;
}
