int duzina(std::string s) {
    int br=0;
   for(int i=0; i<s.size(); i++)
   if((s[i]>='a' && s[i]<='z') || (s[i]>='A' && s[i]<='Z')  || (s[i]>='0' && s[i]<='9') ) br++;
   return br;
}

}
