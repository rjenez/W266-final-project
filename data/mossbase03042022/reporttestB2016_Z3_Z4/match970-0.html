int brojKoraka(string s){
    int c = 0;
    for (int i = 0; i < s.size(); i++) {
        if((s[i]>='0' && s[i]<='9')||(s[i]>='a' && s[i]<='z')||(s[i]>='A' && s[i]<='Z'))
        c++;
    }
    return c-1;
}
