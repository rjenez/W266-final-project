int Znakovi(std::string ime) {
    int BrojZnakova(0);
    for(int i = 0; i < ime.size(); i++) {
        if((toupper(ime[i]) >= 'A' && toupper(ime[i]) <= 'Z') || (ime[i] >= '0' && ime[i] <= '9')) BrojZnakova++;
    }
    return BrojZnakova;
}

        mozel.push_back(1);
    }
    for(int k=0; k<broj_timova-broj_sa_vise; k++){
        mozel.push_back(0);
    }
    auto it=lista.begin();

            i++;
        }
    }
    return povratni;
}
