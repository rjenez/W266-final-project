int NadjiVelicinu(std::string dijete)
{
    int broj=0;
    for(int i=0;i<dijete.size();i++)
    {
        if((dijete[i]<='z' && dijete[i]>='a') || (dijete[i]<='Z' && dijete[i]>='A') || (dijete[i]>='0' && dijete[i]<='9'))
        broj++;
    }
    return broj;
}
