bool JeLiProst(int x) {
	bool prost(true);
	for(int i=2; i<=std::sqrt(x); i++) {

		}

}

bool Simetrican(vector<int> v) {
	bool simetrican(true);
	int duzina1=v.size()/2;
	int duzina=v.size()-1;
	for(int i=0; i<=duzina1; i++) {
		if(v.at(i)!=v.at(duzina-i)) {
			simetrican=false;
			break;
		}
	}
	return simetrican;
}
