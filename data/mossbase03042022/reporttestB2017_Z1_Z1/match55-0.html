bool DaLiJeProst (int n){ 
	bool prost=true;
	if(n<=1) prost=false;
	else{
		for(int i=2; (unsigned int) (i*i)<=n; i++){
			if(n%i==0) prost=false;
		}
	}
	return prost;
}
