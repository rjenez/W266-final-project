#include <iostream>
#include <vector>

bool DaLiJeProst(int n)
{
	if(n<=1) return false;

	for(int i=2; (unsigned int)(i*i)<=n;i++)
	{
		if(n%i==0) return false;
	}
	return true;
}

std::vector<int> IzdvojiSimetricneTernarne(std::vector<int> v, bool jeProst)
{
	std::vector<int> t;
	for(int x : v)
	{
		long temp(x);
		if(temp<0) temp*=-1;
		if(DaLiJeProst(temp)==jeProst && temp != 1)
		{			
			std::vector<int> br;
			while(temp>0)
			{
				br.push_back(temp%3);
				temp/=3;
			}
			bool jeSimetrican=true;
			for(int i=0;i<br.size()/2;i++) 
			{
				if(br.at(i)!=br.at(br.size()-i-1)) 
				{
					jeSimetrican=false;
					break;
				}
			}
			if(jeSimetrican) 
			{
				bool ubaciti=true;
				for(int a : t)
				{
					if(a==x)
					{
						ubaciti=false;
						break;
					}
				}
				if(ubaciti) t.push_back(x);
			}
		}
	}
	return t;
}
