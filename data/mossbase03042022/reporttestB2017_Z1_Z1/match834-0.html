vector<int>IzbaciIste(vector<int>v){
    for(int i=0;i<v.size();i++){
        for(int j=i+1;j<v.size();j++){
            if(v.at(i)==v.at(j)){
                v.at(j)=v.at(j+1);
                v.erase(v.begin()+j);
                j--;
            }
        }
    }
    return v;
}

	n=fabs(n);

    }
	}

}
