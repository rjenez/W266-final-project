std::vector<int> PretvoriUTernarni (int broj)
{
	std::vector<int> v;
	int p=broj;
	int c;
	while(p!=0) {
		c=p%3;
		p=p/3;
		v.push_back(c);
	}
	
	return v;
}

std::vector<int> ObrniUTernarni (std::vector<int> v)
{ 
	std::vector<int> V;
	for(int i=v.size()-1; i>=0; i--){
		V.push_back(v.at(i));
	}

	return V; 
}
