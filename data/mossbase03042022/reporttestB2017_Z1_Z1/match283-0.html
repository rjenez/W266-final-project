bool JeLiProst(int n) {

	int k(sqrt(n));
	for(int i=2; i<=k; i++) {
		if(n%i==0) return false;
	}

}


std::vector<int> Ternarni(int n) {
	std::vector<int> rez;

	while(n!=0) {
		rez.push_back(n%3);
		n/=3;
	}

	return rez;
}

bool Simetricni(std::vector<int> v) {

	bool jeste=true;
	for(int i=0, j=v.size()-1; i<v.size()/2; i++, j--) {

		if(v.at(i)!=v.at(j))
			jeste=false;
	}

	return jeste;

}
