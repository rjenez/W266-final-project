bool DaLiJeProst(int n) {
	
	int max = std::numeric_limits<int>::max();
	if(n<=1) return false;
	for(int i=2;(unsigned int)(i*i)<=n;i++){
		if((unsigned int)(i*i)>=max) return true;
		if(n%i==0) {
		return false;
		}
	}
	
		return true;
}
