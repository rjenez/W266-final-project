bool DaLiJeProst (int x)
{
//kopirana sa tutorijala 2
	if(x==1 || x==0) return false;
	long int m=abs(x);
	for (int i=2; i<=sqrt(m); i++) { //Fermaova teorema
		if (x%i==0) return false;
	}
	return true;

}
