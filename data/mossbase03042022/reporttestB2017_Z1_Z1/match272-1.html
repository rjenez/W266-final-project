bool DaLiJeProst (int x) {
	if (x < 0) x *= -1;
	if (x<=1) return false;
	for (int i=2; i <= x / 2; i++) 
		if (x%i == 0) return false;
	return true;
}
bool DaLiJeSimetrican (std::vector<int>v1) {
	bool simetricni(true);
	int j(v1.size()-1);
	for (int i = 0; i < v1.size()/2; i++) {
		if(v1.at(i) != v1.at(j)) simetricni=false;
		j--;
	}
	return simetricni;
}
std::vector<int> BrojUTernarni (int br) {
	std::vector<int>v ;
	if (br<0) br*=-1;
	while (br>0) {
		int ostatak = br%3;
		v.push_back(ostatak);
		br/=3;
	}
	return v;
}
