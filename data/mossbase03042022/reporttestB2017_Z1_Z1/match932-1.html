{

bool DaLiJeProst(int a)
{
	if(a<0) a=-a;

	if(a%2==0 && a!=2) return false;
	for(int i=2; i<=sqrt(a); i++) {
		if(a%i==0) return false;
	}
	return true;
}
bool DaLiSeNalazi(int a, std::vector<int> d)

}
