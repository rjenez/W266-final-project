int DaLiJeProst(int n)
{
	if(n==1 || n==-1) return 2;
	if(n==0) return 0;
	if(n<-1) n*=(-1);

		if(n%i==0) return 0;
	}

	return 1;
}

std::vector<int> UTernarne(std::vector<int> v)

}
