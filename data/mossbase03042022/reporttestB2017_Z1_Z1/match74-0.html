void mergeBrisanjeKolizija(int* A, int &vel, int l, int p, int q, int u, int* B){
    int indeks=0;
    int pocetak=l;
    while(1){
        bool imaMinus=false;
        if(A[l]==-1){
            l=p;
            imaMinus=true;
        }
        if(A[q]==-1){
            q=u;
            imaMinus=true;
        }
        if(imaMinus)break;
        if(A[l]<A[q]){
            B[indeks]=A[l];
            l++;
        } else {
            B[indeks]=A[q];
            q++;
        }
        indeks++;
        while(l<=p){
            if(A[l]==-1){
                l=p+1;
                break;
            }
            if(A[l]/100==B[indeks-1]/100){
                l++;
                vel--;
            } else break;
        }
        if(l==p+1)break;
        while(q<=u){
            if(A[q]==-1){
                q=u+1;
                break;
            }
            if(A[q]/100==B[indeks-1]/100){
                q++;
                vel--;
            } else break;
        }
        if(q==u)break;
    }
    while(l<=p){
        if(A[l]==-1){
            l=p+1;
            break;
        }
        B[indeks]=A[l];
        indeks++;
        l++;
        while(l<=p){
            if(A[l]/100==B[indeks-1]/100){
                l++;
                vel--;
            } else break;
        }
        if(l==p+1)break;
    }
    while(q<=u){
        if(A[q]==-1){
            q=u+1;
            break;
        }
        B[indeks]=A[q];
        indeks++;
        q++;
        while(q<=u){
            if(A[q]/100==B[indeks-1]/100){
                q++;
                vel--;
            } else break;
        }
        if(q==u+1)break;
    }
    for(int i=0;i<indeks;i++){
        A[pocetak+i]=B[i];
    }
    for(int i=indeks;i<(u-pocetak+1);i++){
        A[pocetak+i]=-1;
    }
}

void merges(int* A, int poc, int kraj, int &vel, int* B){
    if(kraj-poc<1)return;
    merges(A, poc, (poc+kraj)/2, vel, B);
    merges(A, (poc+kraj)/2+1, kraj, vel, B);
    mergeBrisanjeKolizija(A, vel, poc, (poc+kraj)/2, (poc+kraj)/2+1, kraj, B);
    for(int i=0;i<vel;i++)cout<<B[i]<<" "; cout<<endl;
}
