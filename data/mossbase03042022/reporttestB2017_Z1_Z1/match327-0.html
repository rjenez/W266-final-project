std::vector<int> ternarni(int broj){

}

bool simetrican(std::vector<int> n){
	
	int i=0, j=n.size()-1;
	while(i<=j){
		if(n.at(i)!=n.at(j))

		i++;
		j--;
	}
	
	return true;
}
