int Prost(int n)
{

}

bool Simetrican(int n)
{
	int p(n), q(0);
	
	while(p!=0)
	{
		q=q*10+p%10;
		p/=10;
	}
	if(q==n) return true;
	else return false;
}

	}
	return p;
}
