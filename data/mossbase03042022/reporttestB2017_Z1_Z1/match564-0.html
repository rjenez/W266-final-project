bool DaLiJeProst(int n) {
	if(n<=1) n*=-1;
	for(int j=2;j<=sqrt(n);j++) {

	return true;
}  

std::vector<int> Izdvoji(std::vector<int> v, bool prost) {
	std::vector<int> v1;
	std::vector<int> v2(v.size());

		v2.at(i)=v.at(i);
	}
	
	for(int i=0;i<v2.size();i++) {
		for(int j=i+1;j<v2.size();j++) {
			if(v2.at(j)==v2.at(i)) {
				v2.erase(v2.begin() + j);
				j--;
			}
		}

	}
	return v1;
}
