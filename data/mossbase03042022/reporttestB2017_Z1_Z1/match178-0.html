bool DaLiJeProst (unsigned int a) {
	if (a<2) return false;
	for( int i=2; i<a; i++) {
		if (a%i==0) return false;
	}
	return true;
}

bool DaLiPostojiDupli (std::vector<int> v, int a) {
	for (int i=0; i<v.size(); i++) {
		if (v.at(i)==a) return true;
	}
	return false;
}
