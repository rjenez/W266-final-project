int ObrniBroj(int broj)
{
	if(broj < 0) broj*=-1;
	int suma(0);
	while(broj != 0)
	{
		suma = suma*10 + broj%10;
		broj = broj / 10;
	}
	return suma;
}

bool DaLiJeProSt(int n)
{
	if(n < 0) n *=-1;
	bool prost(true);
	if(n <= 1) return false;
	for(int i=2; i<=sqrt(n); i++)
	{
		if(n%i == 0)
		{
			prost = false;
			break;
		}
	}
	return prost;
}

bool DaLiSePonavlja(std::vector<int> v, int n)
{

		if(v.at(i) == n) 
		{
			ponavlja_se = true;
			break;
		}
	}
	return ponavlja_se;
}
