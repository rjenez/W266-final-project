bool DaLiJeProst(int broj){
	if(broj<0) broj*=-1;

		}
	
	}
	return true;
}

bool palindrom(int poslani_broj)
{
	if(poslani_broj<0) poslani_broj*=-1;
	unsigned long long int obrnuti=0, broj=poslani_broj;
	while(broj!=0)
	{
		obrnuti=obrnuti*10+broj%10;
		broj/=10;
	}
	if(poslani_broj==obrnuti)
		return true;
	return false;
}
