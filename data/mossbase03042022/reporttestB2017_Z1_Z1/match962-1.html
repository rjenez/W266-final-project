bool DaLiJeProst (int n) {
	
	if(n<0)n=-n;


	for(int i=2; i<=sqrt(n); i++) {
		if(n%i==0) return false;
	}
	return true;
}

bool Simetrican (int n) {
	
    std::vector<int>v;

	while (n!=0) {
		int c=n%10;
		v.push_back(c);
		n=n/10;
	}
	int vel=v.size();
	for(int i=0;i<=vel/2;i++){
		if(v.at(i)!=v.at(vel-1-i))return false;
	}
	return true;
}
