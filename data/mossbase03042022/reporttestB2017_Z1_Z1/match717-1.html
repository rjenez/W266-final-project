// pomocna funkcija provjerava da li je proslijedjeni broj prost
bool DaLiJeProst (int n) {
	bool djeljiv(true);
	if (n<0) n=n*(-1);
	if (n<=1) return false;
	for (int i=2; i<n; i++) {
		if (n%i==0) return false;
		if (i==n-1) break; 
	}
	return djeljiv;
}
