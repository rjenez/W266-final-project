bool JeLIProstBroj(int n){

	if(n==0)
	return false;
	
	if(n==1)
	return false;

		if(n%i==0)
		return false;
	}
	return true;
}

long long int PretvoriUTernarniBroj(int n){
	
	long long int c(0);
	long long int cifra;

	}

		cifra=n%3;
		c=c+i*cifra;
		i=i*10;
		n/=3;
	}
	while(n!=0);

	return c;
}
