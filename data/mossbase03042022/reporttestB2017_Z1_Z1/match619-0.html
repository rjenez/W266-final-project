bool Prost (int a)
{
    
    for(int i=2;i<=trunc(std::sqrt(a));i++){

}

bool Simetrija(std::deque<int> Sim)
{
    int i;
    bool simetrican;
    simetrican = true;
    i=0;
    while ((i<=trunc(Sim.size()/2)-1) && simetrican) {
      simetrican = Sim.at(i)==Sim.at(Sim.size()-i-1);
      i ++;
    }
    return simetrican;
}
