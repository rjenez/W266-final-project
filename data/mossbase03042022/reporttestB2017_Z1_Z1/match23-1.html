std::vector<int> Vektor(std::vector<int> v)
{
	std::vector<int> c;
	for(int i=0; i<v.size(); i++) {
		for(int j=i+1; j<v.size(); j++) {
			if(v.at(j)==v.at(i)) {
				v.erase(v.begin()+j);
			}
		}
		c.push_back(v.at(i));
	}
	return c;
}
