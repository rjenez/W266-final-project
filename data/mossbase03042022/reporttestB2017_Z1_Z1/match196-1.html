std::vector<int> izbaci_iste(std::vector<int> v)
{
	for(int i=0; i<v.size(); i++) {
		for(int j=i; j<v.size(); j++) {
			if(v.at(i)==v.at(j) && i!=j) {

			}
		}
	}

}
