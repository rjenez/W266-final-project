bool prost(int n){

}
bool simetrican(int n){ //pretvara u ternarne pa provjerava simetr.
	int x;
	std::deque<int> d;
	
	while(n!=0){
		x=n%3;
		d.push_front(x);
		n/=3;
	}
	for(int i=0; i<d.size(); i++){
		if(d.at(i)!=d.at(d.size()-i-1)) return false;
	}
	
	return true;
}
