bool DaLiJeTernarnoSimetrican(int broj)

	}

		if(ternarni.at(i)==ternarni.at(ternarni.size()-i-1))
			simetrija=true;
		else if(ternarni.at(i)!=ternarni.at(ternarni.size()-i-1)){
			simetrija=false;
			break;
		}

}
