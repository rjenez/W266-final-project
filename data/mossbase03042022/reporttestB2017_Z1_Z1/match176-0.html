/* Funkcija koja vraca vrijednost true ili false u zavisnosti od proslijedjenog parametra n */


bool DaLiJeProst(int n)
{
	if(n==0) return false; //Nulu posmatramo kao slozen broj
	n=fabs(n);
	for(int i=2; i<n; i++) {
		if(n%i==0)return false;
	}
	return true;
}

	}

}
