std::vector<int>PretvoriUTernarne(std::vector<int>a )
{

		while (a.at(i)!=0) {
			cifra=a.at(i)%3;
			a.at(i)/=3;
			v.push_back(cifra);
		}
		int duzina=v.size()-1;
		for(int j=v.size()-1; j>=0; j--) {
			cifra=v.at(j)*pow(10,duzina);
			suma+=cifra;
			duzina--;
		}
		v.resize(0);
		novi_vektor.push_back(suma);
		suma=0;
	}
	return novi_vektor;
}
