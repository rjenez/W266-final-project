void zaokruzi1(float niz[], int vel){

int suma_cifara (float a){
	int cifra=0, suma=0;
	int broj=fabs(a*10);
	while(broj!=0){
		cifra=broj%10;
		suma+=cifra;
		broj/=10;
	}
	return suma;
}
