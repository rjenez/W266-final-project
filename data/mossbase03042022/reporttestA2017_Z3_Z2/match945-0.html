int suma_cifara(float c)
{
	int suma=0, cifra=0;
	int pom=c*10;
	if(pom<0) pom=-pom;
	while(pom!=0) {
		cifra=pom%10;
		suma+=cifra;
		pom/=10;
	}
	return suma;
}
