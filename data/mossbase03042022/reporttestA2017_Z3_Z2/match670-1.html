#include <stdio.h>
#include <math.h>

int zbir_decimala(float f)
{
    int i, rez;
    
    rez = 0;
    f = fabs(f) * 10;
    for (i = 0; i < 25; i++)
    {
       rez += f - trunc(f / 10) * 10;
       f = trunc(f / 10);
    }

    return rez;
}

{

		niz[i] = round(niz[i] * 10) / 10;
}
