#include <stdio.h>
#include <math.h>



void zaokruzi1(float *p, int vel_niza)
{
	int i;
	for (i = 0; i < vel_niza; i++)
		*(p+i)=round(*(p+i)*10)/10;
}


void preslozi(float *p, int vel_niza, int k)
{
	zaokruzi1(p,vel_niza);
	float temp_broj;
	int temp, br_cifri, suma, i, j, br_vecih=0;

	for (i = 0; i < vel_niza; i++) {
		suma=0;
		br_cifri=0;
		temp = fabs(*(p+i)*10);
		do {
			br_cifri++;
			temp/=10;
		} while (temp>=1);

		temp = fabs(*(p+i)*10);
		do {
			suma+=temp%10;
			temp/=10;
			br_cifri--;
		} while (br_cifri!=0);

		if (suma>=k) {
			for (j = i; j > 0 + br_vecih; j--) {
				temp_broj=*(p+j-1);
				*(p+j-1)=*(p+j);
				*(p+j)=temp_broj;
			}
			br_vecih++;
		}

	}

}
