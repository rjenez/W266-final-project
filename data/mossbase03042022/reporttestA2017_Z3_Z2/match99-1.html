#include <stdio.h>



int suma(int n)
{
	int s=0;
	if(n<0) n=n*(-1);
	while(n!=0)
	{
		s=s+n%10;
		n=n/10;
	}
	return s;
}

				niz[i]=suma(niz[i-1]);
				
		 }

		niz[vel]=suma(niz[vel-1]);
	
}
	
	int Fibonaci(int n)
	{
		int a=1,b=1,c;
		if(n==1 || n==2) return 1;
		while(a+b<=n)
		{
			if(a+b==n) return 1;
			c=a;
			a=b;
			b=c+b;
		}
		return 0;
	}
	
	int izbaci(int niz[], int vel)
	{
		int i,j;
		for(i=0;i<vel;i++)
		{
			if(Fibonaci(niz[i]))
			{
				for(j=i;j<vel-1;j++)
				niz[j]=niz[j+1];
				vel--;
				i--;
			}
		}
		
		return vel;
	}
