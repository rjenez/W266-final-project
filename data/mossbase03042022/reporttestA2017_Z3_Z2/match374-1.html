int sumiraj (float *pok)
{
	int suma=0, x;

	x=*pok;
	while (x!=0) {
		suma+=fabs(x%10);
		x/=10;
	}
	/* sumiranje preostale decimalne cifre,
	potrebno je zaokruziti prvu decimalu zbog nepreciznosti */
	suma+=fabs(round((*pok-(int)*pok)*10));

	return suma;
}
