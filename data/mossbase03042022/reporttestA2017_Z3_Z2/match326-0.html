int zbirCifara(float x)
{
	int zbirCifara = 0;
	int cijeliBroj = 10*x;
	if(cijeliBroj < 0) {
		cijeliBroj = -cijeliBroj;
	}
	while(cijeliBroj != 0) {
		zbirCifara += cijeliBroj % 10;
		cijeliBroj /= 10;
	}
	return zbirCifara;
}
