/* Funkcija koja sve clanove niza zaokruzuje na jednu decimalu. */



void zaokruzi1 (float *niz, int vel) {

	while (p < niz+vel) {
		*p *= 10;
		*p = round(*p);
		*p++ /= 10;
	}
}
