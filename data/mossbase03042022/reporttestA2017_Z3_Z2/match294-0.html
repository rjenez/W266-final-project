void zaokruzi1(float *niz, int velicina_niza)
{
	int i;

	for(i = 0; i < velicina_niza; i++) {
		niz[i] = (round(niz[i] * 10)) / 10;
	}
}

int suma_cifri(float br)							
{
	int suma = 0;
	int temp = (br * 10);

	if (temp < 0) temp = -temp;							

	while(temp != 0) {
		suma += (temp % 10);
		temp /= 10;
	}
	return suma;
}
