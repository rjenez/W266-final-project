}

/*funkcija koja racuna sumu cifara clanova niza*/
int sumaCifara (double x)
{

	float suma=0,dec;
	x=fabs(x);
	dec=x- (int) x;
	dec=(dec+0.05);
	cifra=dec*10;

	suma=cifra;
