int sumaCifara(int x){
	int zrtva = abs(x), duzina = 0,i, djel;
	int suma  = 0;
	while (zrtva > 0) {
		zrtva /= 10;
		duzina++;
	}
	zrtva = abs(x);
	djel = 10;
	for(i = 0; i < duzina; i++){
		suma += (zrtva % djel) / (djel/10);
		djel *= 10;
	}
	return suma;
}
