#include <stdio.h>
#include <math.h>

void zaokruzi1(float niz[], int size)
{
	int i;
	/*zaokruzivanje niza na jednu decimalu*/
	for(i=0; i<size; i++) {
		niz[i]=roundf(niz[i]*10)/10;
	}
}



void preslozi(float *niz1,int vel1, int k)
{
	int i=0,x1=0,x2=0,s1=0,s2=0;
	float temp=0;
	zaokruzi1(niz1,vel1);

	for(i=0; i<vel1-1; i++) {

		x1=round(niz1[i]*10);
		x2=round(niz1[i+1]*10);
		do {
			s1+=abs(x1%10);
			x1/=10;
		} while(abs(x1)>0);
		do {
			s2+=abs(x2%10);

		} while(abs(x2)>0);

			niz1[i+1]=temp;

		}
		s1=0;
		s2=0;
	}
}
