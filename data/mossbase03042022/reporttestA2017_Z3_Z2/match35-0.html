int dva_od_tri(int niz1[50],int n1, int niz2[50], int n2, int niz3[50], int n3) {

		for(j=0; j<n2; j++){
			if(niz1[i]==niz2[j])
			brojac++;
		}
	}
	for(i=0; i<n1; i++){
		for(k=0; k<n3; k++){
			if(niz1[i]==niz2[k])
			brojac++;
		}
	}
	for(j=0; j<n2; j++){
		for(k=0; k<n3; k++){
			if(niz1[i]==niz2[j])
			brojac++;
		}
	}
	for(i=0; i<n1; i++){
		for(j=0; j<n2; j++){
			for(k=0; k<n3; k++){
				
				if(niz1[i]==niz2[j]==niz3[k])
				brojac--;
			}
		}
	}
return brojac;	
	
}
