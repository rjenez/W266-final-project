void zaokruzi1(float niz[], int n)
{

        niz[i]=(round(niz[i]*10))/10;
    }
}

int sumaf(float broj)
{
    int suma=0;
    int pom=0;
    broj*=10;
    pom=(int)broj;


    while (pom!=0) {
        suma += pom%10;
        pom/=10;
    }
    return suma;
}
