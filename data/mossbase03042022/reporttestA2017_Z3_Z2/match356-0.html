int suma_cifara(float n){ /*FUNKCIJA KOJA RACUNA SUMU CIFARA REALNOG BROJA*/
	if(n<0) n*=-1;	/*AKO JE BROJ NEGATIVAN, PRETVORI NJEGOVU KOPIJU U POZITIVNU ZBOG IZRACUNAVANJA SUME */
	int x=n*10,suma=0;
	
	while(x!=0){
		suma+=(x%10);
		x/=10;
	}
	return suma;
}
