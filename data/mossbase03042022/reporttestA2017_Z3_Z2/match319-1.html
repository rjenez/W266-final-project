void zaokruzi1(float niz[], int velicina)
{

		temp = round(y);
		rez = x + temp/10;

	}
}

int SumaCifara(float x)
{
	int suma = 0, rezultat;
	x = fabs(x*10);
	rezultat = x;
	while(rezultat > 0)
	{
		suma += rezultat % 10;
		rezultat /= 10;
	}
	return suma;
}
