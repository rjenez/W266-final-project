/*Funkcija "suma" vraca zbir cifara nekog realnog broja.*/


int suma(float broj)
{
	int a, i=0, zbir=0;
	broj=fabs(broj*10);
	while(broj!= 0) {
		a=(int)broj%10;
		broj=(int)broj/10;
		zbir+=a;
		i++;
	}
	return zbir;
}
/*Funkcija "zaokruzi1" prima niz realnih brojeva i sve clanove zaokruzuje na jednu decimalu. */
void zaokruzi1(float *niz, int vel)
{
	float *p=niz;
	while (p<niz+vel) {
		*p=round(*p*10)/10;
		p++;
	}
}
