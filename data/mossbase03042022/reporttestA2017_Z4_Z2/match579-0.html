#include <stdio.h>

void unesi(char niz[], int velicina) {
    char znak = getchar();
    if (znak == '\n') znak=getchar();
    int i = 0;


    while (i < velicina-1 && znak != '\n') {
        niz[i] = znak;
        i++;
        znak = getchar();
    }
    niz[i]='\0';
}

int jeLiSlovo(char c){
    if((c>='A' && c<='Z')||(c>='a' && c<='z'))return 1;
    return 0;
}
