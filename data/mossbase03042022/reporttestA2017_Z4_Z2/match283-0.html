#include <stdio.h>



int da_li_je_slovo (char  s)
{
	if ((s>='A' && s<='Z') || (s>='a' && s<='z'))
	{
		return 1;
	}
	return 0;
}

int broj_rijeci (char s[])
{
	int n=0, ne_slovo=1;
	
	while (*s != '\0')
	{
		if (!da_li_je_slovo(*s))
		{
			ne_slovo=1;
		}
		else if (ne_slovo==1)
		{
			ne_slovo=0;
			n++;
		}
		s++;
	}
	return n;
}
