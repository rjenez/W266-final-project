int prebroj_rijeci(char *string){
    char *pok=string;
    int brojac=0;
    
    while (*pok!='\0') {
        if((*pok>='a'&& *pok<='z') || (*pok>='A'&& *pok<='Z')){
            while((*pok>='a'&& *pok<='z')||(*pok>='A'&& *pok<='Z')){
                pok++;
            }
            brojac++;
        }
        if(*pok=='\0')break;
        pok++;
    }
    return brojac;
}
