#include <stdio.h>
void unesi (char niz[], int velicina)
{
    char znak = getchar();
    if (znak == '\n') znak=getchar();
    int i = 0;
    while (i < velicina-1 && znak != '\n') {
        niz[i] = znak;


        i++;
        znak = getchar();
    }
    niz[i]='\0';
}

		if ((*s>='A' && *s<='Z') || (*s>='a' && *s<='z'))
		{
			brojac++;
		}
		else if (brojac>0)
		{
			if (niz[rijec]!=brojac)
			{
				return 0;
			}
			rijec++;
			brojac=0;
		}
		s++;
	}
		 if (brojac>0)
	 {
	 	rijec=rijec+1;
	 }
	
	if (rijec!=vel)
	{
		return 0;
	}
return 1;
}
