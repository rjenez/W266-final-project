#include <stdio.h>



int provjeri_duzine(char *s, int *niz,int vel){
	int i = 0,duz;
	while(*s != '\0'){
		duz = 0;
		if(*s >= 'a' && *s <= 'z' || *s >= 'A' && *s <= 'Z' ){
				while(*s >= 'a' && *s <= 'z' || *s >= 'A' && *s <= 'Z'){s++; duz++;}
				if(duz != niz[i] || i > vel)return 0; // &&i<vel
				if(*s == '\0')s--;
				if(i == vel)return 0;
				i++;
		}
		s++;
	}
	
if(i != vel)return 0;
return 1;
}
