#include <stdio.h>
#include <ctype.h>



int JelSlovo(char s)
{ 
	int slovo=0;
	if(s>='A' && s<='Z' || s>='a' && s<='z')
		slovo=1;
	
	return slovo;
}

int provjeri_duzine(char *str,int *niz,int vel)
{
	int i;
	int br;
	int b=vel;
	char *p;
	if(vel<=0) return 0;
	
	for(p=str,i=0;*p!='\0';p++,i++)
	{
	    br=0;
	    while(!JelSlovo(*p))p++;
		while(JelSlovo(*p))
		{
			p++;
			br++;
		}
	
		b--;
		if(niz[i]!=br || b<0 || (*p=='\0' && i<vel-1))
			return 0;
			
		if(*p=='\0') break;
	}
	
	return 1;
}
