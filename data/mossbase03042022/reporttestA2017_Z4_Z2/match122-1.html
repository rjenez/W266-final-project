#define MAX 100

			neslovo=1;
		} else if(neslovo==1) {
			neslovo=0;
			broj_rijeci++;
		}
		s++;
	}
	if(broj_rijeci!=velicina) return 0;
	while(*spok!='\0') {
		if ( (!((*spok>='A' && *spok<='Z') || (*spok>='a' && *spok<='z')) && *spok!='\0')) duzina=0;

		while(( *spok!='\0') && ((*spok>='A' && *spok<='Z') || (*spok>='a' && *spok<='z'))) {

		if (duzina == *niz ) {
				niz++;
				spok--;
			} else if(duzina != *niz) {

				return 0;
			}

	}

	return 1;

}
