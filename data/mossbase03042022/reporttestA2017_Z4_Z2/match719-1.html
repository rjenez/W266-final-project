Primjer: ako string glasi:

Onda bi niz brojeva morao glasiti: 3, 2, 4, 7, 6 jer su to dužine riječi u ovoj rečenici.

void unesi (char niz[], int velicina) {
	char znak = getchar ();
	if (znak == '\n') getchar ();
	int i=0;
	while (i<velicina-1 && znak != '\n') {
		niz [i] = znak;
		i++;
		znak=getchar ();
	}
	niz[i]='\0';
	
}
