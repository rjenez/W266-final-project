#include <stdio.h>



void unesi(char *s, int duzina)
{
	int znak = getchar();
	if(znak == '\n') znak = getchar();
	int i = 0;
	while(i < duzina - 1 && znak != '\n') {
		s[i] = znak;
		i++;
		znak = getchar();
	}
	s[i] = '\0';
}
