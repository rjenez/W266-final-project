#include <stdio.h>

void unesi(char niz[], int velicina)
{
	char znak=getchar();
	if(znak=='\n') znak=getchar();
	int i=0;
	while (i<velicina-1 && znak!='\n') {
		niz[i]=znak;
		i++;
		znak=getchar();
	}
	niz[i]='\0';
}

			br++;
			while(!((*s>='a' && *s<='z') || (*s>='A' && *s<='Z'))) s++;
