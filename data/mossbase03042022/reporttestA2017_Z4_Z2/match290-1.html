			if (*(rijeci+1)!='\0'){

			}
			else {
				break;
			}
			continue;
		}
		if (*rijeci=='\0'){
			break;
		}

			return 0;
		}
		
		char *p;

		//	rijeci++;
		//}
		
		p = rijeci + niz[brojac];
		
		if (p>q){
			return 0;
		}
		brojac++;
		
		if (((*p>='a' && *p<='z') || (*p>='A' && *p<='Z')) && *p!='\0'){

				return 0;
			}
			rijeci++;
		}
		rijeci++;

	}

	return 1;
}
