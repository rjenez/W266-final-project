#include <stdio.h>

/*funkcija za brojanje rijeci u stringu*/
int broj_rijeci(char *s){
	char *p=s;
	int br=0;


	while( *p !='\0'){
		if((*p>='A' && *p<='Z')||(*p>='a' && *p<='z')){
			while((*p>='A' && *p<='Z')||(*p>='a' && *p<='z'))p++;
			if(!((*p>='A' && *p<='Z')||(*p>='a' && *p<='z')) || *p=='\0') br++;
			p--;
		}
		p++;
	}
	return br;
}
