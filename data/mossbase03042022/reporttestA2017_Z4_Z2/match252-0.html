#include <stdio.h>
//fja za prebrojavanje rijeci u stringu


int broj_rijeci(char*s)
{
	int slovo=0;
	int br_rijeci=0;
	while(*s) {
		if(!((*s>='A' && *s<='Z') || (*s>='a' && *s<='z')))
			slovo=0;
		else if(slovo==0) {
			slovo=1;
			br_rijeci++;
		}
		s++;
	}
	return br_rijeci;
}
