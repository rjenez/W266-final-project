#include <stdio.h>


int provjeri_duzine(char* recenica,int niz[],int velicina)
{
	int p[50000];
	int i,j,temp;
	i=0;
	while(*recenica!='\0')
	{
		temp=0;
		while((*recenica>=65 && *recenica<=90) || (*recenica>=97 && *recenica<=122))
		{
		   temp++;
		   recenica++;
		}
		if(temp>0) {p[i]=temp; i++;}
		if(*recenica=='\0') break;
		recenica++;
	}
	if(i!=velicina) return 0;
    for(j=0;j<velicina;j++)
    if(p[j]!=niz[j]) return 0;
    return 1;
}
