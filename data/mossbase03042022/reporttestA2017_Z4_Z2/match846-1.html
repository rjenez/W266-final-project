#include <stdio.h>


int broj_rijeci(char *recenica){
	int rijeci=0,ne_slovo=1;
	while(*recenica!='\0'){
		if(!((*recenica>='a' && *recenica<='z')|| (*recenica>='A' && *recenica<='Z'))) ne_slovo=1;
		else if(ne_slovo==1){
			ne_slovo=0;
			rijeci++;
		}
		recenica++;
	}
	return rijeci;
}
