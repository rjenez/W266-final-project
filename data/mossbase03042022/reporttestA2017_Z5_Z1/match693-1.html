int slovo(char a)
{
	if((a>='A'&&a<='Z')||(a>='a'&&a<='z'))return 1;
	return 0;
}

int isto(char* a, char* b)
{
	while(*a!='\0'||*b!='\0')
	{
		if(*a!=*b)return 1;
		a++;b++;
	}
	return 0;
