int porodica(char *s1, char *s2){
	/*provjerava string odzada jer je na kraju prezime*/
	do{s1++;}while(*s1);
	do{s2++;}while(*s2);
	while(*s1!=' ' && *s1!='-' && *s2!=' ' && *s2!='-' && *s1==*s2){
		s1--;
		s2--;
	}
	if((*s1==' ' || *s1=='-') && (*s2==' ' || *s2=='-'))return 1;
	return 0;
}
