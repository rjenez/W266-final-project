	int isto_prezime(char *a, char *b){
		int i=0;

		while((*b>='A' && *b<='Z') || (*b>='a' && *b<='z') || *b=='-') b--;
		a++;
		b++;
		while(prezime_a[i++]=*a++);
		prezime_a[i]='\0';
		i=0;
