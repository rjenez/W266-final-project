void kopirajRijec(char* ulaz,char* izlaz)
{
	while(*ulaz != '\0' && *ulaz !='\n')
	*izlaz++=*ulaz++;
	*izlaz='\0';
	
}

int jednakeRijeci(char* prva, char* druga)
{
	int jednake = 1;
	while((*prva != '\n' && *prva != '\0')&&(*druga != '\n' && *druga != '\0'))
	{
		if(*prva != *druga) jednake = 0;
		prva++;
		druga++;
	}
	
	if(*prva != *druga) jednake = 0;
	
	return jednake;
	
}

int vecaRijec(char* r1,char* r2)

}
