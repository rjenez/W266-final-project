void unesite (char niz[], int velicina)
{

	if (znak == '\n') znak = getchar();
	int i = 0;


	while (i< velicina-1 && znak != '\n') {
		niz[i]=znak;

		znak = getchar();
	}
	niz[i]='\0';
}
