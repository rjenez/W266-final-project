double vratiVeciManji(double x, double y, bool veci=true){
    if(veci){
        if(x>y) return x;
        return y;
    }
    else{
        if(x>y) return y;
        return x;
    }
}

bool provjeriRealne(double x, double y){
    if(fabs(x-y)<(EPSILON*(fabs(x)+fabs(y)))) return true;
    else return false;
}

}
