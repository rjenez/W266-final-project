void unesi (char niz[],int velicina)
{
	char znak=getchar();
	if (znak == '\n')
		znak = getchar();
	int i=0;
	while (i<velicina-1 && znak != '\n') {
		niz[i]=znak;
		znak=getchar();
		i++;
	}
	niz[i]='\0';
}
