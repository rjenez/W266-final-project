int slovo(char x)
{
    if((x>='A' && x<='Z') || (x>='a' && x<='z'))

}

int javlja(char *a,char *b)
{
    char*c;

    if(b==a)
        return 0;

    c=b-1;
    while(c>=a) {
        if(*c==*b || *c==*b+32 || *c==*b-32)
            return 1;
        c--;
    }

    return 0;
}
