int Slovo(char s){
    if((s>='a' && s<='z') || (s>='A' && s<='Z')) return 1;

}

int Prebrojistring(char *s){
    int brojac = 0;
    while(*s != '\0'){
        brojac++;
        s++;
    }
    return brojac;
}

    for (i=0; i<vel-1; i++) {

            if (niz[i] == niz[j] || niz[i] == (niz[j] + 32) || niz[i] == (niz[j] - 32)) {

                    niz[k] = niz[k+1];
                }
                niz[vel] = '\0';

                j--;
            }
        }
    }
}
