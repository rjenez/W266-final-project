		}
		q = p + 1;
		while (q < kontejner + vel) {
			char tmp2 = *q;
			if(tmp2 >= 'A' && tmp2 <= 'Z') {
				tmp2 += 'a'-'A';
			}
			if (tmp1 == tmp2) {
				char* pom = q;
				while (pom <= kontejner + vel) {
					*pom = *(pom + 1);

int duzina_rijeci (char* pocetak_rijeci, char* kraj_rijeci) {
	int duzina = 0;
	while (pocetak_rijeci < kraj_rijeci) {
		duzina++;
		pocetak_rijeci++;
	}
	return duzina;
}
