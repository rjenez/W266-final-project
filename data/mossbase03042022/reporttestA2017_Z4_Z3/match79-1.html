int slovo(char c)
{
	if((c>='A'&&c<='Z')||(c>='a'&&c<='z'))return 1;
	return 0;
}

int dva_slova(char a, char b)
{
	if( (a-b=='a'-'A'||a-b=='A'-'a'||a-b==0)&&slovo(a)&&slovo(b))return 1;
	return 0;
}
