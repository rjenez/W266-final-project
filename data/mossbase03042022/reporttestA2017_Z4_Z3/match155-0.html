char veliko(char c) {
     if (c >= 'a' && c <= 'z')
        c -= 32;
        
     return c;
}

int duzina_podniza(char* p, char* k) {
    char* s1 = p;
    char* s2;
    
    while (s1 < k - 1) {
        s2 = s1 + 1;
        while (s2 < k) {
            if (veliko(*s1) == veliko(*s2))
                return s2 - p;
            s2++;
        }
        s1++;
    }
    
    return k - p;
}
