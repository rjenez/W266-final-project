#include <stdio.h>
#define razlika 'a'-'A'



int Korektno(char s){
	return ((s>='a' && s<='z') || (s>='A' && s<='Z'));
}

int JeSuLiVelika(char s){
	return s>='A' && s<='Z';
}

void PostaviNiz(unsigned int niz[], int vel){
	unsigned int i;
	for(i=0;i<vel;i++){
		niz[i]=0;
	}
}
