int JeLiSlovo (char znak){
    return ((znak >= 'A' && znak <= 'Z') || (znak >= 'a' && znak <= 'z'));
}

//provjera duzine stringa, tj. pojedinacne rijeci

int duzina (char * string){
    int duzina = 0;
    while (*string != '\0' && JeLiSlovo(*string)){
        duzina++;
        string++;
    }
    return duzina;
}
