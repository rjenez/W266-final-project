					std::cout<<"Greska, unesite ponovo!"<<std::endl;
					continue;
				}
				if(s[1]-'0'>=n || s[3]-'0'>=n || (s[1]-'0'==0 && s[3]-'0'==0))
				{
					std::cout<<"Greska, unesite ponovo!"<<std::endl;
					continue;
				}
				mine.push_back(std::vector<int>());
				mine[i].push_back(s[1]-'0');
				mine[i].push_back(s[3]-'0');
				i++;
			}
			polja=KreirajIgru(n,mine);
	}
