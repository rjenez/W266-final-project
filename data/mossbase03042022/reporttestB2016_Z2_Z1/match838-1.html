string int2string(int x){
  string tekstualniBroj;
  if(x==0) return "0";
  while(x!=0){
    int cifra = x%10;
    
    tekstualniBroj.push_back('0'+cifra);
    x/=10;
  }
  
  return tekstualniBroj;
}
