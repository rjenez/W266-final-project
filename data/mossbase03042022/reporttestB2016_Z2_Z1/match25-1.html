		}
	
	polja=KreirajIgru(n,vek);
}
	else if(komanda==Komande::ZavrsiIgru){
		for(int i=0;i<polja.size();i++){
			for(int j=0;j<polja[i].size();j++){
				polja[i][j]=Polje::Prazno;
			}
		}
		if(komanda==Komande::ZavrsiIgru)throw std::runtime_error("Dovidjenja!");
	}
}
