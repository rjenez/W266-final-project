				kraj=j;
				int br=0;
				for(int k=poc;k<kraj;k++){
					x=x*pow(10,br)+(pom[k]-'0');
					br++;
				}
				j++;
				poc=j;
				while(pom[j]!=')')j++;
				kraj=j;
				br=0;
				for(int k=poc;k<kraj;k++){
					y=y*pow(10,br)+(pom[k]-'0');
					br++;
				}
			}
			
		}
		rez[a].push_back(x);
		rez[a].push_back(y);
		a++;
	}
	return rez;
}
