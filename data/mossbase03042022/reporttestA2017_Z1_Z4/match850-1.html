	} while(r>10 || r<1);

		printf("Unesite broj kolona: ");
		scanf("%d", &k);
	} while(k>10 || k<1);
	
	do{
		printf("Unesite sirinu jedne kolone: ");


		scanf("%d", &s);
	} while(s>10 || s<1);
	
	for(i=0; i<=2*r; i++){

			for(j=0; j<=k+k*s; j++){
