	int a,b,c,j,i,I,J;

	   printf("Unesite broj redova: ");

	   while(a<=0 || a>10);
	   
	do {   
	printf("Unesite broj kolona: ");
	scanf("%d",&b); }


	while(b<=0 || b>10);
	
	do {
	printf("Unesite sirinu jedne kolone: ");
	scanf("%d",&c); }
	while(c<=0 || c>10);
	I=(a*2)+1;
	J=b*c+1;


	for(i=0;i<I;i++ ) {
		for(j=0;j<J;j++) {

		}
