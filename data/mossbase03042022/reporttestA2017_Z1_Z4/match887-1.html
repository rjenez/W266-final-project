#include <stdio.h>

int main() {
	int r, k, sk, i, j;
	
	do {
		printf("Unesite broj redova: ");
		scanf("%d", &r);
	} while(r<=0 || r>10);
	
	do {
		printf("Unesite broj kolona: ");
		scanf("%d", &k);
	} while(k<=0 || k>10);
	
	do {
		printf("Unesite sirinu jedne kolone: ");


		scanf("%d", &sk);
	} while(sk<=0 || sk>10);
	
	for(i=0; i<=2*r; i++) {
		if(i%2==0) {
			for(j=0; j<=k*(sk+1); j++) 
			if(j%(sk+1)==0) 
			printf("+");
			else 
			printf("-");
		}
		else
			for(j=0; j<=k*(sk+1); j++)
			if(j%(sk+1)==0)
			printf("|");
			else
			printf(" ");
			printf("\n");
	}
	return 0;
}
