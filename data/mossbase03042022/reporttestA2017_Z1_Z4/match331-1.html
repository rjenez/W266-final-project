	} while (k<=0 || k>10);

	for (i=0; i<=r*2; i++)
		{

			{
				for (j=0;j<k;j++)

					for (z=0;z<s;z++)
						printf("-");
					
				}

			}
			
			else

				for (y=0;y<=(s+1)*k;y++)
				{
					if (y%(s+1)==0)
					
					printf("|");
					
					else printf(" ");
				}
				
			}
			
			printf("\n");
		}
	return 0;
}
