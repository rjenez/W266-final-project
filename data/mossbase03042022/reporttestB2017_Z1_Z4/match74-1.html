std::string IzvrniString(std::string s)
{

    for(int i=0; i<((int)s.size()/2); i++) {
        char pomocna(std::move(s[0+i]));
        s[i]=std::move(s[s.size()-1-i]);
        s[s.size()-1-i]=std::move(pomocna);
    }
    return s;
}


bool DaLiJeSlovo(char c)
{
    if(c>='a' && c<='z' || c>='A' && c<='Z') return true;
    return false;
}
