bool Slovo(char znak)															//fja koja provjerava da li je poslani znak slovo
{
	if((znak>='a' && znak<='z') || (znak>='A' && znak<='Z')) return true;
	return false;
}

std::string IzvrnutNaopacke(std::string s)
{
	int duzina(s.length());
	std::string rezultat;
	for(int i=duzina-1; i>=0; i--)
		rezultat.push_back(s.at(i));
	return rezultat;
}
