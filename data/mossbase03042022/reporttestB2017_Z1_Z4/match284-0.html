int PocetakPodstringa(std::string s1, std::string s2, int start){
	int istina=1;
	for(int i=start; i<s1.size(); i++){
		if(s1.at(i)==s2.at(0)){
			for(int j=0; j<s2.size(); j++){
				if(s1.at(i+j)!=s2.at(j)){
					istina=0;
					break;
				}
			}
			if(istina) return i;
			istina=1;
		}
	}
	return -1;
}
