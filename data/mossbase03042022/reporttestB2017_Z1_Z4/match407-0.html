string Obrni(string s)
{
	string pom;
	for(int i=s.size()-1;i>=0;i--)
		pom.push_back(s.at(i));
		
	return pom;
}

bool JelSlovo(string s)
{
	return (s[0]>='A' && s[0]<='Z') || (s[0]>='a' && s[0]<='z');	
}
