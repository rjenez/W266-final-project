string Palindrom(string s)
{
    string s1;
    s1.resize(s.length());
    int j(0);
    for(int i=s.length()-1; i>=0; i--) {
        s1.at(j)=s.at(i);
        j++;
    }
    return s1;
}

        s2.resize(s.length()/2);

        for(int i=0; i<(int)(s2.length()); i++)

    return s1;
}

bool Slovo(char c)                                          //Provjerava da li je znak slovo
{
    return (c>='A' && c<='Z') || (c>='a' && c<='z');
}
