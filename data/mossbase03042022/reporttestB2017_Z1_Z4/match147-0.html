std::string Obrni (std::string s)
{
    for (unsigned int i{0}; i < (s.length() + 1) / 2; i++)
        std::swap(s.at(i), s.at(s.length() - i - 1));
    return s;
}

{
	if (recenica.empty() || rijeci.empty())

    for (const std::string& rijec : rijeci)
       	if (rijec.length())
