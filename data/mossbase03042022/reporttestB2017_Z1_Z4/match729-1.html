/* Ova funkcija modificira primljeni string, vracajuci njegov polupalindrom.
Npr. za string programirane vrati programargorp */

std::string PoluPalindrom(std::string s)
{
	int i(0);
	int j(s.length()-1);
	while(i<j) {
		s.at(j)=s.at(i);
		i++;
		j--;
	}
	return s;
}

{

	int i(0);
	int j(s.length()-1);

}
