std::string Palindrom(std::string s) {
	char pomocna;
	for(int i=0;i<s.size()/2;i++) {
		pomocna=s.at(i);
		s.at(i)=s.at(s.size()-1-i);
		s.at(s.size()-1-i)=pomocna;
		if(i==s.length()-1-i) break;
	}
	return s;
}
