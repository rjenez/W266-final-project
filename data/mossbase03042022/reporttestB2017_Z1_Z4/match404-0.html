std::string IzvrniString(std::string s) {
      std::string novi; 
      for(int i=s.size()-1; i>=0; i--) 
            novi.push_back(s.at(i));
      
      return novi;
}

bool JelSlovo(char znak) {
      if(znak>='A' && znak<='Z' || znak>='a' && znak<='z') return true;
