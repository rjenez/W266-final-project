#include <iostream>

int Nadji(std::string s, std::string f, int poz)
{
    for(int i=poz;i<=s.length()-f.length();i++)
    {
        std::string tmp(s.substr(i,f.length()));
        if(tmp==f) return i;
    }
    return -1;
    
}
