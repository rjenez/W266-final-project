std::string IzokrenutaRijec (std::string rijec)
{

	std::string obrnuti;
	for(int i = rijec.length() - 1; i >= 0;  i--) {

	}

}

bool DaLiJeSlovo(char znak)
{
	if((znak >= 'A' && znak <= 'Z') || (znak >= 'a' && znak <= 'z')) return true;
	return false;
}
