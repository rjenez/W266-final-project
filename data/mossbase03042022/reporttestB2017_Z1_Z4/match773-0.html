{
	int brojac(0);

	for(int i=0; i<s1.size(); i++) {

			for(int k=0; k<s.size(); k++) {

				brojac=0;
				int f(j);
				int m(k);
				while(s1.at(i).at(f)==s.at(k) && k<s.size() && f<s1.at(i).size()) {
					brojac++;

					k++;
					f++;

}
