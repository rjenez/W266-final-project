String ObrniRijec(String s,bool Polovina)
{
	if(Polovina == false){
		String naopako;
		naopako.resize(s.length());
	//	int k(0);
		for(int i=s.length()-1; i>=0; i--){
			naopako.at(s.length()-1-i)=s.at(i);
		}
		return naopako;
	}
	else {
		String naopako;
		for(int i=0; i<=(s.length()-1)/2; i++){
			naopako.resize(i+1);
			naopako.at(i)=s.at(i);
		}
		return ObrniRijec(naopako,false);
	}
}
