std::vector<std::string>Pretvorba(std::string s) {
	std::vector<std::string>v;
    v.push_back(" ");
	for (int i=0; i<s.size(); i++) {
    if (s[i]==' ') v.push_back(" ");
    else v[v.size()-1]=v[v.size()-1]+s[i];
	}
  return v;	
}
std::string ObrnutaPretvorba(std::vector<std::string>v) {
	std::string s;
	for (int i=0; i<v.size(); i++) {
		s+=v[i];
		s+=" ";
	}
	return s;
}
