std::string ObrniFraze(std::string recenica, std::vector<std::string> SpisakRijeci){

        for(int j=0;j<recenica.length();j++){
           int k=j;
           while(k<recenica.length() && (k-j)<SpisakRijeci[i].length() && recenica[k]==SpisakRijeci[i][k-j]){
               k+=1;
           }

}
