bool daLiJeValidno(StringVek v) {
	for(int i=0; i<v.size(); i++) {
		for(int j=0; j<v[i].size(); j++) if(v[i][j]<'A' || v[i][j]>'z' || (v[i][j]>'Z' && v[i][j]<'a')) return false;
	}
	return true;
}
