bool slovo(char a)
{
	if((a >= 'A' && a <= 'Z') ||(a >= 'a' && a <= 'z'))
		return true;
	return false;
}

string ObrniFraze (string s, vektor fraze)
{
	int z;
	for(int i=0; i<fraze.size(); i++)
	{
		for(int j=0; j<s.size(); j++)
		{
			if(s.substr(j, fraze[i].size() ) == fraze[i])
				{
					z=j;
					for (int k  = fraze[i].size()-1 ; k >= 0; k--)
					{
						s[j] = fraze[i][k];
						j++;
					}
					j=z;
				}
		}
	} return s;
}
