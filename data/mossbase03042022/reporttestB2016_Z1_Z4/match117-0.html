std::string ObrniFraze(std::string s, std::vector<std::string> v) {


	for(int i=0; i<v.size();i++){//kroz vektor rijeci

			for(int k=0; k<s.size();k++){//kroz recenicu

				int br=0;
				while(s[k]==v[i][j] && k<s.size() && j<v[i].size()){//bez jednako
					k++;
					j++;
					br++;
				}
					if(j==v[i].size()){
						int q=--k;
						char pom;
						while(p<q){
							pom=s[p];
							s[p++]=s[q];
							s[q]=pom;
							q--;
						}//kraj while
					}//kraj ifa
					j=j-br;
					}//k petlja
					break;
				}//kraj j petlje
				
			}//kraj i petlje
		
		
		return s;
}
