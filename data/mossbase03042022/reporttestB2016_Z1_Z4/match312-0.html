string ObrniString(string s)    /*Obrtanje fraze*/
{
	string s1;
	for(int i=0; i<s.length(); i++) {
		s1.push_back(s[s.length()-1-i]);
	}
	return s1;
}

string RijecUPigLatin(string s)    /*Pretvaranje rijeci u PigLatin*/
{

	return s;
}
