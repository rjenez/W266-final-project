int Rijec_trazi(const std::string &T, const std::string &P,int i = 0)
{

        while (j < m && T[i+j] == P[j]) {
            ++j;
        }
        if (j == m) {

        }
    }
    return -1;
}

bool ima_li_ne_slova (std::string s)

}
