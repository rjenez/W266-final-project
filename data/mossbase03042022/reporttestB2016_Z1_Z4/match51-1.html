string ObrniFraze(string s,vector<string>vs){
	for(int i=0;i<int(vs.size());i++){
		string pom(vs[i]);
		for(int j=0;j<int(s.length());j++){
			for(int k=0;k<int(pom.length());k++){
				if(pom[0]!=s[j])break;
				while(pom[k++]==s[j++]){
					if(k==int(pom.length())){
					    j--;
					    int pom1(j),pom2;
					    pom2=int(j-pom.length()+1);
						while(pom1>pom2){
						char temp(s[pom2]);
						s[pom2]=s[pom1];
						s[pom1]=temp;
							pom1--;pom2++;
						}
					}
				}
			}
		}
	}
		
	return s;
}
