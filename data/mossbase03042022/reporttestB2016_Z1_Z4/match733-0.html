std::string ObrniFraze(std::string s, std::vector<std::string> baza){

		for(int j=0; j<s.length() ; j++){
			if(	(s.substr(j, baza[i].length() ) == baza[i]) ) {
				for(int k=j ; k<j+baza[i].length() ; k++){
					s[k]=baza[i][baza[i].length() - 1 - k + j];
				}
			}
		}
	}
	return s;
}
