std::string Obrcem(std::string s) {
	char temp;
	for(int i=0; i<s.length()/2; i++) {
		temp=s[i];
		s[i]=s[s.length()-1-i];
		s[s.length()-1-i]=temp;}
		return s;
}
int prostor(std::string s) {
	int brojac(0);
	for(int i=0; i<s.size(); i++) brojac++;
	return brojac;
}
