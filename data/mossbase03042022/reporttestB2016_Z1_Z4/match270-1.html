std::string ObrniFraze(std::string s, std::vector<std::string>v) {


	for (int i=0; i<v.size(); i++) {

			if ((v[i]==s.substr(j, v[i].size()))) {
				for (int k=0; k<v[i].size()/2; k++) {
					char pomocni(s[j+k]);
					s[j+k]=s[j+v[i].size()-k-1];
					s[j+v[i].size()-k-1]=pomocni;
				}
			}
			else continue;
		}
	}

	return s;
}
