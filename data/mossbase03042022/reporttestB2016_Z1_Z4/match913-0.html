bool PravaRijec(std::string s){
	int suma=0;
	for(int i=0;i<s.size();i++){
		if((s[i]>='A' && s[i]<='Z') || ( s[i]>='a' && s[i]<='z'))suma++;
	}
	if(s.size()==suma) return 1;
	return 0;
}
