std::string PigLatin (std::string s){

std::string Obrni (std::string s){
	for (int i(0); i<s.size()/2; i++){
		char temp;

		s.at(i)=s.at(s.size()-1-i);
		s.at(s.size()-1-i)=temp;
	}
	return s;
}
