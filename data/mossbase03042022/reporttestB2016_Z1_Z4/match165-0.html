bool DaLiJeSlovo(char s){
	return ((s >= 'a' && s <= 'z') || (s >= 'A' && s <= 'Z'));
}
bool ValidanSpisak(std::vector<std::string> s){
	
	for(auto x : s){
		for(auto y : x){
			if(!DaLiJeSlovo(y)) return false;
		}
	}
	
	return true;
}

					recenica[i] = fraza[j];
					i++;
				}

		}
	}

	return recenica;
}
