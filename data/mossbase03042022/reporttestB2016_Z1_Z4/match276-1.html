string okreni(string s)
{
	string pom;
	for(int i=s.size()-1; i>=0; i--) pom+=s[i];
	return pom;
}

bool dalijeslovo(char sl)
{
	if(toupper(sl)>='A' && toupper(sl)<='Z') return true;
	return false;
}

string pig(string s)
{
	string pom;
	int k=1;
	while(k!=s.size())
	{
		pom+=s[k];
		k++;
	}
	pom+=s[0];
	pom+="ay";
	return pom;
}
