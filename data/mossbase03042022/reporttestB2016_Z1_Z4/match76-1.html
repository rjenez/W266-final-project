std::string ObrniFraze(std::string s, std::vector<std::string> f){

			if(s.substr(i,f[j].size())==f[j]){
				int k=i, r=i+f[j].length()-1;
				while(k<r){
					char temp=s[k];
					s[k]=s[r];
					s[r]=temp;

					r--;
				}
			}

	}
	return s;
}
