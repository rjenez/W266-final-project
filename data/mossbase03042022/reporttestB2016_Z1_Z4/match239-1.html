string zarotiraj(string s) {
	for(int i=0;i<s.length()/2;i++) {
		char temp=s[s.length()-1-i];
		s[s.length()-1-i]=s[i];
		s[i]=temp;
	}


	return s;
}

				{

					s1+=zarotiraj(v[i]);

		}
	}


	return s;
}
