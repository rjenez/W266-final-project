std::string ObrniFraze(std::string s, std::vector<std::string>v){
	for(int i=0;i<v.size();i++){
		std::string x=v[i];
		for(int j=0;j<s.size();j++){
			int k=0;
			int r=j;
			while(s[r]==x[k]){
				if(r==s.size()) break;
				k++;
				r++;
			}
			if(k==x.size()){
				r--;
				for(int l=0;l<x.size();l++){
					s[r]=x[l];
					r--;
				}
				
			}
		}
	}
	return s;
}
