bool StepenDvojke(int broj)
{
	const double Eps(1e-10);
	for(int i=0; i<=broj; i++) {

		if(std::fabs(broj-pow(2,i))<=Eps*(std::fabs(broj)+std::fabs(pow(2,i))))

	}
	return false;
}
