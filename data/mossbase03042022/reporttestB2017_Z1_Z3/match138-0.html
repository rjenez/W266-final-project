typedef std::vector<int> Vektor;

bool StepenDvojke(int n)
{
	if(n==0) return false;
	while(n!=1) {
		if(n%2!=0) return false;
		n=n/2;
	}
	return true;
}
Dek IzbaciIste(Dek d)
{
	for(int i=0; i<d.size(); i++) {
		for(int j=i+1; j<d.size(); j++) {
			if(d.at(i)==d.at(j)) {
				for(int k=j; k<d.size()-1; k++) {
					d.at(k)=d.at(k+1);
				}
				j--;
				d.resize(d.size()-1);
			}
		}
	}
	return d;
}
