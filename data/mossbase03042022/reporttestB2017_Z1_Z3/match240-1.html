bool DaLiJeStepenDvojke(int n)
{
	if (n==0) return false;
	while (n!=1) {
		if (n%2!=0)
			return false;
		n=n/2;
	}
	return true;
}

				while (i+1<vektor.size() && DaLiJeStepenDvojke(vektor.at(i+1))==true && vektor.at(i)<=vektor.at(i+1)) {
