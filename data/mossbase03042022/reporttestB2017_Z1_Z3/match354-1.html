bool StepenDvojke(int broj) {
	if (broj < 1) return false;

}

bool MozeIciDesno(int lijevi, int desni, Smjer s) {
	if (!StepenDvojke(lijevi) || !StepenDvojke(desni)) return false;

		return lijevi >= desni;

	return false;
}
