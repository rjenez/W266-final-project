bool DaLiJeStepenDvojke(int broj){
	if(broj<=0)return false;
	if(broj==1 || broj==2) return true;
	while(broj!=0){
		broj/=2;
		if(broj==1) return true;
		if(broj%2!=0) return false;
	}
	return true;
}
