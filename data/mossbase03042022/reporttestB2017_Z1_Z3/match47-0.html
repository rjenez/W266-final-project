bool StepenDvojke (long long int n)
{
	if(n <= 0) return false;
	int brojac(0);
	long long int broj = n;

		n /= 2;
		brojac++;
	}
	brojac--;
	if(broj == std::pow(2, brojac)) return true;
	else return false;
}
