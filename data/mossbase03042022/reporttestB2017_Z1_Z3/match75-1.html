bool StepenDvojke(int broj)
{
	if (broj==0 || broj<0)
		return false;
	if (broj==1)
		return true;
	while(broj!=0)
	{
		if (broj%2!=0)
			return false;
		broj/=2;
		if (broj==1)
			break;
	}
	return true;
}
