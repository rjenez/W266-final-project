bool DaLiJeStepenDvojke( int n ) {

	while ( n > 2 ) {

		n = n / 2;
	}
	if( n == 2 ) return true;
	else return false;
}

			pol = i;
		}
	}

		v.resize(pol);
		return v;
	}
	if( pol < 2 && n == false ) {
		v.resize(0);
	    return v;
	}
	else return v;
}
