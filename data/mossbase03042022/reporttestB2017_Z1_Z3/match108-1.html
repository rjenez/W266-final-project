#include <iostream>

enum Smjer {Rastuci=1,Opadajuci=2};

typedef std::vector <int> Vektor;

bool DaLiJeStepenDvojke(int x) {    //funkcija za provjeravanje brojeva da li su stepeni dvojke
    while(x%2==0) {
        x/=2;
    }
    if(x==1) return true;
    return false;
}
