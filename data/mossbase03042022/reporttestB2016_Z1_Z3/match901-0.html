#include <iostream>
#include <vector>



std::vector<std::vector<double>> RastuciPodnizovi(std::vector<double> niz)
{
	std::vector<std::vector<double>> v(0);
	int n = niz.size();
	for(int i=0; i<n; i++)
	{
		int duzina = 1;
		for(int j=i+1; j<n; j++)
		{
			if(niz.at(j) >= niz.at(j-1))
			{
				duzina++;
			}
			else
			{
				break;
			}
		}
		if(duzina >= 2)
		{
			std::vector<double> temp(0);
			for(int j=i; j<i+duzina; j++)
			{
				temp.push_back(niz.at(j));
			}
			v.push_back(temp);
			i += duzina -1;
		}
	}


	return v;
}
