	vector<vector<double>> RastuciPodnizovi(vector<double> v){
	vector<vector<double>> m;
	for(int i=0;i<v.size();i++){
		if(i==v.size()-1) break;
		if(v.at(i)<=v.at(i+1)){
			int j=i;
			vector<double> a;
			while(v.at(j)<=v.at(j+1) && j<v.size()){
				a.push_back(v.at(j));
				j++;
				if(j==v.size()-1) break;
			}
			a.push_back(v.at(j));
			m.push_back(a);
			i=j;
		}
	}
	return m;
}
