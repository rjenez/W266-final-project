vector<vector<double>> RastuciPodnizovi(vector<double> v){

	int k(0);
	for(int i=0;i<v.size()-1;i++){
		if(v[i]<=v[i+1]){
			k++;
			rp.resize(k);
			while(i<v.size()-1 && v[i]<=v[i+1]){

				i++;

			rp[k-1].push_back(v[i]);
		}
	}

}
