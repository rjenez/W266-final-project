Matrica RastuciPodnizovi(vector<double> v) {
	Matrica mat;
	vector<double> pom;
	
	for (int i = 0; i < v.size() - 1; i++) {
		pom.resize(0);
		int p(i);
		
		if (v[i] <= v[i + 1]) {
			while (v[i] <= v[i + 1] && i < v.size() - 2) i++;
			
			int k(i);
			if (i == v.size() - 2) k++;
			//k++;
			
			for (int j = p; j <= k; j++) {
				pom.push_back(v[j]);
			}
			
			mat.push_back(pom);
		}
	}
	
	/*cout << "Maksimalni rastuci podnizovi: \n";

		for (int j = 0; j < mat[i].size(); j++) {

	return mat;
}
