vector<vector <double>> RastuciPodnizovi(vector<double> v)
{
	vector<vector <double>> m;
	
	int i(0);

	
	while(i<v.size())
	{
		vector<double> vrste;
		
		
		if(i!=v.size()-1 && v[i]<=v[i+1]){
		while(i!=v.size()-1 && v[i]<=v[i+1])
		{


			vrste.push_back(v[i]);
			
			i++;
			
			if((i+1)<=v.size()-1 && v[i]>v[i+1] && v[i]>=v[i-1]) vrste.push_back(v[i]);
			
			if(i==v.size()-1 && v[i]>=v[i-1])
			{
				vrste.push_back(v[i]);
				break;
			}
		}
		
		
		m.push_back(vrste);
		}
		else i++;
		
	
	}
	
	
	
	
	return m; 
}
