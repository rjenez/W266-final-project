std::vector<double> MaxPodniz(std::vector<double> v, int odakle, int dokle){
	std::vector<double> noviVektor;
	if(dokle-odakle<=1) {
		noviVektor.push_back(0);
	return noviVektor;
	}
	noviVektor.push_back(v[odakle]);
	for(int i=odakle+1; i<dokle; i++){
		if(v[i]>=v[i-1])
			noviVektor.push_back(v[i]);
		else
			return noviVektor;
	}
	return noviVektor;
}
