	for(int i = 0; i < mat.size(); i++) {
		int temp = i;
		if(mat[i].size() == 1) {
		for(int j = i + 1; j < mat.size(); j++){
			mat[i] = mat[j];
			i++;
		} 
		
		mat.resize(mat.size() - 1);
		i = temp;
		i--;
		}
	}
	return mat;
}
