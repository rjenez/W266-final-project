	{
		if(pom.size() == 0) pom.push_back(v[i]);
		else if(pom[pom.size() - 1] <= v[i]) pom.push_back(v[i]);
		else 
		{
			if(pom.size()>=2) vv.push_back(pom);

	if(pom.size()>=2) vv.push_back(pom);
	return vv;
}
