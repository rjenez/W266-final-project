bool jeInt (double x){
    if(x < 0) x = -x;
    return x == std::floor(x);
}

matrix RastuciPodnizovi(std::vector<double> v){
    matrix ret;
    std::vector<double> tmp;
    unsigned int len = v.size();
    if(len < 2)
        return ret;
    unsigned int p, q;
    for(unsigned int i = 0; i < len-1; i++){
        p = i;
        while(v[i] <= v[i+1] && i < len-1){
            i++;
            if(i == len - 1)
                break;
        }
        q = i;
        for(int j = p; j <= q; j++){
            tmp.push_back(v[j]);
        }
        if(q - p > 0)
            ret.push_back(tmp);
        tmp.resize(0);
    }
    return ret;
}
