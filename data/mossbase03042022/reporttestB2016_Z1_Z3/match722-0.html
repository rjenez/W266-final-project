#include <iostream>

typedef std::vector<std::vector<double>> matrica;

matrica Izbaci( matrica c){
	for(int i=0; i<c.size(); i++){
		if(c[i].size()==0){
			 c.erase(c.begin()+i);
			 i--;
		}
	}
	return c;
}
