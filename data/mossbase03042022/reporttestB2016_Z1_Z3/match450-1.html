std::vector<std::vector<double>> RastuciPodnizovi(std::vector<double> niz) {
	std::vector<std::vector<double>> rez;
	std::vector<double> tmp;
	int i(0);
	while(i<niz.size()-1) {


		while(i<niz.size()-1 && niz[i]<=niz[i+1]) {
			tmp.push_back(niz[i]);
			i++;
		}
		tmp.push_back(niz[i]);
		if(tmp.size()>1) rez.push_back(tmp);
		tmp.resize(0);


		i++;
	}
	return rez;
}
