vector<vector<double> > RastuciPodnizovi(vector<double> v)
{
	vector<vector<double> > m;
	int br;
	int di, k(0);


	for(int i=0;i<v.size()-1;i++)
	{
		if(v[i]<=v[i+1])
		{
			m.resize(m.size()+1);
			br=0;di=i;
			while(v[i]<=v[i+1] && i<v.size()-1) { br++; i++; if(i==v.size()-1 || i==v.size()) break;}

			k++;
		}
	}
	return m;
}
