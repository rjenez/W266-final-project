Matrica RastuciPodnizovi(vektor v)
{

	int k(0), i, j;
	if(v.size()<2)
		return temp;
	for(i=0; i<int(v.size()-1); i++)

		{
			if(v.at(j)>v.at(j+1))
				break;
			while(v.at(j)<=v.at(j+1) && j < int(v.size()-1))
			{
				temp[k].push_back(v[j]);
				j++;

		}

	}

}
