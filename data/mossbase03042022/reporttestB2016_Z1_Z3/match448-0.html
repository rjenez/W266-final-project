#include <iostream>
#include <vector>
#include <cmath>

typedef std::vector<std::vector<double>> Matrica;
Matrica RastuciPodnizovi(std::vector<double> v){
    Matrica m;
   if(v.size() < 2) return m;
   int redovi(0);


   for(int i(0);i < v.size()-1;i++){
       
       if(v[i] <= v[i+1]){
           redovi++;
           m.resize(redovi);
           
           while(i < v.size()-1 && v[i] <= v[i+1] ){
               
               m[redovi-1].push_back(v[i]);
               i++;
               
           }
           m[redovi-1].push_back(v[i]);
       }
       
   }   
       
    return m;
}
