#include <iostream>
#include <vector>


typedef std::vector<std::vector<double>> matrica;
matrica RastuciPodnizovi(std::vector<double> v){
	int i=0,br=0,vel=0;
	matrica rez;
	while(i<v.size()-1){
		if(v[i]<=v[i+1]){
			br=0;
			for(int j=i;j<v.size()-1;j++)if(v[j]<=v[j+1]){
				if(j+1>=v.size())break;
				br++;
			}
			else break;
			rez.push_back({});
			for(int k=i;k<=(i+br);k++)rez[vel].push_back(v[k]);
			vel++;
			i+=br;
		}
		if(i>v.size())break;
		i++;
	}
	return rez;
}
