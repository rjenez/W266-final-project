                i++;
            }
            mat.push_back(v2);
            return mat;
        }
        if(prvi<=v[i]){
            v2.push_back(v[i]);
            prvi=v[i];
        }
        else{

            v2.clear();
            v2.push_back(v[i]);
            prvi=v[i];
        }
    }
    if(v2.size()>1) mat.push_back(v2);
    return mat;
}
