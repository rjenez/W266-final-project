Matrica RastuciPodnizovi(std::vector<double> niz) {
    Matrica rez;
    int red(0);
    bool prviPut(true);
    for (int i = 1; i < niz.size(); i++) {
        if (niz[i - 1] <= niz[i]) {
            if (prviPut) {
                rez.resize(red + 1);
                rez[red].push_back(niz[i - 1]);
                prviPut = false;
            }
            rez[red].push_back(niz[i]);
        }
        else {
            prviPut = true;

                red++;
        }
    }
    return rez;
}
