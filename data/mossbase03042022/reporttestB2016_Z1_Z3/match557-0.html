			vek.push_back(v[i]);
			continue;
		}


		if((i+1)<int(v.size()) && i-1>=0 && ((v[i]>=v[i-1] && v[i]<=v[i+1]) || (v[i]<=v[i-1] && v[i]<=v[i+1]))){
			vek.push_back(v[i]);
			continue;
		}

}
