matrica RastuciPodnizovi(vec v){
    matrica m;
    m.resize(1);
    int k(0);
    for(int i=0; i < v.size(); i++){
        m[k].push_back(v[i]);
        


        if (i < v.size()-1 && v[i] > v[i+1]) {
            while (i < v.size()-1 && v[i] > v[i+1]) {
                i++;
            }
            k++;
            m.resize(m.size()+1);
            m[k].push_back(v[i]);
        }
    }

    matrica mat;
    k = 0;
    
    for (int i = 0; i < m.size(); i++) {
        if (m[i].size() > 1) {
            mat.resize(mat.size() + 1);
            for (int j = 0; j < m[i].size(); j++) {
                mat[k].push_back(m[i][j]);
            }
            k++;
        }
    }
    
    return mat;
}
