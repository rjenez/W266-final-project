vector<vector<double>> RastuciPodnizovi(vector<double> v)
{
	vector<vector<double>> matrica;

	for(int i=0; i<v.size();i++) {
		vector<double> niz;
		if(i+1<v.size() && v[i]<=v[i+1]) {
			while(i+1<v.size() && v[i]<=v[i+1]) {
				niz.push_back(v[i]);
				i++;
			}
			niz.push_back(v[i]);
			matrica.push_back(niz);
		}
	}

	return matrica;
}
