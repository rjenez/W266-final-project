	std::vector<std::vector<double>> matrica;

				int j(i);
				std::vector<double> podniz;


				while(j<v.size()-1 && (v[j] <= v[j+1])) {
						podniz.push_back(v[j]);
						j++;
				}
						podniz.push_back(v[j]);

						matrica.push_back(podniz);
						podniz.resize(0);

				}
	}
	return matrica;
}
