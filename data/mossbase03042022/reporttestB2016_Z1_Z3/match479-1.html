#include<iostream>
#include<vector>

vector<vector<double>>RastuciPodnizovi(vector<double>v){
	vector<vector<double>> f;
	for(int i=0;i<v.size();i++){
		vector<double>g(1,v[i]);
		for(int j=i+1;j<v.size();j++){
			if((v[j]>=v[i])){
			g.push_back(v[j]);
			i++;
			}
			else break;
		}
		if(g.size()>1)
		f.push_back(g);
	}


	return f;
}

			i++;
			}
			else break;
		}
		if(vek.size()>1)
		h.push_back(vek);
	}
	return h;
}
