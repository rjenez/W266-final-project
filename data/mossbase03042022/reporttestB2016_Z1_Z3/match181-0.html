Matrica RastuciPodnizovi(std::vector<double> niz){
	Matrica vracena;
	for(int i=1; i<niz.size(); i++){
		std::vector<double> pomocna;
		bool istina(true);
		if(niz[i-1]<=niz[i]){
			while(i<niz.size() && niz[i-1]<=niz[i]){
				if(istina){
					pomocna.push_back(niz[i-1]);
					pomocna.push_back(niz[i]);
					istina=false;
					i++;
				} else {
					pomocna.push_back(niz[i]);
					i++;
				}
			}
		}

}
