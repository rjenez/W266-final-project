{

                isti++;

            }
        }

        for(j=0; j<v3; j++) {
            if(niz1[i]==niz3[j]) {
                isti++;
                break;
            }
        }

        if(isti==1) k++;
        isti=0;
        i++;
    }
    isti=0;
    i=0;

                        isti++;

                    }
                }
            }
        }


        if(isti==1)
            k++;
        isti=0;
        i++;
    }


    return k;

}
