int izbrisi_duplikate(int niz[],int vel){
	
	
	int i=0,j=0,k=0,br=0,temp=vel;
	while(br<temp){
	for(i=0;i<vel;i++)
		for(j=i+1;j<vel;j++){
			if(niz[j]==niz[i]){
				for(k=j;k<vel-1;k++){
					niz[k]=niz[k+1];
				}
				vel--;
			}
			else j++;
			}
		
		br++;
		
	}
	return vel;
}
