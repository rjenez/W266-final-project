int izbaci(int niz[],int vel,int broj)
{
	int i;
	for(i=0; i<vel; i++) {
		if(niz[i]==broj) {
			int j;
			for(j=i; j<vel-1; j++) {
				niz[j]=niz[j+1];
			}
			vel--;
			i--;
		}
	}
	return vel;
}
