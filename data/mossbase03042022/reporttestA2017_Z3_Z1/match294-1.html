int ISTI(int *niz,int duzina,int i)
{
	int j,LOGICKI=1;;
	if(i!=0) {
		for(j=0; j<duzina; j++) {
			if(*(niz+i)==*(niz+j) && i!=j) {
				LOGICKI=0;
				break;
			} else
				LOGICKI=1;
		}
	}
	return LOGICKI;
}
