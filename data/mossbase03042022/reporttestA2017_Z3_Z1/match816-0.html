			if(nizB[i]==NoviNiz[j])break;

	}

		JavljaUJednomNizu=0;
		for(j=0;j<velA;j++){
			if(nizA[j]==NoviNiz[i]){
				JavljaUJednomNizu++;
				break;
			}
		}

			if(nizB[j]==NoviNiz[i]){

				break;
			}

				break;
		   }
