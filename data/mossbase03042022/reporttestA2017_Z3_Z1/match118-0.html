#include <stdio.h>

int postoji_u_nizu(int* niz, int duzina, int broj)
{
	int i;
	for(i=0; i<duzina; i++) {
		if(broj==niz[i]) return 1;
	}
	return 0;
}

int izbaci_iste(int *niz, int duzina)
{
	int i,j,k;
	for(i=0; i<duzina-1; i++) {
		for(j=i+1; j<duzina; j++)
			if(niz[i]==niz[j]) {
				for(k=j; k<duzina-1; k++)
					niz[k]=niz[k+1];
				duzina--;
				j--;
			}
	}
	return duzina;
}
