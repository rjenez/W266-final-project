int ispitaj (int niz[], int vel, int broj){
	int i;
	for (i=0; i<vel; i++){
		if (broj==niz[i]) return 1;
	}
	return 0;
}

				}
			}
		}
		if (ima3!=ima2) brojac++;
		for (j=i+1; j<vel1; j++){
			if (niz1[i]==niz1[j]) {
