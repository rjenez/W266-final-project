void prebroj(int* niz, int velicina, int p[] , int w[])
{

	int t;

	for(i=0; i<velicina; i++)

		while(1)
		{
			if( p[k]==0 && w[k]==0 )
			{
				p[k]=*(niz+i);
				w[k]++;
				break;
			}
			else if(p[k]==*(niz+i))
			{
				t=0;
				for(j=i-1; j>-1; j--)
				{
					if(  (*(niz+i)) == (*(niz+j))  )t=1;
				}

				if(!t)
				{
					w[k]++;
				}
				break;
			}
			else
			{
				k++;
			}

		}
	}


}
