#include <stdio.h>




void IzbaciDuplikate (int *niz, int *velicina)
{
    
    int i,j,k;
    
   for(i = 0;i<*(velicina);i++) {
       for(j = i+1;j<*(velicina);j++) {
           
           if((niz[i])==((niz[j]))) {
               
               for(k=j;k<*velicina;k++) {
                   (niz[k])=niz[k+1];
               }
               (*velicina)--;

               j--;
           }
           
       }
   }
   
   
}
