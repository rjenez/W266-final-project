#include <stdio.h>

/* funkcija provjerava da li se proslijedeni broj N nalazi u proslijedenom nizu */


int pomocna(int *niz, int vel, int N)
{
	int i;
	for(i=0; i<vel; i++) {
		if(niz[i]==N) return 1;
	}
	return 0;
}
