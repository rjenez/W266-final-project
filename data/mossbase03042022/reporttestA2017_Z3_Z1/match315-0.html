int dva_od_tri(int niz1[], int duzina1, int niz2[], int duzina2, int niz3[], int duzina3)
{
	int i, j, x, y=0, t=0;

	for(i=0; i<duzina1; i++) {
		x=0;
		t=0;

		for(j=i-1; j>=0; j--) {
			if(niz1[i]==niz1[j]) t=1;
		}

		if(t==0) {
			j=0;

			while(j<duzina2) {
				if(niz1[i]==niz2[j]) {
					x++;
					break;
				}
				j++;
			}

			j=0;
			while(j<duzina3) {
				if(niz1[i]==niz3[j]) {
					x++;
					break;
				}
				j++;
			}

			if(x==1) y++;
		}
	}

	for(i=0; i<duzina2; i++) {
		t=0;
		x=0;
		j=0;

		for(j=i-1; j>=0; j--) {
			if(niz2[i]==niz2[j]) t=1;
		}
		if(t==1) continue;
		j=0;

		while(j<duzina1) {
			if(niz2[i]==niz1[j]) {
				t=1;
				break;
			}
			j++;
		}

		if(t==0) {
			j=0;
			while(j<duzina3) {
				if(niz2[i]==niz3[j]) {
					t=1;
					break;
				}
				j++;
			}

			if(t==1) y++;
		}
	}

	return y;
}
