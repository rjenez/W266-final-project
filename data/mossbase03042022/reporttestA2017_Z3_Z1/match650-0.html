{

			if(niz1[i]==niz3[j]) {
				ponavljanje++;
				break;
			}
		}
		if (ponavljanje==2) {
			int ponavljanje_cifre=0;
			for(j=0; j<broj_clanova; j++)
				if (niz1[i]==clanovi[j]) ponavljanje_cifre++;
			if (ponavljanje_cifre==0) {
				clanovi[broj_clanova]=niz1[i];
				broj_clanova++;
			}
		}
	}

	for (i=0; i<vel2; i++) {
		int ponavljanje=1;
		for(j=0; j<vel1; j++) {
			if(niz2[i]==niz1[j]) {
				ponavljanje++;
				break;
			}
		}

			if(niz2[i]==niz3[j]) {
				ponavljanje++;
				break;
			}
		}
