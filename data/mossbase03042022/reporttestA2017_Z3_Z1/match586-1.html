int dva_od_tri(int a[],int vela,int b[],int velb,int c[],int velc){

		}
		
			if(izadji==0){

				}

					}
				}

			}
	}
		for(i=0;i<velb;i++){

		pbr=0;
		
		for(j=i-1;j>=0;j--){
			if(b[j]==b[i] ){
				izadji=1;
				break;
			}
		}
		if(izadji==0){
			for(j=0;j<velc;j++){
				napolje=0;
				for(z=j-1;z>0;z--){
					if(c[j]==c[z])

					pbr=1;

					}

					}
				}
			}
		}
	
	return br;
}
