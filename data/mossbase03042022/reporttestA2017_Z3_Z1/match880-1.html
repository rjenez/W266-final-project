int dva_od_tri(int *niz1, int vel1, int * niz2, int vel2, int * niz3, int vel3)
{
	int ii, jj;


	int i, j, k;
	int br=0;
	for(i=0; i<vel1; i++) {
		for(ii=0; ii<i; ii++) {
			if(niz1[ii]==niz1[i])
				ii=i;
		}
		if(ii == i+1)
			continue;
		for(j=0; j<vel2; j++) {


			if(niz1[i]==niz2[j]) {

					}
				}
				break;
			}
		}
	}

				br++;

					if(niz2[j]==niz1[i]){
						br--;
						break;
					}
				}
				break;
			}
		}
	}
	for(i=0; i<vel1; i++) {
		for(ii=0; ii<i; ii++) {
			if(niz1[ii]==niz1[i])
				ii=i;

					if(niz1[i]==niz2[j]){
						br--;
						break;
					}
				}
				break;
			}
		}
	}

	return br;

}
