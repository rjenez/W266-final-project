#include <stdio.h>


int NemaBroja(int niz[],int suma, int br)
{
	int i,k;
	for(i=0; i<suma; i++) {
		if(niz[i]==br) {
			k=0;
			return k;
		}
	}
	k=1;
	return k;
}
int NemaDuplikata(int niz[], int suma, int br, int i)
{
	int j,k;
	for(j=0; j<suma; j++) {
		if(j>i && niz[j]==br) {
			k=0;
			return k;
		}
	}
	k=1;
	return k;
}
