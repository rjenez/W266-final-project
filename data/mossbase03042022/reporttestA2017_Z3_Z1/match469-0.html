#include <stdio.h>

//funkcija koja izbacuje iste elemente niza



int check_for_equal(int niz[], int velicina)
{
	int i, j;
	for(i=0; i<velicina; i++) {
		for(j=i+1; j<velicina; j++) {
			if(niz[i]==niz[j]) {
				niz[j]=niz[velicina-1];
				j--;
				velicina--;
			}
		}
	}
	return velicina;
}
