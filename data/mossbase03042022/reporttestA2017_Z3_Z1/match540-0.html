int izbaci_iste(int niz[], int vel_niza)
{
	int i,j;

	for(i = 0; i < vel_niza; i++) {
		for(j = i+1; j < vel_niza; j++) {
			if(niz[i] == niz[j]) {
				niz[j]=niz[vel_niza-1];
				j--;
				vel_niza--;
			}
		}
	}
	return vel_niza;
}
