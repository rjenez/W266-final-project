#include<stdio.h>


int pomocna(int niz[],int vel,int broj){
    int *k=niz; int true=0;
    while(k<niz+vel){
        if(*k==broj){
            true=1;
            break;
        }
        k++;
    }
    return true;
}
void prepisi(int niz1[],int vel1,int niz2[]){
    int i;
    for(i=0;i<vel1;i++){
        niz2[i]=niz1[i];
    }
}
int izbaciiste(int niz1[],int vel){
    int i,j,k;
    for(i=0;i<vel-1;i++){
        for(j=i+1;j<vel;j++){
            if(niz1[i]==niz1[j]){
                for(k=j;k<vel-1;k++){
                    niz1[k]=niz1[k+1];
                }
                vel--;
                j--;
            }
        }
    }
    return vel;
    
}
