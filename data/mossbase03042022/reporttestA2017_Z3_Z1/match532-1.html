int kopiraj(int niz1[], int vel1, int niz2[])
{
	int i, j, brojac=0;
	for(i=0; i<vel1; i++) {
		for(j=0; j<brojac; j++) {
			if(niz1[i]==niz2[j]) {
				break;
			}
		}
		if(j==brojac) {
			niz2[brojac]=niz1[i];
			brojac++;
		}
	}
	return brojac;
}
