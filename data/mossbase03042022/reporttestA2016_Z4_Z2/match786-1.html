void unesi(char niz[], int velicina){
	int i=0;
	char znak = getchar();
	if (znak == '\n') znak=getchar();
	while (i < velicina - 1 && znak != '\n') {
		niz[i] = znak;
		i++;
		znak = getchar();
	}
	niz[i]='\0';
}

	/*Broji koliko puta se ponavljaju odredjene rijeci*/
	while (*s1 != '\0') {
		i=0;
		while (znak != '~'){

		s1++;
	}

			}
		}
		if (index != -1)
		{
			brojac++;
			/*Dodaje slovo u niz. Index+32 jer ne pocinjemo od ASCII-0*/
