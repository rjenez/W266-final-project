void unesi(char s[], int n)
{
	int i=0;
	char znak=getchar();
	if(znak=='\n') znak=getchar();
	while(i<n-1 && znak!='\n') {
		s[i]=znak;
		i++;
		znak=getchar();
	}
	s[i]='\0';
}

char *kopiraj(char *s,const char *r)
{
	char *pocs=s;
	while(*r) {
		*s++=*r++;
	}
	*s='\0';
	return pocs;
}

int broj_ponavljanja(char *s, char c)
{
	int br=0;
	while(*s!='\0') {
		if(*s==c)
			br++;
		s++;
	}
	return br;
}

char *dodaj(char *s, const char *r, int n)
{

		s++;
	}

		if(!(*s++ = *r++))
			return pocs;
	}
	*s='\0';
	return pocs;
}
