int DajBrojPonavljanja (char* tekst, char znak)
{
    int brojac = 0;
    while (*tekst != '\0')
    {
        if (*tekst++ == znak)
        {
            brojac++;
        }
    }
    return brojac;
}

int DajSljedeciBroj (char* tekst, int minPonavljanja, int maxPonavljanja)

    while (*tekst != '\0')
    {

}

int DajPozicijuUtablici (char* tablica, char l)
{
    int pozicija = 0;
    while (*tablica != '\0')
    {
        if (*tablica == l)
            return pozicija;
        pozicija++;
        tablica++;
    }
    return pozicija;
}
