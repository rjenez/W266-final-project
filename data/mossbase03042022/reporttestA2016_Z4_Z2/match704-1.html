void izbaciIzNiza(char znak, char *niz){

		if(*p == znak){
			temp = p;
			while(*p != '\0'){
				*p = *(p+1);
				p++;
			}
			p = temp - 1;
		}
		p++;
	}
}
void swap(char a, char b, char *niz){
	char *p = niz;
	//funkcija mijenja mjesta dva chara u niz
	while(*p != '\0'){
		if(*p == a)
		*p = b;
		else if(*p == b)
		*p = a;
		p++;
	}	
	
}
