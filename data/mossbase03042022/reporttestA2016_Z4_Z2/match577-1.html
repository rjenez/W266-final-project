#include <stdio.h>



void unesi(char niz[], int velicina) {
	char znak = getchar();
	
	if (znak == '\n') {
		znak = getchar();
		
	}
	
	int i = 0;
	while(i<velicina - 1 && znak != '\n') {
		niz[i] = znak;
		i++;
		znak = getchar();
		
	} 
	
	niz[i] = '\0';
	
}


int izbroji(char* ptr, char znak) {
	int br = 0;
	
	while(*ptr != '\0') {
		if (*ptr == znak) {
			br++;
			
		}
		ptr++;
	}
	
	return br;
}

		for (j=i; j<i; j++) {

				temp = *(kod+i);
				*(kod + i)=*(kod+j);
				*(kod + j)= temp;
