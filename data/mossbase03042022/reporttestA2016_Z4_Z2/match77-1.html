void sortiraj(int *niz, int *sortirani, int *indeks) {
	int i, j, pom;
	


	for (i=0; i<127; i++) {
		for (j=i; j<127; j++) {
			if (niz[i]<niz[j]) {
				pom=niz[i];
				niz[i]=niz[j];
				niz[j]=pom;
				pom=indeks[i];
				indeks[i]=indeks[j];
				indeks[j]=pom;
			}
			else if (niz[i]==niz[j] && indeks[i]>indeks[j]) {
				pom=indeks[i];
				indeks[i]=indeks[j];
				indeks[j]=pom;
			}
		}
	}
}



void konevertuj(char *s, char *sifrarnik) {

	int brojac;
	
	while (*p!='\0') {
		brojac=0;
		sifra=sifrarnik;
		while (*sifra!='\0') {
			if (*p==*sifra++) {
				*p=126-brojac;
				break;
			}
			brojac++;
		}
		p++;
	}
}
