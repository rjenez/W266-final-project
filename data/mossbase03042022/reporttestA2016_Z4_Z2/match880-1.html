/* funkcija za sortiranje nam treba kako bi brojac tablice poredali po velicini*/
 void sortiraj (int *niz) {
 	int temp;
 	int i, j, max=0;
 	for(i=0; i<96; i++) {
 		max=i;
 		for(j=i+1; j<96; j++) {
 			if (*(niz+j)>*(niz+max)) max=j;
 		}
 		temp=*(niz+i);
 		*(niz+i)=*(niz+max);
 		*(niz+max)=temp;
 	}
 }
