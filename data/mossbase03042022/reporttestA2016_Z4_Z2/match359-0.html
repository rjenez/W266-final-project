void unesi(char niz[], int dim){
	int i = 0;
	char znak = getchar();
	if(znak == '\n')
		znak = getchar();
	while(i<dim-1 && znak != '\n'){
		niz[i] = znak;
		i++;
		znak = getchar();
	}
	niz[i] = '\0';
}
