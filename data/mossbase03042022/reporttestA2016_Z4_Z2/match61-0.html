#include <stdio.h>



char* dekodiraj(char* s, char* tablica)
{
    char *p=s;
    int i, pomjer;
    while(*p!='\0')
    {
        for(i=0;i<127;i++)
        {
            // Trazimo karakter u tablici jednak nasem karakteru
            if(*p==126-i) pomjer=i;
        }
        *p=(*(tablica+pomjer));
        p++;
    }
    return s;
}
