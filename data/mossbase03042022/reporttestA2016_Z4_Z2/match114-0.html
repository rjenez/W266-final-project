int vrati_najveci(int *niz,int vel){
	int i,j = 0;

	int max = niz[0];

	/*promijenili smo mu vrijednost,da sl put kad pozevemo da vrati sl. max */
	niz[j] = vrijednost;
	vrijednost--;
	/*treba nam pozicija navjeceg */
	return j;
}
