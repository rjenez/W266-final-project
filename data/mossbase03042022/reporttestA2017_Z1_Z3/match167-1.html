#include <stdio.h>
#include <math.h>
#define EPSILON 0.0001

    printf("Unesite koeficijente a, b i c: ");
    scanf("%lf %lf %lf", &a,&b,&c);
    D=b*b-(4*a*c);
    if(D<0){

            return 0;
           }

    }

              pomocna=R1;
              R1=R2;
              R2=pomocna;
              }
