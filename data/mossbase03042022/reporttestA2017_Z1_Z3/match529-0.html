#define epsilon1 0.001
#define epsilon2 0.1

		{
			printf(" Priblizno manje\n");
		}
		else if(((x1-unesi_x1) < 0)  && ((x1 - unesi_x1)> -epsilon2))
		{
			printf(" Priblizno vece\n");
		}
