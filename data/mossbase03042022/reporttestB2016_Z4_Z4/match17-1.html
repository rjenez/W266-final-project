            return ProdatPrije(*a,*b);
        });
    }
    std::for_each(vek_prodaje.begin(),vek_prodaje.end(),[&brojac,i]
    (std::shared_ptr<Prodaja> a){
        if(i==0)std::cout<<brojac<<"."<<std::endl;
        a->Ispisi();
        brojac++;
    });
}
