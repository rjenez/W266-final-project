Vektor SumaCifara(Vektor v) {
    Vektor v1;
    for(int i=0; i<v.size(); i++){
        int suma(0);
        while(v[i]!=0){
            int c;
            c=v[i]%10;
            suma+=c;
            v[i]/=10;
        }

    return v1;
}
