#define d 10

		} while (n>10 || n<=0);
	for(c=0;c<n;c++){
		printf("Unesite %d. tacku: ", c+1);
		scanf("%d %d",&niza[c],&nizb[c]);


		if(niza[c]<0 || niza[c]>19 || nizb[c]<0 || nizb[c]>19){
		printf("Pogresan unos\n");

		}
	}

			}

		}

	}

}
