		printf("Unesite %d. tacku: ",i+1);

	{
		for (j=b;j<a-1;j++){


			tacx[j] = tacx[j+1];
			tacy[j] = tacy[j+1];
		}
		b--;
		a--;
	}
}

		a--;

}
