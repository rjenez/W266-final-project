		}

			mat[i][j]=0;
		}
	}
	for (i=0; i<broj; i++){
		printf ("Unesite %d. tacku: ",i+1);
		scanf ("%d %d", &kolone, &redovi);

			do {
				printf ("Pogresan unos\n");

		}


		while (kolone<0 || redovi<0 || kolone>=20 || redovi >=20);
	}
	mat[redovi][kolone]=1;
	}

		for (j=0; j<20; j++) {
			if (mat[i][j]==1) printf ("*");
			else printf (" ");
		}
		printf ("\n");
	}

}
