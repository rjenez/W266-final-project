#define vel 20

	for(i=0;i<n;i++){
		printf("Unesite %d. tacku: ",i+1);
		scanf("%d",&b[i]);
		scanf("%d",&c[i]);

		{printf("Pogresan unos\n");

		continue;}}

		if(c[i]>max2) max2=c[i];
	}

		}}}

			else printf(" ");

	}

}
