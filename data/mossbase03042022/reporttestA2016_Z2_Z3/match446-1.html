int main() {

	for(i=0;i<broj;i++) {
		do {
		printf("Unesite %d. tacku: ", i+1);
		scanf("%d %d", &nizj[i], &nizi[i]);
		if(nizi[i]<0 || nizi[i]>19 || nizj[i]<0 || nizj[i]>19) 	printf("Pogresan unos\n");
		} while( nizi[i]<0 || nizi[i]>19 || nizj[i]<0 || nizj[i]>19);

		for(j=0;j<20;j++) {

					printf("*");
