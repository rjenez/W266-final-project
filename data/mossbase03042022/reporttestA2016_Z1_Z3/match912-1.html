   		{

		siva++;

		crvena++;

		plava++;
		
		
		}
		else {printf("Neispravan unos\n");}
	}while(boja != 'k' && boja !='K');

	max=crna;

		}
	else if(max==plava) {printf("Najpopularnija boja je plava (%.2f%%).",procent);}

	else if(max==crna) {printf("Najpopularnija boja je crna (%.2f%%).",procent);}

}
