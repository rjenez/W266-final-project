	float bv, c, b, s, v, p, cp, bp, sp, vp, pp;
	bv=0;
	c=0;
	b=0;
	s=0;
	v=0;
	p=0;
	printf("Unesite vozila: ");

		switch(br){
			case 'c':
			case 'C':
				bv+=1;

				break;
			case 'b':
			case 'B':
				bv+=1;
				b+=1;
				break;

			case 'p':

				p+=1;
				break;
			case 'k':
			case 'K':
				break;
			default:

		}
	}
	while(br!='K' && br!='k');
	if(bv!=0){
		cp=c/bv*100;

	if(p>c && p>b && p>s && p>v)
		printf("Najpopularnija boja je plava (%.2f%%).", pp);
