int br_elem(Cvor* pocetak)
{
    int result(0);
    if(!pocetak)return result;
    Cvor* klizac(pocetak->sljedeci);
    for(;;) {
        result++;
        if(klizac==pocetak)break;
        if(klizac==nullptr)return -1;
        klizac=klizac->sljedeci;
    if(result>52)return -1;
    }
    return result;
}
