int Znakovi(std::string ime) {
    int BrojZnakova(0);
    for(int i = 0; i < ime.size(); i++) {
        if((toupper(ime[i]) >= 'A' && toupper(ime[i]) <= 'Z') || (ime[i] >= '0' && ime[i] <= '9')) BrojZnakova++;
    }
    return BrojZnakova;
}
