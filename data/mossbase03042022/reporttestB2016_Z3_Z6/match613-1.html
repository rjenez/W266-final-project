bool DaLiJeSlovoIliBroj (char znak)
{
    if((znak>='a' && znak<='z') || (znak>='A' && znak<='Z') || (znak>='0' && znak<='9'))
        return true;
    return false;
}

int DuzinaImena (std::string ime)
{
    int duzina(0);
    auto it(ime.begin());
    while (it!=ime.end())
    {
        if(DaLiJeSlovoIliBroj(*it))
            duzina++;
        it++;
    }
    return duzina;
}
