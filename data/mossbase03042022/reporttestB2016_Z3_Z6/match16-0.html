int brkoraka(std::string s){
    int br=0;
    for(int i=0; i<s.size();i++){
        while(i<s.size() && ((s[i]>='a' && s[i]<='z') || (s[i]>='A' && s[i]<='Z') || (s[i]>='0' && s[i]<='9'))){
            br++;
            i++;
        }
    }
    return br;
}
