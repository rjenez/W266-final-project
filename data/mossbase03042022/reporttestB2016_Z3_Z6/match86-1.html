int BrSlova(std::string s)
{
    int br = 0;
    for(char i : s)
    {
        if((i >= 'A' && i <= 'Z') || (i >= 'a' && i <= 'z') || (i >= '0' && i <= '9'))
            br++;
    }
    return br;
}
