int mod(int n, int k){
    int a(n/k);
    return n-a*k;
}

int DuzinaImena(std::string ime){
    int n(0);
    for(int i=0; i<ime.size(); i++)
        if((ime[i]>='A' && ime[i]<='Z') || (ime[i]>='a' && ime[i]<='z') || (ime[i]>='0' && ime[i]<='9')) n++;

    return n;
}
