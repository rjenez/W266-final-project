bool SlovoCifra(char c) {
    if((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') || (c >= '0' && c <= '9'))
        return true;
    return false;    
}

int DajBrojSlova(string s) {
    int ukupno(0);
    
    for(auto it = s.begin(); it != s.end(); it++) {
        if(SlovoCifra(*it))
            ukupno++;
    }
    
    return ukupno;
}

    bool ulaz(true);
