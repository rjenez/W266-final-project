std::vector<std::set<std::string>> Razvrstavanje(std::vector<std::string> vs, int n) {
    if(n<1 or n>vs.size()) throw std::logic_error("Razvrstavanje nemoguce");
    int temp1(vs.size()%n), temp2(vs.size()/n);
    int x((vs.size()-temp1*(temp2+1))/temp2);
    std::vector<std::set<std::string>> vss(temp1+x);
    /*std::list<std::string> ls;
    for(int i(0); i<vs.size(); i++) ls.push_back(vs[i]);
    std::list<std::string>::iterator it(ls.begin());*/
    std::shared_ptr<Dijete> pocetak(nullptr), prethodni;
    for(int i(0); i<vs.size(); i++) {
        std::shared_ptr<Dijete> novi;
        try {
            novi = std::make_shared<Dijete>();
        }
        catch(std::bad_alloc) {
            throw;
        }
        novi->ime = vs[i]; novi->sljedeci =nullptr;
        if(!pocetak) pocetak=novi;
            else prethodni->sljedeci=novi;
        prethodni=novi;
    }
    prethodni->sljedeci = pocetak;
    auto it(pocetak);
    int i(0), broj_djece(0);
    while(1) {
        //if(it==ls.end()) it = ls.begin();
        int broj_pomjeranja(0);
        if(broj_djece<temp2+1 and i<temp1) {
            vss[i].insert(it->ime);
            broj_djece++;
        } else if(broj_djece==temp2+1 and i<temp1){
            broj_djece=0;
            i++;
            vss[i].insert(it->ime);
            broj_djece++;
        } else if(i>=temp1) {
            if(broj_djece<temp2) {
                vss[i].insert(it->ime);
                broj_djece++;
            } else if(broj_djece==temp2) {
                broj_djece=0;
                i++;
                vss[i].insert(it->ime);
                broj_djece++;
            }
        }
        for(int j(0); j<(it->ime).length(); j++)
            if(!((it->ime)[j]<'0' or ((it->ime)[j]>'9' and (it->ime)[j]<'A') or ((it->ime)[j]>'Z' and (it->ime)[j]<'a') or (it->ime)[j]>'z'))
                broj_pomjeranja++;
        broj_pomjeranja-=1;
        //it=ls.erase(it);
        prethodni->sljedeci = it->sljedeci;
        it=it->sljedeci;
        int rasporedjeni(0);
        for(auto x : vss) rasporedjeni+=x.size();
        if(rasporedjeni==vs.size()) {
            prethodni->sljedeci=nullptr;
            break;
        }
        while(broj_pomjeranja) {
            //if(it==ls.end()) it = ls.begin();
            it=it->sljedeci;
            prethodni = prethodni->sljedeci;
            broj_pomjeranja--;
        }
    }
    return vss;
}
