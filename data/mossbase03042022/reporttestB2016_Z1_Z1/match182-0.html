vector <int> IzbaciVisak(vector<int> brojevi)
{
	for(unsigned int i=0;i<brojevi.size();i++)
	{
		for(unsigned int j=i+1;j<brojevi.size();j++)
		{
			if(brojevi[i]==brojevi[j])
			{
				brojevi.erase(brojevi.begin()+j);
				j--;
			}
		}
	}
	return brojevi;
}

}
