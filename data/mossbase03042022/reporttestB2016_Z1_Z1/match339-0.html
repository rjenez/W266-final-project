long int Pretvori(int n)
{
	long int pom(0);
	int expon(0);
	int n1;
	while( n!=0)
	{
		n1=n%3;
		pom+=(n1*pow(10,expon));
		expon++;
		n/=3;
	}
	return pom;
}
vector<int> IzdvojiGadne(vector<int> v, bool a)

			}

	}
	return w;
	
}
