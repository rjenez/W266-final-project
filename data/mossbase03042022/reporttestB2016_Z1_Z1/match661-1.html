std::vector<int> Ternarni (int n) 
{
	std::vector<int> br(3) ;
	
	while(n!=0)
	{
		int b=abs(n%3);
		if(b==0) br[0]++;
		if(b==1) br[1]++;
		if(b==2) br[2]++;
		n=n/3;
		
	}
	
	return br;
	
	
}

{

std::vector<int> IzdvojiGadne ( std::vector<int > v, bool brojevi )
{
	int n=v.size();
	if(n==0)
	{
		return v;
	}
	for(int i=0; i<n; i++)
	{
		for(int j=i+1; j<n; j++)
		{
			if(v[i]==v[j])
			{
				for( int k=j; k<n-1; k++)
				{
					v[k]=v[k+1];
				}
				n--; 
				j--;
			}
		}
	}
	
	v.resize(n);
	
	std::vector<int> vector ;
	for(int i=0; i<=v.size()-1; i++)
	{
		
		if(Gadan(v[i])==brojevi) vector.push_back(v[i]);
		if(i==v.size()-1) break;
	}
	
	return vector;
	
}
