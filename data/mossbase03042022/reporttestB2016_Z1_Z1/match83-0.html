std::vector<int> Procisti (std::vector<int> v){
	for(int i=0; i<(int)v.size(); i++)
		for(int j=i+1; j<(int)v.size(); j++)
			if(v[i]==v[j]){
				int p=j;
				for(int k=j+1; k<(int)v.size(); k++){
					v[p]=v[k];
					p++;
				}
				v.resize(v.size()-1);
				j--;
			}
	return v;
}
