#include <iostream>

std::vector<int> Pretvori (int broj) //TERNARNI ZAPIS
{
	std::vector<int> v2(0);
	int cifra{0};
	
	do {
		cifra = abs(broj%3);
		broj /= 3;
		v2.push_back(cifra);
	} while (broj != 0);

	return v2;
}

std::vector<int> IzdvojiGadne (std::vector<int> v, bool x)
{
	for (int i=0; i<v.size(); i++)

			if (v[i] == v[j])
			{
				v.erase(v.begin()+j);
				j--;
			}
		}

		for (int j=0; j<p.size(); j++)
		{
			if (p[j] == 0)	b0 ++;

			if (p[j] == 2)	b2 ++;
		}
	
		if (b0%2==0 && b1%2==0 && b2%2==0)
		{

		}
	}
