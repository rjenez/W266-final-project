std::vector<int> ternarni(int x){
	std::vector<int> tern;
	do {
		int m;
		m=x%3;
		tern.push_back(m);
		x=x/3;
} while(x!=0);
return tern;
}

     		if(p[j]==p[i]) brojac++;

    } return true;
}
