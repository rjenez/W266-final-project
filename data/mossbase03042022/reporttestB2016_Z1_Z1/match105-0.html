bool DaLiJeOpak(int n)
{
	std::vector<int> broj;
	long int m{abs(n)};
	while(1)
	{
		broj.push_back(m%3);
		m/=3;
		if(m==0)
			break;
	}
	
	for(int i(0);i<3;i++)
	{
		int brojac(0);
		for(int j(0);j<broj.size();j++)
		{
			if(i==broj.at(j))
				brojac++;
		}
		
		if(brojac%2!=0 && brojac!=0)
		{
			return false;
		}
	}
	
	return true;
}
