std::vector<int> PretvoriBrojeve (long long int n) {
	std::vector<int> v;
	if (n<0) n=(abs(n));
	
	while (n != 0) {
		v.push_back(n%3);
		n/=3;
	}
	return v;
}



std::vector<int> Ponavljanje(std::vector<int> v) {
	for(int i=0; i<v.size(); i++) {
		for(int j=i+1; j<v.size(); j++) {
			if(v[i]==v[j]) {
				v.erase(v.begin()+j);
				j--;
			}
		}
	}
	return v;
}
