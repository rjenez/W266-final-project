#include <iostream>
#include <vector>
#include <cmath>

int Obrni(int n) {
	int novi(0);
	while(n!=0) {
		novi = novi*10 + n%10;
		n/=10;
	}
	return novi;
}

int Ternarni(int n) {
	int rez(0);
	while(n!=0) {
		rez = rez*10 + n%3;
		n/=3;
	}
	return Obrni(rez);
}

bool Testiraj(int n) {
	int cifra, brojaci[3]={0};
	while(n!=0) {
		cifra=n%10;
		brojaci[cifra]++;	//histogram - brojimo koliko se puta javlja svaka cifra u ternarnom obliku broja
		n/=10;
	}
	bool paran(false);
	for(int i(0); i<3; i++) {
		if(brojaci[i]%2==0 && brojaci[i]!=0) paran=true;
		else paran=false;
	}
	return paran;
}

bool Ponavljanje (int n, std::vector<int> v) {
	int br(0);
	for(int i(0); i<v.size(); i++) {
		if(n==v[i]) br++;
	}
	if(br>=1) return true;
	else return false;
}
