std::vector<int> IzbaciDuplikate(std::vector<int> a){
	for(int i=0;i<a.size();i++){
		for(int j=i+1;j<a.size();j++){
		if(a[i]==a[j]){
			a.erase(a.begin()+j);
			j--;
	}
}
}
return a;
}

std::vector<int> Ternary(int n){
	std::vector<int> v;
	int ostatak;
	while(n!=0){
		ostatak=n%3;
		v.push_back(ostatak);
		n=n/3;
	}
	return v;
}
