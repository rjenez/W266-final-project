vector<int> dajt(int h)
{
	if(h<0) h=-h;
	vector<int> pom;
	while(h!=0)
	{
		pom.push_back(h%3);
		h/=3;
	}
	return pom;
}
int tester(vector<int> k)
{

	{
		if(k[a]==0) nule++;
		if(k[a]==1) kecevi++;
		if(k[a]==2) dvice++;
		a++;
	}
	if(nule%2==0 && kecevi%2==0 && dvice%2==0) {
		nule=1;
		kecevi=0;
		dvice=0;
	}
	else if((nule%2!=0 || nule==0)&&(kecevi%2!=0 || kecevi==0)&&(dvice%2!=0 || dvice==0))
	{
		nule=1;
		kecevi=1;
		dvice=0;
	}
	else 
	{
		nule=1;
		kecevi=1;
		dvice=1;
	}
	return nule+kecevi+dvice;
}
