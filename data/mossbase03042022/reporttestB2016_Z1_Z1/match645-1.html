vector<int> ternarni(int n)
{
	vector<int>v;
	int temp=n;
	if(temp<0) temp=fabs(temp);
	while(temp!=0) {
		int cifra=temp%3;
		v.push_back(cifra);
		temp/=3;
	}
	return v;
}
