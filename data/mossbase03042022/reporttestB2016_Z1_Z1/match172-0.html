bool Opak(int n)
{
	vector<int> cifre (10, 0);
	while(n)
	{
		cifre[abs(n%3)]++; //pomjerio zagradu
		n/=3;
	}
	for(int i=0; i<=9; i++)
	{
		if(cifre[i]%2!=0)
			return false;
	}
	return true;
}

bool Odvratan(int n)
{
	vector<int> cifre (10, 0);
	while(n)
	{
		cifre[abs(n%3)]++; //isto
		n/=3;
	}
	for(int i=0; i<=9; i++)
	{
		if(cifre[i]%2==0 && cifre[i]!=0)
			return false;
	}
	return true;
}

vector<int> IzdvojiGadne(vector<int> v, bool opaki_odvratni)
{
	vector<int> gadni;
	if(opaki_odvratni)
	{
		for(int i=0; i<v.size(); i++)
		{
			int isti(0);
			for(int j=0; j<i; j++)
			{
				if(v[j]==v[i])
				{
					isti++;
					break;
				}
			}
			if(isti==0)
				if(Opak(v[i]))
					gadni.push_back(v[i]);
		}
	}

		for(int i=0; i<v.size(); i++)

			}

		}
		
	return gadni;
}
