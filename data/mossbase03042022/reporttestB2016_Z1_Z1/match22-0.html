vector<int> IzbaciIste(vector<int> v){   //funkcija za izbacivanje istih brojeva iz vektora
	for(int i=0; i<v.size()-1; i++){
		for(int j=i+1; j<v.size(); j++)
			if(v[i]==v[j]){


				for(int k=j; k<v.size()-1; k++)
				v[k]=v[k+1];
				j--;
				v.resize(v.size()-1);
		}
	}
	return v;
}
