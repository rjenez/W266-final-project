vector<int> IzbaciIste(vector<int> v){


	for(int i=0;i<v.size();i++){
		for(int j=i+1;j<v.size();j++){
			if(v[j]==v[i]){
				for(int h=j;h<v.size()-1;h++){
					v[h]=v[h+1];
				}
				v.resize(v.size()-1);
				j--;
			}
		}
	}
	return v;
}

bool Opak(int n){
	int i(0),v[3]={0};
	if(n<0)n=-n;

		if(v[i]>0 && v[i]%2!=0)
		return false;
		}

}
