std::vector<int> brojaccifara (std::vector<int> v) {
	std::vector<int> brojaci(v.size(),1) ;
	for(int i=0;i<v.size();i++) {
		for(int j=0;j<v.size();j++) {
			if(i==j) continue;
			else if (v[i]==v[j]) brojaci[i]++;

	return brojaci;
}
