bool DaLiJeOpak (int broj)
{

	std::vector<int>brojaci(3,0);
	int opak(0);
	while(br!=0)
	{
		cifre.push_back(br%3);
		br/=3;
		
	}
	
	


	for(int i=0; i<cifre.size(); i++)
	{
		brojaci[cifre[i]]++;
	}

	
    for(int j=0; j<3; j++)
    {

    }

	else return false;
	
}
