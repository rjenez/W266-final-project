std::vector<int> IzbaciDuple(std::vector<int> v)
{
	for(int i=0; i<v.size(); i++)
	{
		for(int j=i+1; j<v.size(); j++)
		{
			if(v[i]==v[j])
			{
				for(int k=j; k<v.size()-1; k++)
				{
					v[k]=v[k+1];
				}
				v.resize(v.size()-1);
				j--;
			}
		}
	}
	return v;
}
