vector<int> IzbaciDuple(vector<int> v)
{
	if(v.size() == 0) return v;
	for(int i = 0 ; i < v.size()-1 ; i++)
	{
		for(int j = i+1 ; j < v.size() ; j++)
		{
			if(v[j]==v[i])
			{
				v.erase(v.begin()+j);
				j--;
			}
		}
	}
	return v;
}
