#include <iostream>

std::vector<int> Ternarni(int n )  /*Pretvorimo unesenu cifru iz dekadnog u ternarni*/
{
    std::vector<int> rez;

    while(n != 0) {
        rez.push_back(n%3);
        n/=3;
    }
    return rez;
}
