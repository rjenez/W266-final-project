 std::vector<long int> baza_tri (int n) {
	std::vector<long int> vektor;
	long int pomocni (1);
	do {
		long double ostatak = fabs (n % 3);
		n /= 3;
		vektor.push_back(ostatak);

	} while (n != 0);
	
	return vektor;
}
