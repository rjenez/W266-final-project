std::vector<int> VratiListuCifri (int N){
	std::vector<int> vrste_brojeva{0, 0, 0};
	int broj = abs(N);
	do{
		int cifra = broj % 3;
		vrste_brojeva[cifra] += 1;
		broj = broj / 3;
	} while(broj!=0);
	return vrste_brojeva;
}
