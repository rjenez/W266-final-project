bool ProvjeraOp (int broj) {

	std::vector<long long int> brojac(3, 0);
	do{

		if(ostatak<0) ostatak*=-1;
		broj/=3;
		brojac[ostatak]++;
	}while(broj!=0); 
	for(int i(0); i<brojac.size(); i++) {
		if(brojac[i]==0) continue;
		else if(brojac[i]%2==0) continue;

	}
	return true;
}
