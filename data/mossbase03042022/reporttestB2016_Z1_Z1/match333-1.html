std::vector<int> IzbaciDuple(std::vector<int> br)
{
	for(int i=0; i < br.size(); i++){
		for(int j=i+1; j < br.size(); j++)
		{
			if (br[i] == br[j]) {
				br.erase(br.begin()+j);
				j--;}
		}}
		
		
		return br;
}
