int maxstepen(int broj){
	int brojac(0);
	while(broj!=0){
		broj=broj/3;
		if(broj==0)
			return brojac;
		brojac++;
	}
	return brojac;
}
std::vector<int> IzbaciIste(std::vector<int> v){


	for(int i(0);i<v.size()-1;i++){
		for(int j(i+1);j<v.size();j++){
		if(v[i]==v[j]){
			v.erase(v.begin()+j);
			j--;
		}
		}
	}
	return v;
}
