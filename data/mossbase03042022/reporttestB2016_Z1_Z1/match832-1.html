//Funkcija za konverziju broja iz dekadnog u ternarni brojni sistem
std::vector<int> PretvoriUTernarni (int n) {
    std::vector<int>cifre;
    do{
        cifre.push_back(std::abs(n%3));
        n/=3;
    }while(n!=0);
    return cifre;
}
