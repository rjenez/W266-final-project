gadni provjera(int k){
	
	std::vector<int> p(3,0);

	 	int a;

	}


	gadni m;
	
	if(p[0]%2==0 && p[1]%2==0 && p[2]%2==0){
		m=opaki;
	}else
	{
		for(int i=0;i<p.size();i++){
			if(p[i]==0) p[i]++;

			}
	
		}
		
		if(p[0]%2!=0 && p[1]%2!=0 && p[2]%2!=0){
			m=odvratni;
		}
	}

}

		}

	}
}

}
