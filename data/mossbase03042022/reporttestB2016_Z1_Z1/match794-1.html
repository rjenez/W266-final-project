#include <iostream>

bool ProvjeriOpaki(int n) {
	bool gadni(false);
	int broj_jedinice(0), broj_dvice(0), broj_nule(0);
	
	while(n) {
		if(n%3==0) broj_nule++;
		else if(n%3==1) broj_jedinice++;
		else broj_dvice++;
		n/=3;
	}
	if((broj_nule%2==0 ||broj_nule==0) && (broj_jedinice%2==0 || broj_jedinice==0) && (broj_dvice%2==0 || broj_dvice==0)) gadni=true;
	return gadni;
}
