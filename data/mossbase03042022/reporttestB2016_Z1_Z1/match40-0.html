std::vector< long long int> UkloniDuple(std::vector<long long int>niz){
    long long int size=niz.size(),i,j,k;
    for(i=0;i<size;i++){
        for(j=i+1;j<size;){
            if(niz[j]==niz[i]){
                for(k=j;k<size-1;k++){
                    niz[k]=niz[k+1];
                }
                size--;
            }else
            j++;
        }
    }
    std::vector< long long int>Vraceni;
   for(i=0;i<size;i++){
      Vraceni.push_back(niz[i]);
}
return Vraceni;
}
