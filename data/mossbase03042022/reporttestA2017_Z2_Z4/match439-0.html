/* Funkcija za sortiranje niza putem selection sort-a */


void sortiraj(int p,int q,int niz[])
{
	int i,j,min,temp;
	for (i=0; i<p*q; i++) {
		min=i;
		for (j=i+1; j<p*q; j++) {
			if (niz[j]>niz[min])


				min=j;
		}
		temp=niz[i];
		niz[i]=niz[min];
		niz[min]=temp;
	}
}
