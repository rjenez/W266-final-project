int main()
{

				maxa=A[i][j];
		}
	}


	printf ("Unesite clanove matrice B: ");
	for (i=0; i<M; i++) {
		for (j=0; j<N; j++) {
			scanf ("%d", &B[i][j]);
		}
	}
	maxb=B[0][0];
	for (i=0; i<M; i++) {
		for (j=0; j<N; j++) {
			if (B[i][j]>maxb)
				maxb=B[i][j];
		}
	}

		
		
		printf ("Unesite clanove matrice C: ");
		for (i=0; i<M; i++) {
			for (j=0; j<N; j++) {
				scanf ("%d", &C[i][j]);
			}
		}
		maxc=C[0][0];
		for (i=0; i<M; i++) {
			for (j=0; j<N; j++) {

					maxc=C[i][j];

		}

			
			for (i=0; i<M; i++) {
				for (j=0; j<(N); j++) {

						if (A[i][j]==a1) {
							bra[a1]++;
						}
					}
				}
			}

			for (i=0; i<M; i++) {
