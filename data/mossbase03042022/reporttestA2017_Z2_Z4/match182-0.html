#include <stdio.h>
#define STOTINA 100



int koliko(int broj, int M, int N, int mat[][STOTINA]){
	int i, j;
	int brojac=0;
	for(i=0; i<M; i++){
		for(j=0; j<N; j++){
			if(mat[i][j]==broj) brojac++;
		}
	}
	return brojac;
}
