#include <stdio.h>
#define M 100
#define N 100

		printf("Unesite brojeve M i N: ");
		scanf("%d %d", &m, &n);

			scanf("%d", &C[i][j]);
		}
	}

	k=0;

		for(j=0; j<n; j++){

		}
