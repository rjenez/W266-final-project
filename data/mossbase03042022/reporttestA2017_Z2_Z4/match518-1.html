#include <stdio.h>
#include <stdlib.h>
#define MAX 100
#define MAXI 10000

			scanf("%d", &A[i][j]);

		}

		for(j=0; j<n; j++){
			scanf("%d", &C[i][j]);
		}
	}
	
	for(i=0; i<m; i++){


		for(j=0; j<n; j++){
			Ai[b]=A[i][j];
			Bi[b]=B[i][j];
			Ci[b]=C[i][j];
			b++;
		}
	}
	
	for(i=0; i<b; i++){

			if(Ci[j]<Ci[minc]) minc=j;
		}
		mc=Ci[i];

		Ci[minc]=mc;
