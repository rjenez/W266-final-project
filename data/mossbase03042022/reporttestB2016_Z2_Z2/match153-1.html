template<typename t>
t Rotiraj(t mat)
{ 
    t pom(mat[0].size());


    for(int i=0;i<pom.size();i++)
        pom[i].resize(mat.size());
        
    for(int i=0;i<pom.size();i++)
    {
        for(int j=0;j<pom[i].size();j++)
        {
            pom[i][j]=mat[j][i];
        }
    }
    return pom;
}
