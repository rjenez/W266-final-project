template<typename TipMat>
TipMat Okreni(TipMat m){


    TipMat Nova(m[0].size());
    for(int i=0;i<Nova.size();i++){
        Nova[i].resize(m.size());
    }
    for(int i=0;i<Nova.size();i++){
        for(int j=0;j<Nova[i].size();j++){
            Nova[i][j]=m[j][i];
        }
    }
    return Nova;
}
