    NekiTip Novi(Niz.at(0).size(), typename std::remove_reference<decltype(Niz.at(0))>::type(Niz.size()));
    


    for(int i = 0; i < Niz.size(); i++){
        for(int j = 0; j < Niz.at(0).size(); j++){
            Novi.at(j).at(i) = Niz.at(i).at(j);
        }
    }
    return Novi;
}
