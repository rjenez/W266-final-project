#include <stdio.h>
#define PI 3.1415926
#include <stdlib.h>
#include <math.h>
int main(){
	double ang,rem,rez,minu,secu;
	int ugao[500]={0},min[500]={0},sec[500]={0},i,j;
	int bruglova;
	printf("Unesite broj uglova: ");
	scanf("%d",&bruglova);
	
	for (i=0;i<bruglova;i++){
		scanf("%lf",&ang);
		rez=(ang*180.)/PI;
		//hrs
		ugao[i]=rez;
		//min
		rem=fabs(rez-ugao[i]);
		minu=(rem*60);
		min[i]=minu;
		//sec
		rem=fabs(minu-min[i]);
		secu=(rem*60)+0.5;
		sec[i]=secu;
		if (sec[i]==60){
			min[i]++;
			sec[i]=0;
		}
		if (min[i]==60){
			if (ugao[i]>0){
			ugao[i]++;}
			else {


				ugao[i]--;
			}
			min[i]=0;
		}
	}
	//izbacivanje elemenata gdje je sec[i]>30 ili 
	
	for (i=0; i<bruglova; i++) {
		if (sec[i]>30) {
			for (j=i; j<bruglova-1; j++) {
			ugao[j] = ugao[j+1];
			min[j] = min[j+1];


			sec[j] = sec[j+1];
		}
	bruglova--;
	i--;
	}
}
	printf("Uglovi su: \n");
	for (i=0;i<bruglova;i++){
		printf("%d stepeni %d minuta %d sekundi\n",ugao[i],min[i],sec[i]);
	}
	return 0;
}
