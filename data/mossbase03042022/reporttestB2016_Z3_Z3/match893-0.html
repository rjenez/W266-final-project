template<typename tip>

struct Matrica{
    int br_redova, br_kolona;
    tip **pok=nullptr;
};

template<typename tip>
void UnistiMatricu(Matrica<tip>mat){
    if(!mat.pok) return;
    for(int i=0;i<mat.br_redova;i++) delete [] mat.pok[i];
    delete [] mat.pok;
    mat.pok=nullptr;
}

template<typename tip>

Matrica<tip>StvoriMatricu (int m, int n=0){
    if(n==0) n=m;
    Matrica <tip> mat;
    mat.br_redova=m;
    mat.br_kolona=n;
    try{
        mat.pok=new tip *[m];
      for(int i=0;i<m;i++) mat.pok[i]=nullptr;


        try{
            for(int i=0;i<m;i++) mat.pok[i]=new tip[n]; //svi elementi na nulu
            for(int i=0;i<m;i++){
                for(int j=0;j<n;j++)
                    mat.pok[i][j]=0;
            }
    }
    catch(...){
        UnistiMatricu(mat);
        throw;
      }
    }
    catch(...){
        delete [] mat.pok;
        throw;
    }
    return mat;
}
