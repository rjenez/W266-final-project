template <typename TipElementa>
    struct Matrica {

template <typename TipElementa>
    void UnistiMatricu(Matrica<TipElementa> mat) {
        if(!mat.elementi) return;
        for(int i = 0; i < mat.br_redova; i++) delete[] mat.elementi[i];
        delete[] mat.elementi;
        mat.elementi = nullptr;
    }
