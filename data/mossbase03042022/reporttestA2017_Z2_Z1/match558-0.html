#include <stdio.h>


int main() 
{

		{
			printf("Neispravno N!\n");
		}

	printf("\n");

	{
	for(j=0;j<n;j++)
	{
	if(j<n/2)
	{

		{
			printf("\\");
		}
		else if(j==(n/2)-i-1)
		{
			printf("/");
		}
		else
		{
			printf(" ");
		}
	}
	else if(i==0 || i==n/2-1)
	{
		if(j==n/2 || j==n-1)
		{
			printf("+");
		}
		else
		{
			printf("-");
		}
	}
	else
	{
		if(j==n/2 || j==n-1)
		{
			printf("|");
		}
		else
		{
			printf(" ");
		}
	}
	}
	printf("\n");
	}
	m=n/2;
	for(i=m;i<n;i++)
	{
	for(j=0;j<n;j++)
	{
	if(j<n/2)	
	{
		if(i==n/2 || i==n-1)
		{
			if(j==0 || j==n/2-1)
			{
				printf("+");
			}
			else
			{
				printf("-");
			}
		}

			if(j==0 || j==n/2-1)

			}
			else
			{
				printf(" ");
			}
		}
	}
	else
	{
			if(i==x1 || i==x2)
			{
				if(j==x1 || j==x2)
				{
					printf("+");
				}
				else
				{
					printf("-");
				}
			}
			else
			{
				if(j==x1 || j==x2)
				{
					printf("|");
				}
				else
				{
					printf(" ");
				}
			}
	}
	}
