			}
			else printf (" ");
		}			
		if(i==0 || i==N/2-1){

	}
	for(i=0;i<N/2;i++){
		if(i==0 || i==N/2-1){
			printf("+");
			for(k=1;k<=N/2-2;k++) {
				printf("-");
			}
			printf("+");
