{

	} while(1);

	for (i=1; i<=N; i++) {
		for (j=1; j<=N; j++) {

				printf("-");
			else if((i>1 && i<N/2 && j==(N/2)+1) || (i>1 && i<N/2 && j==N))
				printf("|");

			//treci dio
			else if((i==(N/2)+1 && j==1) || (i==(N/2)+1 && j==N/2) || (i==N && j==1) || (i==N && j==N/2))

				printf("-");
			else printf(" ");


		}

	}

}
