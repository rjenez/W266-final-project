		printf("\n");

	for (i=0;i<n/2;i++){
			for (j=0;j<n/2;j++){
				if (i==j) printf("\\");

				else printf(" ");
			}
			for (j=n/2;j<n;j++){
				if (j==n/2 || j==n-1){
					if (i==0 || i== n/2-1) printf("+");
					else printf ("|");
				}
				else if (i==0 || i==n/2-1) printf("-");
				else printf(" ");
			}
		printf("\n");
	}
	r=3*n/4;
	for (i=n/2;i<n;i++){
		for (j=0;j<n/2;j++){
			if (j==0 || j==n/2-1) {
				if (i==n/2 || i==n-1) printf("+");
				else printf("|");
			}
			else if(i==n/2 || i==n-1) printf("-");
			else printf(" ");
		}
		for(j=n/2;j<n;j++){
			if (i==r-1 || i==r){

				else printf("-");
			}

			else printf(" ");
		}
		printf("\n");
	}
	return 0;
}
