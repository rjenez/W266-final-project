int izbaci (int* niz, int duzina){
    int i, fib, fib1, fib0, j;

    for (i=0; i<duzina; i++) {
        
        /*provjera da li je broj Fibonaccijev*/

        fib=1;
        fib1=1;
        fib0=1;
        while (fib<niz[i]) { 
            fib=fib1+fib0;
            fib0=fib1;
            fib1=fib;
        }
        /*izbacivanje Fibonaccijevih brojeva*/
        if (fib==niz[i]) {
            for (j=i; j<duzina-1; j++) {
                niz[j]=niz[j+1];
            }
            duzina--;
            i--;
        }

    }

    return duzina;
}
