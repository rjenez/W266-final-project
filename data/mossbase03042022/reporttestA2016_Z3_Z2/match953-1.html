{

	}
}
int fibonacijev(int broj)
{
	

	int broj1=0;
	int broj2=1;
	int fib=0;
	if(broj==0) return 0;
	for(;;)
	{
		if(fib>=broj) break;
		fib=broj1+broj2;
		broj1=broj2;
		broj2=fib;
	}
	if(fib==broj) return 1;
	else return 0;
	
}
