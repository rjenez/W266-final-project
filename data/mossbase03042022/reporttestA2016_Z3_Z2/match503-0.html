int sum(int a){
	int suma=0;
	a=abs(a);
	while(a!=0){
		suma+=a%10;
		a/=10;
	}
	return suma;
}

int fibonacci(int a){
	int i=1,j=1,k=0;
	if (a==1) return 1;
	while(2){
		k=i+j;
		if(k==a) return 1;
		if(k>a) return 0;
		i=j;
		j=k;
	}
	return k;
}
int izbaci(int *niz,int vel){
	int i,j;
	int y;


	for(i=0;i<vel;i++){
		y=fibonacci(niz[i]);

		for(j=i;j<vel-1;j++){
			niz[j]=niz[j+1];
		}
		vel--;
		i--;
		}
	}
	return vel;
}
