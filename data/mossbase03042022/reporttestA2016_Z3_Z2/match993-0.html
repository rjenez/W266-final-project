int sumacifara (int x)
{
	int suma=0;
	do
	{
		x=abs(x);
		suma+=x%10;
		x=x/10;
	}
	while(x!=0);
	return suma;
}
void ubaci (int niz[], int VEL)
{
	int i,j;
	for(i=0;i<VEL;i++)
	{
		for(j=VEL+i;j>2*i+1;j--)
		{


			niz[j]=niz[j-1];
		}
		niz[j]=sumacifara(niz[j-1]);
	}
}
int fibonaci (int x)
{
	if(x==1 || x==2)
	{
		return 1;
	}
	else
	{
		return fibonaci(x-1) + fibonaci(x-2);
	}
}

}
