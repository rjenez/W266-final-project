    int Fibonacijev_niz(int m){
        int a;
        int b;
        int c;
        if(m<0)return 0;
        if(m==1)return 1;
        a=1;
        b=1;


        while(1){
            c=a+b;
            if(m==c)return 1;
            else if(c>m)return 0;
            a=b;
            b=c;
        }
    }
