int suma (int x) {
    int i, s=0, br=0, a, b;
      if (x<0)
        x=-1*x;
    b=x;
    /* Racunanje broja cifara unesenog clana niza */
    while (x>0) {
        x=x/10;
        br++;
    }
    /* Racunanje sume cifara unesenog clana niza */
    for (i=0; i<br; i++) {
        a=b%10;
        b=(b-a)/10;

    }

}
