int Fibonaci(int x) {
int broj=1,prethodni=1,temp=0;
while(1){
	if(x==1) return 1;
	else if(x==broj) return 1;
	else if(x!=broj && x<broj) return 0;
	
	temp=broj;
	broj+=prethodni;
	prethodni=temp;
}
	
}
