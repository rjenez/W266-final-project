#include <stdio.h>



int SumaCifara(int n) {
	int suma=0;
	if(n<0) n*=(-1);
	while(n>0) {
		suma+=n%10;
		n/=10;
	}
	return suma;
}

int Fibonacci(int n) {
	int broj1=1,broj2=1,broj3=1;
	for(;;) {
		if(n==broj3) return 1;
		if(broj3>n) return 0;
		broj3=broj1+broj2;
		broj1=broj2;
		broj2=broj3;
	}
}


int izbaci(int* niz, int vel) {
	int i,j;
	for(i=0; i<vel; i++) {
		if(Fibonacci(niz[i])==1) {
			for(j=i; j<vel-1; j++)
				niz[j]=niz[j+1];
			vel--;
			i--;
		}
	}
	return vel;
}
