#include <stdio.h>
void ubaci(int*niz, int vel)
{
	int niz1[30], i,j=0,suma=0,broj;
	for (i=0; i<vel; i++)
	{
		niz1[j]=niz[i];
		j++;
		broj=niz[i];
		suma=0;
		while (broj!=0)
		{
			suma+=broj%10;
			broj/=10;
		}
		if (suma<0)
		   suma*=-1;
		niz1[j]=suma;
		j++;
		
	}
	for (i=0;i<2*vel; i++)
	{
		niz[i]=niz1[i];
	}
}

  }

     for (j=0; j<300; j++ )

}
