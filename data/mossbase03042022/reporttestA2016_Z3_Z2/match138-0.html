    int fibniz(int n)
    {   /*određivanje fibonačijevog niza ako je suma prethodna dva jednaka n vraća jedan a ako nije vraća nulu*/    
        int x,y,z;


        if(n<0)return 0;
        if(n==1)return 1;
        x=1;
        y=1;
        while(1)
        {
            z=x+y;
            if(n==z)return 1;
            else if(z>n)return 0;
            x=y;
            y=z;
        }
    }
