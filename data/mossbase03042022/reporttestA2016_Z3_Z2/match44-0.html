	}

	//ubacivam elemente

		if (i%2==0) {
		//	if (niz[i]<0) niz[i]*=-1;
			tmp = niz[i];
			if (tmp<0) tmp*=-1;
			sum=0;
			do {
			//	tmp = niz[i*2];

				sum+=tmp%10;

			} while (tmp!=0);

}

int izbaci (int *niz, int v) {

	for (i=0; i<v; i++) {
		for (j=0; j<29; j++ ) {
			if (niz[i]==fib[j]) {
				for (z=i; z<v; z++) {
					niz[z]=niz[z+1];
				}
				v--;
				i--;
				break; 
			}
		}
	}
	return v;
}
