int suma(int n){
	
	int cifra,s=0;
	while(n!=0){
		
		cifra=n%10;
		s=s+cifra;
		n=n/10;
	}
    s=abs(s);
	return s;
}
int kvadrat(long int x){
	long int s=sqrt(x);
	if(s*s==x)return 1;
	else return 0;
	
}
int Fibonaci(long int n){
	if(n<1)return 0;
	else if (kvadrat(5*n*n+4)==1|| kvadrat(5*n*n-4)==1) return 1;
	else return 0;
}

long int izbaci(int niz[], int n){


	long int i,j;

	for(i=0;i<n;i++){
		
		if(Fibonaci(niz[i])==1){

			}

}
