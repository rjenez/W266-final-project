/*Trazenje fibonacijevog niza*/
int fib(int a)
{
	int sljedeci=1,prethodni=0;
	while(sljedeci<=a)
	{
		sljedeci+=prethodni;
		prethodni=sljedeci-prethodni;
		if(a==sljedeci)return 1;
	}
	return 0;
}
/*izbacivanje brojeva fibonacijevog niza iz drugog niza*/
int izbaci(int niz[],int n)
{
	int i,j,v=n;
	for(i=0;i<v-1;i++)
	{
		if(fib(niz[i])){
			for(j=i;j<v-1;j++)
			{
				niz[j]=niz[j+1];
			}
			i--;v--;
		}
	}
	if(fib(niz[i]))v--;
	return v;
}
