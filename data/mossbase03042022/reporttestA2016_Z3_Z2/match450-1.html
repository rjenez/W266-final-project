#include <stdio.h>
int Fibonacci(int n) {
	int broj1=1,broj2=1,broj3;
	if(n==1) return 1;
	for(;;) {
		broj3=broj1+broj2;
		broj1=broj2;
		broj2=broj3;
		if(n==broj3) return 1;
		if(broj3>n) return 0;
	}
}


int zbir(int x) {
	int sabrani=0;
	if(x<0) x*=(-1);
	while(x>0) {
		sabrani+=x%10;
		x/=10;
	}
	return sabrani;
}

	int i,j;
	for(i=0;i<velicina;i++) {
		if(Fibonacci(niz[i])==1) {

}
