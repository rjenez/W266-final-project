int suma (int a)
{
    int s=0;
    while (a>0)
    {
        s=s+a%10;
        a=a/10;
    }
    return s;
}

        b=niz + i*2;

        *b=*s;
    }

    for (i=0; i<velicina; i++)
        {
            if (max<niz[i])

        }
        fibonacijev[0]=1; 
        fibonacijev[1]=1;
        
        
        
    for (i=2; i<1000; i++)
        {
            fibonacijev[i]=fibonacijev[i-1] + fibonacijev[i-2];
            if (fibonacijev[i]>max) break;
            else (b++);
        }
        
    v2=velicina;    
    
    /*izbacivanje Fibonaccijevih brojeva*/

}
