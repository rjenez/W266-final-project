void ubaci(int niz[], int velicina)
{
	int i,sum,j,razdvajanje;


	int tmp=velicina*2-2;
	for(i=velicina-1; i>=1; i--) {
		niz[tmp]=niz[i];
		tmp=tmp-2;
	}
	for(j=1; j<2*velicina; j+=2) {
		sum=0;
		razdvajanje=abs(niz[j-1]);
		while(razdvajanje>0) {
			sum=sum+razdvajanje%10;
			razdvajanje=razdvajanje/10;
		}
		niz[j]=sum;
	}
}

}
