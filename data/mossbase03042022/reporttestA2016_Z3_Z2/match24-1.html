#include <stdio.h>

int SumaCifara (int broj) {
	int cifre [10]={0}, j=0, suma=0, brojac=0;
	while (broj!=0) {
		cifre[j]=broj%10;
		broj=broj/10;
		j++;
		brojac++;
	}
	for (j=0; j<brojac; j++) {
		suma=suma+cifre[j];
	}
	return abs(suma);
}
