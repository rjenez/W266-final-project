int izbaci(int *niz,int vel){


	int fniz[100],i,j,k;
	fniz[0]=1;
	fniz[1]=1;
	for(i=2;i<100;i++){
		fniz[i]=fniz[i-2]+fniz[i-1];
	}
	for(i=0;i<vel;i++){
		for(j=0;j<100;j++){
			if(niz[i]==fniz[j]){
				for(k=i;k<vel-1;k++){
					niz[k]=niz[k+1];
				}
				vel--;				
				i--;
				break;
			}
		}
	}
	return vel;
}
