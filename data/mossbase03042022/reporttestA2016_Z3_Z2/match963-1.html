#include <stdio.h>

int DajSumuCifara (int broj)
{
	if (broj < 0) broj *= (-1);
	int suma = 0;
	while (broj != 0)
	{
		suma += broj%10;
		broj/=10;
	}
	return suma;
}

int DaLiJeFibonaccijevBroj(int broj)
{
	int i, prethodni = 1, trenutni = 1, temp;
	for (i = 0; i < broj; i++)
	{
		if (trenutni == broj) return 1;
		temp = trenutni;
		trenutni += prethodni;
		prethodni = temp;
	}
	return 0;
}

void ubaci(int* niz, int vel)
{
	int i, j;
	for (i = 0; i < vel; i++)
	{
		for (j = vel; j > i; j--)
			niz[j] = niz[j-1];
		vel++;
		niz[i+1] = DajSumuCifara(niz[i]);
		i++;
	}
}

int izbaci (int* niz, int vel)
{


	int i, j;
	for (i = 0; i < vel; i++)
	{
		if (DaLiJeFibonaccijevBroj(niz[i]))
		{
			for(j = i; j < vel-1; j++)
				niz[j] = niz[j+1];
			vel--;
			i--;
		}
	}
	return vel;
}
