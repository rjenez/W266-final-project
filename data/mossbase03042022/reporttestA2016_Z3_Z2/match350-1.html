#include <stdio.h>

int suma_cifara(int broj){

	if(broj < 0)
		broj *= -1;
	while(broj>0) {
		suma+=(broj%10);
		broj/=10;
	}
	return suma;
}


void ubaci(int *niz, int brel) {
	int i, n = brel;
	int j = 0;
	int *p = niz;
	for(i = 0; i < brel*2; i+=2){
        n = brel + j;


		for(; n > i+1; n--)
			p[n] = p[n-1];
		p[i+1] = suma_cifara(p[i]);
		j ++;
	}
}

int fib(int broj){
	int a, b, temp = 0;
	a = b = 1;
	while(b<broj){
		temp = b;
		b += a;
		a = temp;
	}
	return (broj == b);
}

int izbaci(int *niz, int brel) {
	int *p = niz;
	int i=0, j=0;


	for(;i<brel;i++){
		if(fib(p[i])){
			for( j = i; j < brel-1; j ++)
				p[j] = p[j+1];
			brel --;
			i--;
		}
	}
	return brel;	
}
