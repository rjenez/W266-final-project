#include <stdio.h>
int modif(int broj) {
    int suma=0,cifra=0;
    int negativan=0;
    if(broj<0){ negativan=1; broj*=-1; } 
    while(broj!=0) {
        cifra=broj%10;
        suma+=cifra;
        broj/=10;
    }
    if(negativan==1) suma*=1;
    return suma;
}
void ubaci(int* niz, int vel) {
    int i,a=vel;
    vel=vel*2;

    for(i=0;i<a;i++) {
        niz[vel-i*2-2]=niz[vel/2-i-1];
    }
    for(i=0;i<vel;i++) {
        if(i%2!=0) {
            niz[i]=modif(niz[i-1]);
        }
    }
    return;
}
