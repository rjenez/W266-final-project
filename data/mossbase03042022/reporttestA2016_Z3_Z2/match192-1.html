void ubaci( int *niz,int vel)
{
	int i,j;
	int n,s;
	for(i=0;i<vel;i++)
	{
		for(j=vel-1;j>i;j--)
		{
			niz[j+1]=niz[j];
		}
		n=niz[i];
		if(n<0)n*=-1;
		s=0;
		while(n!=0)
		{
			s+=n%10;
			n/=10;
		}
		niz[i+1]=s;
		i++;
		vel++;
	}
}
