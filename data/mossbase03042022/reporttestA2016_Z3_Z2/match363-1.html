int sum(int br)
{
	int suma=0,cifra=0;
	while(br>0)
	{
		
		cifra=br%10;
		suma+=cifra;
		br=br/10;
	}

}
void ubaci(int p[],int n)
{
	int *q,*z;
	int i=2*n;
	int k;
	for(i=n-1;i>=0;i-=1)
	{
		z=p+i*2;
		q=p+i;
		*z=*q;
	}


	for(i=0;i<2*n;i+=2)
	{
		z=p+i;
		q=z+1;
		k=abs(*z);
		*q=sum(k);
	}
}
