#include <stdio.h>

int sumacifara(int a)
{


	int suma=0;
	while (a!=0)
	{
		suma+=a%10;
		a/=10;
	}
	return suma;
}

int DaLiJeFibonacijev(int broj)
{
	int prvi=0,drugi=1, sljedeci,i;
	if(broj==1 || broj ==2 || broj==3||broj==5) return 1;
	for ( i = 0 ; i < broj ; i++ )
   {
      if ( i <= 1 )
         sljedeci = i;
      else
      {
         sljedeci = prvi + drugi;
         prvi = drugi;
         drugi = sljedeci;
      }
      if(broj==sljedeci) return 1;
   }
   return 0;
}

}
