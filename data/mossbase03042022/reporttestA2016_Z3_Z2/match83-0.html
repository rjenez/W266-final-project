int fibonacci (int x){
	int fib[100], i;
	fib[0]=1;
	fib[1]=1;
	for(i=2; i<100; i++){
		fib[i]=fib[i-1]+fib[i-2];
	}
	for(i=0; i<100; i++){
		if(x==fib[i]) return 1;
	}
	return 0;
	}

int izbaci (int*niz,int br){
	int*p=niz, i, j, *q;
	for(i=0; i<br;i++){
		q=p;
		if(fibonacci(*p)==1){
			for(j=i; j<br-1; j++){
				*q=*(q+1);
				*q++;
			}
		*p--;
		br--;
		i--;
		}
	*p++;
	}
	return br;
}
