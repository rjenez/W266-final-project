int fib(int broj) {
	int a=1, b=1, x=2;
	if(broj==1) return 1;
	if(broj < 1) return 0;
	while(x<broj) {
		x=a+b;
		a=b;
		b=x;
	}
	
	if(x==broj) return 1;
	else return 0;
	
}
