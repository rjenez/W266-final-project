#include <stdio.h>

int suma_cifara(int n)
{
    int m=n;
    if(m<0) m*=-1;
    int suma=0;
    if(m<=9 && m>=0)
    {
        return m;
    }
    else if(m>10){
        while(m!=0)
        {
            suma+=m%10;
            m/=10;
        }
    }
    else if(m==10) return 1;
    return suma;
}

int izbaci(int* niz, int vel)
{
    int fibonizi[50];


    int i=0, j, k;
    fibonizi[0]=1;
    fibonizi[1]=1;
    for(i=2;i<50;i++)
    {
        fibonizi[i]=fibonizi[i-1]+fibonizi[i-2];
    }
    for(i=0;i<vel;i++)
    {
        for(j=0;j<50;j++)
        {
            if(niz[i] == fibonizi[j])
            {
                for(k=i;k<vel-1;k++)
                {
                    niz[k]=niz[k+1];
                }
                vel--;
                i--;
                if(i==-1) break;
            }
        }
    }
    return vel;
}
