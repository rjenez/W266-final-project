void ubaci (int niz[], int vel) {
	int suma=0,i,n,j=vel-1;
	for(i=2*vel-2;i>=0;i=i-2) {
		niz[i]=niz[j];
		j--;
	}
	for(i=1;i<2*vel;i=i+2) {
		n=niz[i-1];
		if(n<0) n=n*(-1);
		for(;;) {
			suma+=n%10;
			n=n/10;
			if(n==0) break;
		}
		niz[i]=suma;
		suma=0;
	}
}
int izbaci (int niz[],int vel) {

			a=b;
			b=c;
			if(niz[i]==c || niz[i]==1) {
				ind=1;
				break;
			}
			if(c>niz[i]) break;
		}
		a=1;
