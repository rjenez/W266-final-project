		}

		for(j=i; j>d; j--) {
			temp=p[j-1];
			p[j-1]=p[j];
			p[j]=temp;
		}
		d=d+2;
	}
	b=a;
	for(i=0; i<2*b; i=i+2) {
		k=p[i];
		suma=0;
		do {
			c=fabs(k%10);

		}while(k!=0);

}

int izbaci(int*p, int a){


	int x,y,z=1,u=1,i,j,max;
	max=p[0];
	for(i=0; i<a; i++){
		if(p[i]>max){
			max=p[i];
		}
	}
	do{
		x=0;
		y=z+u;
		for(i=0; i<a; i++){
			if(p[i]==y){
				x=1;
				break;
			}
			else if(p[i]==1){
				x=1;
				break;
			}
		}
		if(x){
			for(j=i; j<a-1; j++){
				p[j]=p[j+1];
			}

		}
		u=z;
		z=y;
	}while(y<=max);

}
