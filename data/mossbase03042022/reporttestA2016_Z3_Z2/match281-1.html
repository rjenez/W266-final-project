void ubaci (int*niz, int vel){
	int *x=niz, i, *m=niz, sumacifara=0, vel1;

		*(x+(2*i))=*(x+i);
	}

	for(i=1;i<vel*2; i+=2){
		broj=*m;

			sumacifara+=(broj)%10;
			broj=broj/10;}

	}
		
}
