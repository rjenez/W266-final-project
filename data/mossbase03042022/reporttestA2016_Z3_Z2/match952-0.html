int sum(int p){

    int suma=0,broj,zadnji_broj;

    }else{

        while(broj!=0){
        
        zadnji_broj = (broj)%10; 
        suma = suma + zadnji_broj;

        }
    }
return suma;
}

int fibonacci(int a) {
    int prvi=0,drugi=1,sljedeci,i;
    for(i=0;i<a;i++) {
       
        sljedeci=prvi+drugi;
        prvi=drugi;
        drugi=sljedeci;
    if(sljedeci==a) {
        return 1;
    }  
    }
    return 0;
 
}
