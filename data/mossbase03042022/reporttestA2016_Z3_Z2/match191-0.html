void ubaci (int niz[], int vel) {
	int i,x,y,suma;


	x=2*vel-2;
	for (i=vel-1; i>=0; i--) {
		niz[x]=niz[i];
		x=x-2;
	}
	for(i=0; i<2*vel; i+=2) {
		suma=0;
		y=fabs(niz[i]);
		do {
			suma=suma+y%10;


			y=y/10;
		} while (y!=0);
		niz[i+1]=suma;
		
	}

}
