void ubaci(int* p,int vel){

	for(i=0;i<vel;i++){
		*(p+2*vel-2*i-2)=*(p+vel-i-1);
	}
	for(i=0;i<vel;i++){
		*(p+2*i+1)=*(p+2*i);
	}
	for(i=0;i<vel;i++){
		*(p+2*i+1)=fabs(*(p+2*i+1));
		do{
			c=(*(p+2*i+1))%10;
			s=s+c;
			*(p+2*i+1)=(*(p+2*i+1))/10;
		}
		while((*(p+2*i+1))!=0);
		*(p+2*i+1)=s;
		s=0;
	}
}
