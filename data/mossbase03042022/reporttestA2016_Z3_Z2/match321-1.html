int izbaci ( int niz[], int vel ){
	int i, fniz[10000] = {0}, j, m, k, maxint;
	
	fniz[0] = 1;
	fniz[1] = 1;
	maxint = 46;                                                    
	


	for( i = 2; i < maxint; i++ ) {                                 /*Fibonacijev niz*/
		fniz[i] = fniz[i-1] + fniz[i-2];                           
	}


		m = vel;
		for( i = 0; i < maxint; i++ ) {                             
			for( j = 0; j < m; j++ ) {                              
			  if ( niz[j] == fniz[i] ) {                            
				for( k = j; k < m-1; k++ ) {                        
				   niz[k] = niz[k+1];                               /*Izbacivanje iz niza*/
				} 
			       m--;                                             
				   j--;
			}
		}
	}
	return m;                                                     
}

}
