int SumaCifara(int x)
{
	int suma=0;
	while(x>0)
	{
		suma+=x%10;
		x/=10;
	}
	return suma;
}
void ubaci (int *niz,int velicina)
{
	int i,j;

	for(i=0;i<velicina;i++)
	 {
	for(j=velicina;j>(i+1);j--)
	  {
	niz[j]=niz[j-1];	
	  }
	i++;
	niz[i]=SumaCifara(abs(niz[i-1]));
	velicina++;
     }
}
