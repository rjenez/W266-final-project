		x=x/10;
		}

}

int Fib(int a) {
	int x=1;
	int y=1, b;
	if(a==1) return 1;
	while(y<=a) {
		b=x;
		x=y;
		y=b+x;
		if(y==a) return 1;
	}
	return 0;
}
