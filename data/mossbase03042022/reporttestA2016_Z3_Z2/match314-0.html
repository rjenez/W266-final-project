int cifre(int i) {
    int suma=0;
    if(i!=0) {
    do {
      i=abs(i);
      suma=suma+(i%10);
      i=i/10;
    }while(i!=0);
}
return suma;
}

int fibonaci(int a) {
    int temp;
    int temp1=1, temp2=1;
    if(a==1) {
        return 1;
    }
    do {
        temp=temp1+temp2;
        if(temp==a) return 1;
        if(temp>a) return 0;
        temp1=temp2;
        temp2=temp;
    }while(1);
}
