#include <stdio.h>
void ubaci(int niz[],int n){
	int i=0;
	for(i=n-1; i>=0; i--){
		niz[i*2]=niz[i];
	}
	for(i=0;i<n*2;i++){
		int suma=0,a=niz[i];
		if(a<0)a*=-1;
		while(a>0){
			suma+=a%10;
			a/=10;
		}
		niz[i+1]=suma;
		i++;
	}
}
