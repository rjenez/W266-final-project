#include <stdio.h>

int suma_cifara(int broj)
    {
        int suma=0;
        while(broj!=0){
            suma+=broj%10;
            broj/=10;
        }
        return abs(suma);
    }
void ubaci(int niz[], int br_el)
    {
        int i,j;
        for(i=0; i<br_el; i++){


            for(j=br_el; j>i+1; j--){
                niz[j]=niz[j-1];
            }
            niz[i+1]=suma_cifara(niz[i]);
            i++;
            br_el++;
        }
    }

    for(i=0; i<br_el; i++){

    }

}
