#include <stdio.h>

int Fibonacci(int p){
	int FibonacciNiz[46]={1,1},a=1,b=1,c,i=0,rezultat=0;


	for(i=2;i<46;i++){
		c=a+b;
		a=b;
		b=c;
		FibonacciNiz[i]=c;
	}
	for(i=0;i<46;i++){
		if(p==FibonacciNiz[i]){
			rezultat=1;
			break;
		}
		else 	
			rezultat=0;
	}
		return rezultat;
}
