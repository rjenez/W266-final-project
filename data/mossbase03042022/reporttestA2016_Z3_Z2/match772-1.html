int suma (int x){
	int s=0,l;
	l=abs(x);
	while (l!=0){
		s+=(l%10);
		l=l/10;
	}
	return s;
}

int fibonacci (int x){
	int g=0, a, b, c;
		if (x==1){
			g = 1;
		} else {
			a=0;
			b=1;
			c=a+b;
				while (c<x){
					a=b;
					b=c;
					c=a+b;
				}
			if (c==x){
				g = 1;
			} else {
				g = 0;
			}
		}
	return g;
}

}

int izbaci (int *p, int A){
	int i,j,n=A;
	


		for (i=0; i<n; i++){
				if (fibonacci(*(p+i))==1){
					for (j=i; j<n-1; j++){
						*(p+j)=*(p+j+1);
					}
					n--;
					i--;
				}
		}
	return n;
	
}
