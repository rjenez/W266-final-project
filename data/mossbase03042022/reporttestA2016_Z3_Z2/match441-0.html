#include <stdio.h>

int suma(int n){
	int s=0,c;
	while(n!=0){
		c=n%10;
		s+=c;
		n/=10;
	}
	return s;
}


void ubaci(int*niz,int velicina){
 	int i,j;
 	for(i=0;i<velicina;i++){
 		velicina++;
 		for(j=velicina;j>i;j--){
 			niz[j]=niz[j-1];
 		}
 		niz[i+i]=suma(niz[i]);
 		i++;
 	}
 }
 
 void fibonacci(int*niz,int velicina){
 	int i;

 	niz[1]=1;
 	for(i=2;i<velicina;i++){
 		niz[i]=niz[i-1]+niz[i-2];
 	}
 }
