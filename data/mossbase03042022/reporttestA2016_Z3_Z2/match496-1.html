void ubaci (int * nizf,int size1)
{

	{
		
		if (i%2==1) /*neparni*/
		{
			a=nizf[i-1];
			while (a!=0)
			{
				cf=a%10;
				suma+=cf;
				a/=10;
			}
			
			if (suma>=0)
			{
				suma=suma;
			}

				suma=-suma;
			}

	}
}
