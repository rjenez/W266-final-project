#include <stdio.h>

int sumacifara(int n)
{
	/*funkcija koja racuna sumu cifara svakog elementa niza*/
	int s=0;

	if(n<0) n=-n;
	while(n>0)
	{
		s=s+n%10;
		n=n/10;
	}
	return s;
}
