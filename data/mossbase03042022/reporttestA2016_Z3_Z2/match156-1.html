/* Funkcija SUMA CIFARA */



int suma_cifara(int a){
	int x=0, suma=0;
	do{
		x=a%10;
		a=a/10;
		suma=suma+x;
	} while(a!=0);
	if(suma<0){
		suma=suma*(-1);
	}
	return suma;
}
