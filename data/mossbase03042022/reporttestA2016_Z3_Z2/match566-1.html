int fib (int x) {
    int g=x,a=0,b=1,c,znak=1;
    while(2) {
        c=a+b;
        if(g==c) break;
        else if (g<c) {
            znak=0;
            break;
        }
        else {
            a=b;
            b=c;
        }
    }
    return znak;
}
