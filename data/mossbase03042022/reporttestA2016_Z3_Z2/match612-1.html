#include <stdio.h>

void ubaci(int *niz, int velicina)
{
	int i,j;
	for(i=0;i<velicina;i+=2)
	{
		int zbir_cifara=0;
		int pomocna=niz[i];
		if(pomocna<=0) pomocna*=-1;
		while(pomocna>0)
		{
			zbir_cifara += pomocna%10;
			pomocna/=10;
		}
		for(j=velicina;j>i;j--)
			niz[j]=niz[j-1];
		velicina++;
		niz[i+1]=zbir_cifara;
	}
}

int daLiJeFib(int broj)
{
	int tren=0;
	int p = 1;
	int pp = 1;
	if(broj==0) return 0;
	if(broj==1) return 1;


	while(tren<broj)
	{
	tren = p + pp;
	pp = p;
	p = tren;
	}
	if(tren==broj) return 1;
	return 0;
}



int izbaci(int *niz, int velicina)
{
