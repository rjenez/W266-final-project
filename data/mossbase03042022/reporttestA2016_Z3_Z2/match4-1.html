void ubaci(int *niz, int vel){

		else negative=0;

		do{
			y=*(niz+2*i)%10;
			suma+=y;
			*(niz+2*i)/=10;
		}while(*(niz+2*i)!=0);
		
		*(niz+2*i)=a;/*vracamo staru vr*/
		p++;/*povecavamo vr velicine niza za jedno mjesto*/

		*(niz+2*i+1)=suma;/*u mjesto t ubacujemo sumu cifara prethodnog clana niza*/
	}
}
