bool JesuLiJednake(double x, double y, double eps=1e-10)
{
    if((x<0 && y>0) || (x>0 && y<0)) return false;
    return std::fabs(x-y) <= eps*(std::fabs(x)+ std::fabs(y));
}

bool DaLiSuIdenticne(const Kugla &k1, const Kugla &k2)
{

    return false;
}
