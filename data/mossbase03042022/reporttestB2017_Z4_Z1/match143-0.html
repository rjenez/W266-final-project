bool Jednakost(double x, double y)
{
    double eps=1e-10;
    if ((x<0 && y>0) || (x>0 && y<0))
        return false;
    if (!(std::fabs(x-y)<=eps*(std::fabs(x) + std::fabs(y))))
        return false;
    return true;    
}
